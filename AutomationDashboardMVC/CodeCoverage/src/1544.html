using Aurigo.Common;

{
    public class WfUIIcons
    {
        private static string _Default = "~\\Images\\toolbar\\IcnLedGreen.gif";
        private static string _EndWorkflow = "~\\Images\\toolbar\\IcnLedRed.gif";
        private static string _StartWorkflow = "~\\Images\\toolbar\\IcnLedGreen.gif";
        private static string _State = "~\\Images\\toolbar\\IcnLedYellow.gif";
        private static string _Button = "~\\Images\\toolbar\\Icn_WFAction.png";
        private static readonly string _Warning = _Default;
        private static readonly string _ConditionWarning = _Default;
        private static string _IfBlock = "~\\Images\\toolbar\\Icn_WFIfBlock.png";
        private static string _IfCondition = "~\\Images\\toolbar\\Icn_WFIfCondition.png";
        private static string _SetState = "~\\Images\\toolbar\\Icn_WFSetStage.png";
        private static string _SetFormState = "~\\Images\\toolbar\\Icn_WFSetFormStatus.png";
        private static string _Activity = "~\\Images\\toolbar\\Icn_WfActivity.png";
        private static string _DisplayMessage = "~\\Images\\toolbar\\Icn_WfMessage.png";
        private static string _Publish = "~\\Images\toolbar\\IcnComplete.gif";
        private static string _UnPublish = "~\\Images\toolbar\\IcnCompleteUNP.png";
        private static string _LinkedForm = "~\\Images\\toolbar\\Icn_WfMessage.png";

        public static string UnPublish(string sLang = "en-US")
        {
            return _UnPublish;
        }

        public static string Publish(string sLang = "en-US")
        {
            return _Publish;
        }

        public static string StartWorkflow(string sLang = "en-US")
        {
            return _StartWorkflow;
        }

        public static string EndWorkflow(string sLang = "en-US")
        {
            return _EndWorkflow;
        }

        public static string State(string sLang = "en-US")
        {
            return _State;
        }

        public static string Button(string sLang = "en-US")
        {
            return _Button;
        }

        public static string Warning(string sLang = "en-US")
        {
            return _Warning;
        }

        public static string ConditionWarning(string sLang = "en-US")
        {
            return _ConditionWarning;
        }

        public static string IfBlock(string sLang = "en-US")
        {
            return _IfBlock;
        }

        public static string IfCondition(string sLang = "en-US")
        {
            return _IfCondition;
        }

        public static string SetState(string sLang = "en-US")
        {
            return _SetState;
        }

        public static string SetFormState(string sLang = "en-US")
        {
            return _SetFormState;
        }

        public static string Activity(string sLang = "en-US")
        {
            return _Activity;
        }

        public static string DisplayMessage(string sLang = "en-US")
        {
            return _DisplayMessage;
        }

        public static string LinkedForm(string sLang = "en-US")
        {
            return _LinkedForm;
        }

        public static string Default(string sLang = "en-US")
        {
            return _Default;
        }

        public static string GetIconFor(IActivityInstance ai, string sLang = "en-US")
        {
            if (null == ai) return _Default;

            string sIcon = _Default;
            string sType = ai.GetType().FullName;

            switch (sType)
            {
                case "Aurigo.Workflow.ExternalCallInstance":
                    sIcon = Activity(sLang);
                    break;
                case "Aurigo.Workflow.ConditionsBlock":
                    sIcon = IfBlock(sLang);
                    break;
                case "Aurigo.Workflow.IfCondition":
                    sIcon = IfCondition(sLang);
                    break;
                case "Aurigo.Workflow.StateInfo":
                    var si = (StateInfo)ai;
                    sIcon = si.IsEndState ? EndWorkflow(sLang) : si.IsStartState ? StartWorkflow(sLang) : State(sLang);
                    break;
                case "Aurigo.Workflow.SetStateActivityEx":
                    sIcon = SetState(sLang);
                    break;
                case "Aurigo.Workflow.SetFormStatus":
                    sIcon = SetFormState(sLang);
                    break;
                case "Aurigo.Workflow.SendValidationMessage":
                    sIcon = DisplayMessage(sLang);
                    break;
                case "Aurigo.Workflow.FormAction":
                    var fa = (FormAction)ai;
                    sIcon = Button(sLang);
                    break;
                case "Aurigo.Workflow.LinkedForm":
                    sIcon = LinkedForm(sLang);
                    break;
                case default(string):
                    sIcon = _Default;
                    break;
            }
            return sIcon;
        }
    }
}

using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using Aurigo.Brix.Construction.BudgetManager.BusinessLayer.BL;
using Aurigo.Brix.Platform.BusinessLayer.DataResouces;
using Aurigo.Brix.Platform.CoreUtilities.Utility;
using Aurigo.Common;

namespace Aurigo.Brix.Construction.PlanningManagement.WFActivities.Project_WF
{
    [CustomResourceType("Forms", "XPROJCT")]
    public class WFAValidateProjectOnPublish: FormsCustomResource
    {
        public WFAValidateProjectOnPublish()
        {
            _Namespace = "Forms";
            _Path = "Forms.PLANING.WFactivities.ProjectWF.WFAValidateOnPublish";
            _Name = "Validate on Publish";
            _Desc = "Validate on Publish";
            // _InParameters = new[] { "Forecast Complete Action ID,System.String", "Forecast Submit Action ID,System.String", "Forecast Approve Action ID,System.String", "Budget Estimate Un-Publish Action ID,System.String" };
            _OutParameters = new[] { "IsSuccessful,System.Boolean", "Message,System.String" };
        }
        public override void ExecuteDerived()
        {
            try
            {
                int projectId = InstanceId.ToInt32_2();
                
                int releasedBudgetEstimateID = PlanningManagementManager.Instance.GetReleasedToPlanningBudgetEstimate(projectId);

                if (releasedBudgetEstimateID == 0)
                    throw new Exception("Unable to identify published budget estimate, cannot publish the project");

                SetOutParam("IsSuccessful", true, "System.Boolean");
                SetOutParam("Message", string.Empty, "System.String");
            }
            catch (Exception ex)
            {
                SetOutParam("IsSuccessful", false, "System.Boolean");
                SetOutParam("Message", ex.Message, "System.String");
                throw;
            }
        }
    }
}

namespace Aurigo.Brix.Construction.BudgetManager.BusinessLayer.ConcreateModels
{
    public class InflationRuleFormManagerModel : XMLFormManagerModel
    {
        private WebDateChooser EffectiveFromDate;

        public override void OnPageLoad(BrixFormModel model, XmlFormArgs e)
        {
            //The "effective from date" can range within only as many years from current year as defined in "Program Duration (years)" in Program Cycle under Planning Configuration.
            base.OnPageLoad(model, e);
            EffectiveFromDate = (model.form.GetControl("EffectiveFromDate").ControlReference[0] as Panel).Controls[0] as WebDateChooser;
            Tuple<DateTime, DateTime> planningProgramCycleStartAndEndDates = new BudgetEstimateItemModel(). GetPlanningProgramCycleStartAndEndDates();
            var startDate = planningProgramCycleStartAndEndDates.Item1;
            var endDate = planningProgramCycleStartAndEndDates.Item2;
            EffectiveFromDate.MinDate = startDate;
            EffectiveFromDate.MaxDate = endDate;
        }
    }
}

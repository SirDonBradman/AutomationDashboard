using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using System.Web.UI;

namespace Aurigo.Brix.Platform.BusinessLayer.XmlForm_Framework.XMLControls
{
    public class XMLPanel : System.Web.UI.WebControls.Panel
    {
        public List<IControl> ControlReference = new List<IControl>();

        protected override void OnInit(EventArgs e)
        {
            base.OnInit(e);
        }
        
        protected override void AddedControl(Control control, int index)
        {
            if (!(control is System.Web.UI.WebControls.WebControl) || !(control is IControl))
            {
                if (Controls.Contains(control))
                    Controls.Remove(control);

                throw new Exception("Control should be of type WebControl and must implement IControl inerface");
            }

            base.AddedControl(control, index);

            ControlReference.Add(control as IControl);
        }
    }
}

        public override string ReturnURL
        {
            get
            {
                string parentModule = string.Empty;
           
                if (!string.IsNullOrEmpty(Request["ParentModuleID"]))
                    return String.Format(@"/Common/BrixListPage.aspx?context=QTYPLAN&ContractID={0}&ParentID={1}&PID={2}&ParentModuleId=BDGTEST", Request["ContractID"], Request["ParentID"], Request["PID"]);
                else if (Request["ModelType"] != null && Request["ModelType"] == "Gallery")
                    return string.Format(@"/Common/BrixReportGallery.aspx?Context=BDGMGMT&PID={0}&ParentID={1}&nt=1", Request["PID"], Request["ParentID"]);
                else
                    return string.Format(@"/Common/BrixListPage.aspx?context=QTYPLAN&ContractID={0}&ParentID={1}&PID={2}", Request["ContractID"], Request["ParentID"], Request["PID"]);

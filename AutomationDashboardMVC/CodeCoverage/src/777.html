using System;
using System.Web.UI;
using System.Web.UI.WebControls;

using Aurigo.AMP3.Common;
using Aurigo.AMP3.DataAccess.Core;
using Aurigo.AMP3.ModuleManagementBL;

namespace Aurigo.AMP3.Core.UserControls
{
    public partial class FileUploadControl : UserControl
    {
        public FileUpload FileUpload { get { return fileUploadCtrl; } }

        protected void Page_Load(object sender, EventArgs e)
        {
            RegisterDynamicJSFunction(fileUploadCtrl.ClientID, AMP3ApplicationSettings.Instance.BlockedFileTypes);
            
        }

        private void RegisterDynamicJSFunction(string fileControlClientID, string filetypes)
        {
            this.Page.ClientScript.RegisterClientScriptBlock(this.GetType(), "FileTypeFilter",
            "function OnUpload() { "
            + "var validFiles = \"" + filetypes + "\".split(\",\"); "
            + "var obj = document.getElementById(\"" + fileControlClientID + "\"); "
            + "var source = obj.value; "
            + "var ext = source.substring(source.lastIndexOf(\".\") + 1, source.length).toLowerCase(); "
            + "var found = false; "
            + "for (var i = 0; i < validFiles.length; i++) { "
            + "     if (validFiles[i].toLowerCase() == ext) "
            + "         { found = true; break; } } "
            + "if (found) { "
            + "ShowError(\"The file type '.\" + ext + \"' is not supported.\"); "
            + "document.getElementById(\"" + fileControlClientID + "\").onchange = \"\"; "
            + "document.getElementById(\"" + fileControlClientID + "\").value = ''; "
            + "document.getElementById(\"" + fileControlClientID + "\").onchange = OnUpload; "
            + "}  return true; }", true);
        }
    }
}

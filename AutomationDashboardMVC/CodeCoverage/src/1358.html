using Aurigo.AMP3.Common;
using Aurigo.Brix.Core.WebUiHelper;
using ExpressionDesigner;
using System;
using System.Web;

namespace Aurigo.Brix.WorkflowUI
{
    public partial class ExprPickerDialog : BrixPageBase
    {
        protected override void OnInit(EventArgs e)
        {
            PermissionsToCheck.Add(Constants.MODFEAT_VISIBILITY);

            PageTitle = "Select Resource";
            //this.Page.ClientScript.RegisterClientScriptInclude("CommonScriptFile",
            //    Page.ResolveClientUrl("~/Scripts/Workflow_20160721_1717.js"));

            //var p = (PickerParams)Session[Request.QueryString["AjaxContext"]];
            string key = Session.SessionID + "_" + Request.QueryString["AjaxContext"];
            var p = (PickerParams)HttpContext.Current.Cache[key];

            if (null == p) return;
            p.IsCancelled = true;
            ResPicker.InitParams(p);
            ResPicker.Initialize();
            base.OnInit(e);
        }
    }
}

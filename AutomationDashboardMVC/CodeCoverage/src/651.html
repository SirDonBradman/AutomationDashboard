using System;
using System.Reflection;
using System.Resources;
using Aurigo.Brix.Platform.CoreUtilities.Utility;

namespace Aurigo.AMP3.Resources.MessageResources
{
    public class MessageResourceManager
    {
        public const string AMP3_CONNECTIONSTRING = "AMP3ConnectionString";
        public const string AMP3_Default_Resource_String = "Error Occurred";

        private static readonly ResourceManager WarningResourceManager =
            new ResourceManager("Aurigo.Brix.Platform.BusinessLayer.Resource.Messages.WarningResources",
                Assembly.GetExecutingAssembly());

        private static readonly ResourceManager ErrorResourceManager =
            new ResourceManager("Aurigo.Brix.Platform.BusinessLayer.Resource.Messages.ErrorResources",
                Assembly.GetExecutingAssembly());

        private static readonly ResourceManager InfoResourceManager =
            new ResourceManager("Aurigo.Brix.Platform.BusinessLayer.Resource.Messages.InfoResources",
                Assembly.GetExecutingAssembly());

        public static string GetString(string key, Enumerations.MessageType MessageType)
        {
            string Ret;
            switch (MessageType)
            {
                case Enumerations.MessageType.ErrorMessage:
                    Ret = ErrorResourceManager.GetString(key);
                    if (Ret == null)
                    {
                        return AMP3_Default_Resource_String;
                        //Error Occured
                    }
                    else
                    {
                        return Ret;
                    }

                case Enumerations.MessageType.WarningMessage:
                    Ret = WarningResourceManager.GetString(key);
                    if (Ret == null)
                    {
                        return AMP3_Default_Resource_String;
                    }
                    else
                    {
                        return Ret;
                    }

                case Enumerations.MessageType.InfoMessage:
                    Ret = InfoResourceManager.GetString(key);
                    if (Ret == null)
                    {
                        return AMP3_Default_Resource_String;
                    }
                    else
                    {
                        return Ret;
                    }
                default:
                    return AMP3_Default_Resource_String;
            }
        }

        public static string GetString(string key, Enumerations.MessageType MessageType, params Object[] Parameters)
        {
            string Ret;
            switch (MessageType)
            {
                case Enumerations.MessageType.ErrorMessage:
                    Ret = ErrorResourceManager.GetString(key);
                    if (Ret == null)
                    {
                        return AMP3_Default_Resource_String;
                        //Error Occured
                    }
                    else
                    {
                        return (Ret.Format2(Parameters));
                    }

                case Enumerations.MessageType.WarningMessage:
                    Ret = WarningResourceManager.GetString(key);
                    if (Ret == null)
                    {
                        return AMP3_Default_Resource_String;
                    }
                    else
                    {
                        return (Ret.Format2(Parameters));
                    }

                case Enumerations.MessageType.InfoMessage:
                    Ret = InfoResourceManager.GetString(key);
                    if (Ret == null)
                    {
                        return AMP3_Default_Resource_String;
                    }
                    else
                    {
                        return (Ret.Format2(Parameters));
                    }
                default:
                    return AMP3_Default_Resource_String;
            }
        }
    }
}

using Aurigo.Brix.Platform.BusinessLayer.AbstractModels;
using Aurigo.Brix.Platform.BusinessLayer.XMLForm;

namespace Aurigo.Brix.Platform.UI.Modules.FRMBLDR
{
    public class AddSpacer : AddProcessor
    {
        public AddSpacer(BrixFormModel model) : base(model)
        {
        }

        public override string ControlName => "spacer";

        public override string Add(Command cmd, out ControlContainer control)
        {
            string err = string.Empty;

            control = null;

            ControlContainer dest = FormBuilderSupport.GetDestination(_model, cmd.destinationControl);
            if (dest != null)
            {
                if (dest.GetType() == typeof(Form))
                {
                    var s = new Spacer();
                    int i = 0;
                    ControlContainer sc;
                    do
                    {
                        s.Name = ("Spacer" + i++);
                        sc = _model.form.GetContainer(s.Name);
                    } while (sc != null);
                    s.Height = "20px";
                    dest.Containers.Add(s);

                    control = s;
                }
                else
                    err = "Spackers must be a child of a Form";
            }
            else
                err = "Cant find destination control";

            return err;
        }
    }
}

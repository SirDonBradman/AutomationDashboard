using Aurigo.Brix.Platform.BusinessLayer.XMLForm;
using System;
using System.Collections.Generic;
using System.ComponentModel;
using System.Linq;
using System.Web;
using System.Xml.Serialization;

namespace Aurigo.Brix.Platform.BusinessLayer.XMLForm
{
    /// <summary>
    /// Holds the QRCodeControl XML container
    /// </summary>
    public class QRCodeControl
    {
        [XmlAttribute("QRColumns")]
        public string QRColumns;

        [XmlAttribute("QRLabels"),DefaultValue("")]
        public string QRLabels=string.Empty;

        [XmlAttribute("ShowText"), DefaultValue(false)]
        public bool ShowText=false;

        [XmlAttribute("Name")]
        public string Name;

        [XmlAttribute("Height"), DefaultValue("200px")]
        public string Height= "200px";

        [XmlAttribute("Width"), DefaultValue("200px")]
        public string Width="200px";

        [XmlAttribute("LosslessColumns"), DefaultValue("")]
        public string LosslessColumns=string.Empty;       
    }    
}

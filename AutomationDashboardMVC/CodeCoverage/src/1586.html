using System.Data;
using Aurigo.AMP3.Common;
using Aurigo.AMP3.Core.AbstractModels;
using Aurigo.Brix.Construction.ContractManager.CostRevisionUI.BL;

namespace Aurigo.Brix.Construction.ContractManager.CostRevisionUI
{
    [Context(Name = "COSTVER")]
    public class CostRevisionResourceSummaryModel : ResourceSummaryModel
    {
        public override bool IsReadOnly
        {
            get { return !CostRevisionManager.Instance.IsRevisionEditable(InstanceId); }
        }

        public override DataSet GetInstanceDetails(int InstanceID)
        {
            DisplayInventoryDimensions = true;
            return CostRevisionManager.Instance.GetRevisionDetails(InstanceID, 0);
        }
    }
}

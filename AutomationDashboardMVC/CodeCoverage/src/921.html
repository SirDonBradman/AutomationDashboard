using Aurigo.Brix.Platform.BusinessLayer.ChartBuilder;
using Aurigo.Brix.Platform.BusinessLayer.XMLForm;
using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using System.Xml.Serialization;

namespace Aurigo.Brix.Platform.BusinessLayer.XmlForm_Framework.FormComponents
{
    public class TreeChart : Chart
    {
        [XmlAttribute("IDColumnName")]
        public string IDColumnName { get; set; }

        [XmlAttribute("TextColumnName")]
        public string TextColumnName { get; set; }

        [XmlAttribute("ParentIDColumnName")]
        public string ParentIDColumnName { get; set; }


        [XmlAttribute("BeforeRenderScript")]
        public string BeforeRenderScript { get; set; }

        [XmlAttribute("TitleColumnName")]
        public string TitleColumnName { get; set; }

        public override void Render(object parent, RenderFormat format)
        {
            if (form.Renderer.ContainerRenderers[(int)ContainerTypes.TreeChart] != null)
                form.Renderer.ContainerRenderers[(int)ContainerTypes.TreeChart](this, parent);
        }

    }
}

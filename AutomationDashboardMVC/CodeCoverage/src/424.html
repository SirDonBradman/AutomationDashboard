using Aurigo.AMP3.Common;
using System.Data;
using System.Collections.Generic;

namespace Aurigo.AMP3.EstimatorUI
{
    [ListModelContext(Name = _CONTEXT, Table = "CORITEMGlobalSubItemGroup")]
    public class GlobalSubItemGroupListModel : Core.AbstractModels.GlobalSubItemGroupListModel
    {
        private const string _CONTEXT = "GLOBALGROUPESTMATE";

        public override string CONTEXT
        {
            get { return _CONTEXT; }
        }

        public override string ModuleId
        {
            get { return Constants.MODID_ESTMATE; }
        }

        public override DataSet GetList(int pageSize, string sortOrder, string filter, ref int CurrentPage,
                                      out int count)
        {
            sortOrder = DefaultSortColumn + " DESC";
            return base.GetList(pageSize, sortOrder, filter, ref CurrentPage, out count);
        }

    }
}

using System;
using System.Collections.Generic;
using System.Data;
using Aurigo.AMP3.DataAccess.Core;
using Aurigo.Brix.Platform.BusinessLayer.DataAccessHelper;
using Aurigo.Brix.Platform.BusinessLayer.DTO;
using Aurigo.Common.Utility;

namespace Aurigo.Brix.Platform.BusinessLayer.BL
{
    [Serializable]
    public class ChangeLogComponent : SingletonManagerBase<ChangeLogComponent>
    {
        private ChangeLogComponent()
        {
        }

        public DataTable GetRecordHistory(ChangeLogDTO dto)
        {
            DataSet ds =
                ComponentHelper.Instance.ExecuteDataSetWithDTO<ChangeLogDTO>(
                    StoredProcedure.usp_CORITEMGetRecordHistory, null, dto);
            if (ds.Tables.Count > 0)
            {
                ds.Tables[0].TableName = dto.TableName;
                return ds.Tables[0];
            }
            else
                return null;
        }

        public DataSet GetAllRecordsDetails(ChangeLogDTO dto)
        {
            return ComponentHelper.Instance.ExecuteDataSet(StoredProcedure.usp_CORITEMGetRecordsDetails, null,
                dto.TableName, dto.WhereCond);
        }
    }
}

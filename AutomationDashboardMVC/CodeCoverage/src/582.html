using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using Aurigo.Brix.Construction.BudgetManager.BusinessLayer.BL;
using Aurigo.Brix.Platform.BusinessLayer.DataResouces;
using Aurigo.Brix.Platform.CoreUtilities.Utility;
using Aurigo.Common;


namespace Aurigo.Brix.Construction.PlanningManagement.WFActivities
{
    [Serializable]
    [CustomResourceType("Forms", "BDGTREV")]
    public class WFAUpdateBudgetRevisionStatusToUnPublished : FormsCustomResource
    {
        public WFAUpdateBudgetRevisionStatusToUnPublished()
        {
            _Namespace = "Forms";
            _Path = "Forms.PLANING.WFActivities.WFAUpdateBudgetREvisionStatusToUnPublished";
            _Name = "Update Status to Un-Published";
            _Desc = "Updates the status to Un-Published";
            _OutParameters = new[] { "IsSuccessful,System.Boolean", "Message,System.String" };
        }
        public override void ExecuteDerived()
        {
            BudgetManagementManager.Instance.UpdateRevisionStatus(InstanceId.ToInt32_2(), "UnPublished");

            SetOutParam("IsSuccessful", true, "System.Boolean");
            SetOutParam("Message", string.Empty, "System.String");
        }
    }
}

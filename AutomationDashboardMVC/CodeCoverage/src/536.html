using Aurigo.AMP3.Common;
using System;
using System.Collections.Generic;
using System.Data;
using System.Linq;
using System.Web;

namespace Aurigo.Brix.Construction.PlanningManagement
{
    public abstract class ForecastModel
    {
        private static readonly Dictionary<string, Type> contextDictionary;
        public delegate void ShowNotificationHandler(string message);

        public event ShowNotificationHandler OnShowSuccess;
        public event ShowNotificationHandler OnShowError;
        public event ShowNotificationHandler OnShowWarning;
        public event ShowNotificationHandler OnShowInfo;
        protected void ShowSuccess(string message)
        {
            if (OnShowSuccess != null)
                OnShowSuccess(message);
        }

        protected void ShowWarning(string message)
        {
            if (OnShowWarning != null)
                OnShowWarning(message);
        }

        protected void ShowError(string message)
        {
            if (OnShowError != null)
                OnShowError(message);
        }

        protected void ShowInfo(string message)
        {
            if (OnShowInfo != null)
                OnShowInfo(message);
        }
        public int InstanceId { get; set; }

        public DataSet ScheduleData { get; set; }

        public int SnapshotID { get; set; }

        static ForecastModel()
        {
            if (contextDictionary == null)
                contextDictionary = ModelHelper.GetModelObjects(typeof(ForecastModel));
        }

       
        public static ForecastModel GetInstance(string context)
        {
            if (!contextDictionary.ContainsKey(context) || contextDictionary[context] == null)
                return null;
            return (ForecastModel)Activator.CreateInstance(contextDictionary[context]);
        }

        public static Dictionary<string, Type> GetModelObjects()
        {
            return contextDictionary;
        }

        public abstract void HandleSave(object sender, EventArgs e);
    }
}

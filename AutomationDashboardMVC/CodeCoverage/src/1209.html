using System;
using System.Collections.Generic;
using System.Runtime.Remoting;
using Aurigo.AMP3.Common;

namespace Aurigo.Brix.Platform.BusinessLayer.AbstractModels
{
    public class ModuleDetails
    {
        public static Dictionary<string, string> contextDictionary;

        static ModuleDetails()
        {
            if (contextDictionary == null)
            {
                contextDictionary = new Dictionary<string, string>();

                //TODO: Add proper class name
                contextDictionary[Constants.MODID_CONTMGT] =
                    "Aurigo.Brix.Construction.ContractManager.BusinessLayer,Aurigo.Brix.Construction.ContractManager.BusinessLayer.ConcreateModels.ContractManagement.ContractManagementEisCompanyDetails";
                contextDictionary[Constants.MODID_ESTMATE] =
                    "Aurigo.Brix.Construction.Estimator.BusinessLayer,Aurigo.Brix.Construction.Estimator.BusinessLayer.ConcreateModels.Estimate.EstimatorEisCompanyDetails";
            }
        }

        public virtual string GetMappedProjectInEis(string additionalInfo, string projectId, string parentInstanceID)
        {
            return additionalInfo;
        }

        public static ModuleDetails GetInstance(string moduleId)
        {
            if (contextDictionary == null || !contextDictionary.ContainsKey(moduleId))
            {
                //Returning an object of type ModuleDetails instead of null.
                return new ModuleDetails();
            }
            string[] moduleInfo = contextDictionary[moduleId].Split(",".ToCharArray());

            if (moduleInfo.Length < 2)
            {
                return null;
            }

            ObjectHandle handle = Activator.CreateInstance(moduleInfo[0], moduleInfo[1]);
            var listModel = (ModuleDetails)handle.Unwrap();

            if (listModel == null)
            {
                listModel = new ModuleDetails();
            }

            return listModel;
        }
    }
}

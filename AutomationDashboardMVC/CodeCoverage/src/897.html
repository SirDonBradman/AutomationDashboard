using Aurigo.AMP3.DataAccess.Core;
using Aurigo.AMP3.Logging;
using Aurigo.Brix.Platform.BusinessLayer.AbstractModels;
using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using Aurigo.Brix.Platform.CoreUtilities.Utility;

namespace Aurigo.Brix.Platform.BusinessLayer.XmlForm_Framework
{
    public class EmbeddedFormHelper
    {
        public static int GetEmbeddedFormInstance(BrixFormModel embeddedFormModel, string parentModuleId, int parentInstanceId)
        {
            //TODO(Nidhi) : need to remove dependency on ParentInstanceID and  ParentModuleName
            string query = String.Format("select " + embeddedFormModel.form.PrimaryKeyName + " from " + embeddedFormModel.form.ActualTableNameInDatabase + " where ParentInstanceID =" + parentInstanceId + " and ParentModule = '" + parentModuleId + "'");
            object formInstance = null;
            try
            {
                formInstance = ComponentHelper.Instance.ExecuteScalar(query);
            }
            catch (Exception ex)
            {
                Logger.Log("RendererHtml", ex);
            }

            return formInstance != null ? formInstance.ToInt32_2() : 0;
        }
    }
}

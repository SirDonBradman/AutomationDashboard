using System;
using System.Web.UI;
using Aurigo.AMP3.Common;

namespace Aurigo.AMP3.UI
{
    public partial class BrowserInfo : Page
    {
        protected void Page_Load(object sender, EventArgs e)
        {
            if (!IsPostBack)
            {
                if (Request.QueryString["nodes"] != null && !string.IsNullOrEmpty(Request.QueryString["nodes"]))
                {
                    Session["TreeNodes"] = Request.QueryString["nodes"];
                }
                if (Request.QueryString["mainMenuValues"] != null)
                {
                    var ud = (UserDetail)Session["UserDetail"];
                    if (ud != null)
                        CoreDatabaseHelper.SaveControlSettings(ud.UID, ud.RID, "CustomMenu",
                            Request.QueryString["mainMenuValues"]);
                }
                Response.End();
            }
        }
    }
}

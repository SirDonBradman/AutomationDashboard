using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using Aurigo.Brix.Platform.BusinessLayer.DataResouces;
using Aurigo.Common;

namespace Aurigo.Brix.Construction.PlanningManagement.WFActivities.Master_Program_WF
{

    [Serializable]
    [CustomResourceType("Forms", "MSTRPLN")]
    public class WFAUpdateMasterProgramStatusToClosed : FormsCustomResource
    {
        public WFAUpdateMasterProgramStatusToClosed()
        {
            _Namespace = "Forms";
            _Path = "Forms.PlanningManagement.WFActivities.Master_Program_WF.WFAUpdateMasterProgramStatusToClosed";
            _Name = "Update status to Closed";
            _Desc = "Update status to Closed";
            //_InParameters = new[] {  };
            _OutParameters = new[] { "IsSuccessful,System.Boolean", "Message,System.String" };
        }

        public override void ExecuteDerived()
        {
            try
            {
                int masterProgramID = int.Parse(InstanceId);

                PlanningManagementManager.Instance.UpdateMasterProgramStatus(masterProgramID, DTO.MasterProgramStatus.Closed);

                SetOutParam("IsSuccessful", true, "System.Boolean");
                SetOutParam("Message", string.Empty, "System.String");
            }
            catch (Exception ex)
            {
                SetOutParam("IsSuccessful", false, "System.Boolean");
                SetOutParam("Message", ex.Message, "System.String");
                throw new Exception(ex.Message);
            }
        }
    }
}

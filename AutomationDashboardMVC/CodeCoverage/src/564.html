namespace Aurigo.Brix.Construction.PlanningManagement
{
    public partial class Dashboard1 : BrixPageBase
    {
        protected Dashboard DBControl;

        protected override void OnPreRender(EventArgs e)
        {
            UIHelper.CreateDashboadTabs(1, PageTabBar, DBControl.Layout.pages, DBControl.CurrentPage, "Dashboard");
            base.OnPreRender(e);
        }

        protected override void OnInit(EventArgs e)
        {
            PermissionsToCheck.Add(Constants.MODFEAT_VISIBILITY);

            DBControl.DefaultDashboardPage = "Dashboard.aspx";
            DBControl.SettingName = "Layout_Planning";
            DBControl.PagePrefix = "Dashboard";
            DBControl.IsMultiplePages = true;
            DBControl.CurrentPage = 2;
            base.OnInit(e);
        }
    }

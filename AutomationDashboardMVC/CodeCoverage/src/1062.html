using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;

namespace Aurigo.Brix.Platform.BusinessLayer.Controller
{
    public class DocumentSyncStatus
    {
        public int PID { get; set; }
        public Dictionary<int, DocumentErrorDetails> DocumentWithErrors { get; set; } = new Dictionary<int, DocumentErrorDetails>();
    }

    public class DocumentErrorDetails
    {
        public int FolderID { get; set; }
        public string ErrorMessage { get; set; }

        public DocumentErrorDetails()
        {

        }
        public DocumentErrorDetails(int folderId, string errorMessage)
        {
            FolderID = folderId;
            ErrorMessage = errorMessage;
        }
    }
}

using System.Collections.Generic;
using System.Data;
using Aurigo.AMP3.Common;
using Aurigo.AMP3.Core.AbstractModels;
using Aurigo.AMP3.Core.UserControls;

namespace Aurigo.Brix.Platform.BusinessLayer
{
    [ListModelContext(Name = "RPTVIEW", Table = "REPORTSAdHocReportingViews")]
    public class AdHocReportingViewsListModel : ListModel
    {
        public override string QueryStringName
        {
            get { return "ID"; }
        }

        public override string ModuleId
        {
            get { return "MODMGMT"; }
        }

        public override List<MenuGroup> MenuGroups
        {
            get
            {
                List<MenuGroup> menuGroups = base.MenuGroups;

                MenuGroup generalGroup = menuGroups.Find(mg => mg.Title == GROUP_GENERAL);
                if (generalGroup == null)
                {
                    generalGroup = new MenuGroup(GROUP_GENERAL);
                    menuGroups.Insert(0, generalGroup);
                }

                //generalGroup.AddMenu(new TextIcon("lnkPermissions", "Permissions", "Icn_permission_16.png"));

                return menuGroups;
            }
        }

        public override DataSet GetList(int pageSize, string sortOrder, string filter, ref int CurrentPage,
            out int count)
        {
            return base.GetList(pageSize, sortOrder, filter, ref CurrentPage, out count);
        }

        public override void SetUIDetails()
        {
            displayNew =
                displayEdit =
                    displayView =
                        displayDelete =
                            displayAddDynamicColumns = false;

            displayApplyFilter = true;

            Header = "AdHoc Reporting Views";

            ModifyColumnProperties("Multi", true, null, null, null, false);
            ModifyColumnProperties("ID", true, null, string.Empty, string.Empty, false);
            ModifyColumnProperties("ModuleID", false, null, string.Empty, string.Empty, false, "Module");
            ModifyColumnProperties("Category", false, null, string.Empty, string.Empty, false, "Category");
            ModifyColumnProperties("ViewName", true, null, string.Empty, string.Empty, false, "ViewName");
            ModifyColumnProperties("ViewAliasName", false, null, string.Empty, string.Empty, false, "View Name");
            ModifyColumnProperties("Description", false, null, string.Empty, string.Empty, false, "Description");
            ModifyColumnProperties("RowNum", true, null, string.Empty, string.Empty, false);
        }

        public override string HandleMenuItem(string eventString)
        {
            if (eventString == "Permissions")
            {
                Response.Redirect("~/Modules/Reports/ReportPermissions.aspx");
            }
            return base.HandleMenuItem(eventString);
        }

        public override void HandleNew()
        {
        }

        public override void HandleEdit()
        {
        }

        public override void HandleView()
        {
        }
    }
}

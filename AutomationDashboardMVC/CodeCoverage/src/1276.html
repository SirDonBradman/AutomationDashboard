using System;
using Aurigo.AMP3.Common;
using Aurigo.AMP3.LibraryBL;
using Aurigo.Brix.Core.WebUiHelper;

namespace Aurigo.AMP3.LibraryUI
{
    public partial class LibraryHomePage : BrixPageBase
    {
        protected override void OnInit(EventArgs e)
        {
            if (CoreModuleComponents.Contains("ShowUnifiedLibraryPermissions"))
                PermissionsToCheck.Add(Constants.MODFEAT_VISIBILITY);
            else if (!LibraryInterface.Instance.HasAccessToAnyLibraryForm(UserDetail.GetCurrentUserDetail().UID))
                OnAccessDeniedRedirect();

            base.OnInit(e);
        }

        protected void Page_Load(object sender, EventArgs e)
        {
        }
    }
}

using System;
using System.Collections.Generic;

namespace Aurigo.AMP3.CONTMOHDTO
{
    [Serializable]
    public class MOHDTO
    {
        public Decimal? Amount;
        public string Comments;
        public int ContractID;
        public string Contractor;
        public int ContractorID;
        public int? CreatedBy;
        public DateTime CreatedDate;
        public bool IsContractPrePayment;
        public bool IsIntegrationMode;
        public bool IsPrimeContractor;
        public DateTime MOHDate;
        public List<MOHDeletedItem> MOHDeletedItems;
        public int? MOHID;
        public List<MOHItem> MOHItems;
        public string MOHType;
        //public Decimal? ContractAdvance;
        //public Decimal? RecoveryPercentage;
        public string MaterialDesc;
        public int MaterialID;
        public string PrepayDesc;
        public Decimal? RemainingAmount;
        public int? Status;
        public string WorkOrderNo;
        public int? WorkOrdID;

        public MOHDTO()
        {
            MOHItems = new List<MOHItem>();
            CreatedDate = DateTime.UtcNow;
        }
    }

    [Serializable]
    public struct MOHItem
    {
        public Decimal? Amount;
        public int? GroupID;
        public int? ItemID;
        public Decimal? RecoveryQty;
    }

    [Serializable]
    public struct MOHDeletedItem
    {
        public int GroupID;
        public int ItemID;
    }

    [Serializable]
    public struct MOHDeduction
    {
        public int? GroupID;
        public int? ItemID;
        public int? MOHID;
        public int? PEID;
        public int? PostingID;
        public Decimal? RecoveredAmount;
        public Decimal? RecoveredQuantity;
    }
}

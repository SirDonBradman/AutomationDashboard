using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Xml.Serialization;

namespace Aurigo.Brix.BackgroundProcessor
{
    public class Job
    {
        [XmlAttribute(AttributeName = "name")]
        public string Name { get; set; }

        [XmlElement(ElementName = "trigger")]
        public Trigger Trigger { get; set; }

        [XmlArray("params"), XmlArrayItem(typeof(CustomKeyValuePair<string, string>), ElementName = "add")]
        public List<CustomKeyValuePair<string, string>> Params { get; set; }
    }

    public class Trigger
    {
        [XmlAttribute(AttributeName = "name")]
        public string Name { get; set; }

        [XmlAttribute(AttributeName = "startAt")]
        public string startTime { get; set; }

        [XmlIgnore]
        public DateTime StartAt
        {
            get
            {
                return DateTime.Parse(startTime);
            }
        }


        [XmlAttribute(AttributeName = "repeatCount")]
        public int RepeatCount { get; set; }

        [XmlAttribute(AttributeName = "interval")]
        public int Interval { get; set; }

        [XmlIgnore]
        public bool IsInfinite
        {
            get
            {
                return (RepeatCount == -1);
            }
        }
        [XmlArray("weekDays"), XmlArrayItem(typeof(DayOfWeek), ElementName = "weekDay")]
        public List<DayOfWeek> daysOfWeek { get; set; }

        public List<DayOfWeek> DaysOfWeek
        {
            get
            {
                if (daysOfWeek != null && daysOfWeek.Count > 0)
                    return daysOfWeek;
                else
                    return new List<DayOfWeek>() {
                        DayOfWeek.Monday,
                        DayOfWeek.Tuesday,
                        DayOfWeek.Wednesday,
                        DayOfWeek.Thursday,
                        DayOfWeek.Friday,
                        DayOfWeek.Saturday,
                        DayOfWeek.Sunday
                    };
            }
        }
    }

    public class CustomKeyValuePair<K, V>
    {
        [XmlAttribute(AttributeName = "key")]
        public K Key
        { get; set; }

        [XmlAttribute(AttributeName = "value")]
        public V Value
        { get; set; }
    }
}

using System;
using System.Web.UI;
using Aurigo.AMP3.Common;
using Aurigo.AMP3.DataAccess.Core;
using Aurigo.AMP3.ModuleManagementBL;

namespace Aurigo.AMP3.Core.UserControls
{
    public partial class Copyright : UserControl
    {
        protected System.Web.UI.HtmlControls.HtmlTableCell tdCompany;

        protected void Page_Load(object sender, EventArgs e)
        {
            if (!Page.IsPostBack)
            {
                lblVersion.Text = ModuleManager.Instance.CORE_VERSION;
                lblCopyright.Text = string.Format("{0} ï¿½ Aurigo Software Technologies, Inc. All rights reserved.",
                    Constants.PRODUCTNAME);
                if (string.IsNullOrEmpty(ConnectionHelper.GetCurrentCompany()))
                    tdCompany.Visible = false;
                else
                    lblCompany.Text = "Organization: " + ConnectionHelper.GetCurrentCompany();
                lblCulture.Text = "Location/Culture: " +
                                  (AMP3ApplicationSettings.Instance.Culture.Equals("en-ZW") ? "India" : "United States");
            }
        }
    }
}

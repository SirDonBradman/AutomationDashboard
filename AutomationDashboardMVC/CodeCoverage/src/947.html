using System;
using System.Collections.Generic;
using System.ComponentModel;
using System.Linq;
using System.Web;
using System.Xml.Serialization;

namespace Aurigo.Brix.Platform.BusinessLayer.XMLForm
{
    // the xml deserializer requres that each XML elmement name have a corresponding class
    // otherwise the code breaks.  So these empty classes are created based on xControl 
    // to make the xml deserializer happy
    public class xColumn : xControl
    {
        //default dbtype should be varchar(4000) - for static grid

        [XmlAttribute("ColumnName"), DefaultValue("")]
        public string ColumnName;

        /// <summary>
        /// Required for the picker column, where we can specify any sql query as where condition
        /// </summary>
        [XmlAttribute("FilterExpression"), DefaultValue("")]
        public string FilterExpression;

        /// <summary>
        /// To make a column as hyperlink in grid, this will hold the url
        /// </summary>
        [XmlAttribute("LinkURL"), DefaultValue("")]
        public string LinkURL { get; set; }

        /// <summary>
        /// Currently this used only in static grid.
        /// We can specify the attributes for the parent td.
        /// Mainly for RowSpan and ColumnSpan controls where we need to align the controls
        /// </summary>
        [XmlAttribute("ParentAttributes"), DefaultValue("")]
        public string ParentAttributes { get; set; }


        public xColumn()
        {
        }
    }
}

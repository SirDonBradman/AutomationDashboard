using Aurigo.Brix.Platform.BusinessLayer.AbstractModels;
using Aurigo.Brix.Platform.BusinessLayer.XMLForm;

namespace Aurigo.Brix.Platform.UI.Modules.FRMBLDR
{
    public class AddMatrixGrid : AddProcessor
    {
        public AddMatrixGrid(BrixFormModel model) : base(model)
        {
        }

        public override string ControlName => "matrixgrid";
        public override string Add(Command cmd, out ControlContainer control)
        {
            control = null;
            string err = string.Empty;

            ControlContainer dest = FormBuilderSupport.GetDestination(_model, cmd.destinationControl);
            if (dest != null)
            {
                if (dest.GetType() == typeof(Form))
                {
                    var mg = new MatrixGrid
                    {
                        Name = GetUniqueName(ContainerTypes.MatrixGrid),
                        Height = "50px"
                    };

                    mg.Caption = mg.Name;
                    dest.Containers.Add(mg);

                    control = mg;
                }
            }
            else
                err = "Could not find destination control";

            return err;
        }
    }
}

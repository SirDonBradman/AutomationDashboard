using System.Collections.Generic;

namespace Aurigo.Brix.Platform.BusinessLayer.XMLForm
{
    public class ForeignKeyReference
    {
        public xControl ReferedBy;
        public string ReferenceTable;
        public string ReferenceColumn;
        public List<ReferenceColumnDetails> RequiredColumns = new List<ReferenceColumnDetails>();

        public ReferenceColumnDetails FindColumn(string columnTextWithAlias)
        {
            return this.RequiredColumns.Find(c => c.OriginalFieldText_WithAlias == columnTextWithAlias);
        }
    }
}

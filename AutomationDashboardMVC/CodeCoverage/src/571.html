using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using Aurigo.Brix.Construction.PlanningManagement;
using Aurigo.Brix.Platform.BusinessLayer.DataResouces;
using Aurigo.Brix.Platform.CoreUtilities.Utility;
using Aurigo.Common;

namespace Aurigo.Brix.Construction.PlanningManagement.WFActivities
{
    [CustomResourceType("Forms", "BDGTREV")]
    public class WFAUpdateProjectCostInPrograms : FormsCustomResource
    {
        public WFAUpdateProjectCostInPrograms()
        {
            _Namespace = "Forms";
            _Path = "Forms.PLANING.WFActivities.WFAUpdateProjectCostInPrograms";
            _Name = "Update Project Costs In Programs";
            _Desc = "Update Project Costs In Programs";
            _OutParameters = new[] { "IsSuccessful,System.Boolean", "Message,System.String" };
        }

        public override void ExecuteDerived()
        {
            try
            {
                PlanningManagementComponent.Instance.UpdateProjectCostInProgram(InstanceId.ToInt32_2());

                SetOutParam("IsSuccessful", true, "System.Boolean");
                SetOutParam("Message", string.Empty, "System.String");
            }
            catch (Exception ex)
            {
                SetOutParam("IsSuccessful", false, "System.Boolean");
                SetOutParam("Message", "Project Costs could not be updated in a program(s) that the project belongs to.", "System.String");
                throw new Exception("Project Costs could not be updated in a program(s) that the project belongs to.");
            }
        }
    }
}

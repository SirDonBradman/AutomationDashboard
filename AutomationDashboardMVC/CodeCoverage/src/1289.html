using System.Xml.Serialization;

namespace Aurigo.AMP3.ProjectUI
{
    [XmlRoot("Dashboard")]
    public class DashboardChartManager
    {
        [XmlElement("Chart")] public Chart[] Charts;
    }

    public class Chart
    {
        [XmlAttribute("ControlID")] public string ControlID = string.Empty;

        [XmlElement("DataSource")] public ChartDataSource DataSource;
        [XmlAttribute("DrillDownLevel")] public uint DrillDownLevel;

        [XmlElement("ExcludedColumns")] public ChartColumns ExcludedColumns;
        [XmlAttribute("IsDrillDown")] public bool IsDrillDown;

        [XmlElement("RowLabelsColumn")] public string RowLabelsColumn;
        [XmlAttribute("SkinID")] public string SkinID = string.Empty;
        [XmlAttribute("Title")] public string Title = string.Empty;
    }

    public class ChartDataSource
    {
        [XmlAttribute("Key")] public string Key = string.Empty;

        [XmlElement("Parameter")] public ChartDataSourceParameter[] Parameters;
        [XmlAttribute("Procedure")] public string Procedure = string.Empty;
    }

    public class ChartDataSourceParameter
    {
        [XmlAttribute("Name")] public string Name;
    }

    public class ChartColumns
    {
        [XmlElement("Column")] public string[] Column;
    }
}

using System;
using System.Collections.Generic;
using System.ComponentModel;
using System.Linq;
using System.Web;
using System.Xml.Serialization;

namespace Aurigo.Brix.Platform.BusinessLayer.XMLForm
{
    public class CheckList : ControlContainer
    {
        [XmlAttribute("Description"), DefaultValue("")]
        public string Description;

        public xControl PKControl;

        public CheckList()
        {
            PKControl = new xControl();
            PKControl.ParentObject = this;
            PKControl.IsRunTimeControl = true;
        }
    }
}

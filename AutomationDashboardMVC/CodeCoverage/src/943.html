using System;
using System.Collections.Generic;
using System.ComponentModel;
using System.Linq;
using System.Web;
using System.Xml.Serialization;

namespace Aurigo.Brix.Platform.BusinessLayer.XMLForm
{
    public class Spacer : ControlContainer
    {
        [XmlAttribute("LineAtBottom"), DefaultValue(false)]
        public bool LineAtBottom;

        /// <summary>
        /// These attributes are used to draw a line at the top or bottom of the spacer.  The only valid values are 
        /// true and false
        /// </summary>
        [XmlAttribute("LineAtTop"), DefaultValue(false)]
        public bool LineAtTop;

        /// <summary>
        /// this attribute if true will insert a page break if appropriate for the
        /// renderer.  For instance on an HTML like display this attribute will probably
        /// be ignored, but for a paper oriented renderer this is a significant attribute.
        /// 
        /// In general support of the PageBreak attribute is completely optional
        /// </summary>
        [XmlAttribute("PageBreak"), DefaultValue("")]
        public string PageBreak;

        public override void Render(object parent, RenderFormat format)
        {
            if (form.Renderer.ContainerRenderers[(int)ContainerTypes.Spacer] != null)
                form.Renderer.ContainerRenderers[(int)ContainerTypes.Spacer](this, parent);
        }
    }
}

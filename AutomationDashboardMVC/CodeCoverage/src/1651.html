using System;
using System.Collections.Generic;

namespace Aurigo.Brix.MainModules.ContractManager.MeetingMinutes.DTO
{
    [Serializable]
    public class MOMDTO
    {

            AttendeesList = new List<Attendee>();
            MeetingNotes = new List<MeetingNotes>();
        }

        public int PID { get; set; }
        public int ContractID { get; set; }
        public int MeetingID { get; set; }
        public string MeetingType { get; set; }
        public string MinutesType { get; set; }
        public string Subject { get; set; }
        public string Description { get; set; }
        public DateTime MeetingDate { get; set; }
        public string MeetingTime { get; set; }
        public string MeetingLocation { get; set; }
        public DateTime NextDate { get; set; }
        public string NextTime { get; set; }
        public string NextLocation { get; set; }
        public string Preparer { get; set; }
        public int AgendaType { get; set; }
        public int CreatedBy { get; set; }
        public DateTime CreatedOn { get; set; }


        public List<Attendee> AttendeesList { get; set; }
        public List<MeetingNotes> MeetingNotes { get; set; }
    }
}

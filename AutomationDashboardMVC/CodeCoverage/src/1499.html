using Aurigo.Brix.Platform.BusinessLayer.AbstractModels;
using Infragistics.WebUI.WebSchedule;
using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using System.Web.UI.WebControls;
using Aurigo.Brix.Platform.CoreUtilities.Utility;
using Infragistics.WebUI.WebDataInput;
namespace Aurigo.Construction.MainModules.ResourceManagement.Models
{
    public class ProjectResourceXMLManager : XMLFormManagerModel
    {
        public override void OnInit(BrixFormModel model, XmlFormArgs e)
        {
            model.form.page.ClientScript.RegisterClientScriptInclude("CreateProject", VirtualPathUtility.ToAbsolute("~/Modules/RESMGMT/Scripts/ProjectResources.js"));
            base.OnInit(model, e);

        }
        public override void OnPageLoad(BrixFormModel model, XmlFormArgs e)
        {
            base.OnPageLoad(model, e);

            WebDateChooser startDate = ((model.form.GetControl("StartDate").ControlReference[0] as Panel).Controls[0] as WebDateChooser);
            WebDateChooser endDate = ((model.form.GetControl("EndDate").ControlReference[0] as Panel).Controls[0] as WebDateChooser);
            WebNumericEdit txtHPD = (model.form.GetControl("HoursPerDay").ControlReference[0] as WebNumericEdit);
            WebNumericEdit totHrs = (model.form.GetControl("TotalHours").ControlReference[0] as WebNumericEdit);
            txtHPD.ClientSideEvents.ValueChange = "StartOrEndDateChanged('" + startDate.ClientID + "','" + endDate.ClientID + "','" + txtHPD.ClientID + "','" + totHrs.ClientID + "')";
            startDate.ClientSideEvents.ValueChanged = "StartOrEndDateChanged('" + startDate.ClientID + "','" + endDate.ClientID + "','" + txtHPD.ClientID + "','" + totHrs.ClientID + "')";
            endDate.ClientSideEvents.ValueChanged = "StartOrEndDateChanged('" + startDate.ClientID + "','" + endDate.ClientID + "','" + txtHPD.ClientID + "','" + totHrs.ClientID + "')";


        }
    }
}

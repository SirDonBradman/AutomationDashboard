using Aurigo.AMP3.Common;
using Aurigo.AMP3.Core.AbstractModels;
using Aurigo.Brix.Construction.ContractManager.CostRevisionUI.BL;

namespace Aurigo.Brix.Construction.ContractManager.CostRevisionUI
{
    [Context(Name = "COSTVER")]
    public class CostRevisionItemTotalModel : ItemTotalModel
    {
        public override string Header
        {
            get { return "Cost Planning Total"; }
        }

        public override bool IsPriceRelatedFieldsHidden
        {
            get { return true; }
        }

        public override bool GetApprovalStatus(int projectEstimateID, string moduleID)
        {
            return !CostRevisionManager.Instance.IsRevisionEditable(projectEstimateID);
        }
    }
}

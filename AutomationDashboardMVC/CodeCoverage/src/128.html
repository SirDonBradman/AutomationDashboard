        private TextBox txtQuanity;

        public void InitializeReport()

            txtContractor = ((TextBox) (Detail.Controls[0]));
            txtPersonnel = ((TextBox) (Detail.Controls[1]));
            txtQuanity = ((TextBox) (Detail.Controls[2]));
        }

}

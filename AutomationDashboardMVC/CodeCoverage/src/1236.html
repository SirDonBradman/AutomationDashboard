using System;
using System.Web;

namespace Aurigo.AMP3.UserManagementUI
{
    internal class Global : HttpApplication
    {
        //counter for online users
        private static int LoggedInUsers;

        public static int OnLineUsers
        {
            get { return LoggedInUsers; }
            set { LoggedInUsers = value; }
        }

        protected void Application_Start(object sender, EventArgs e)
        {
        }

        protected void Application_End(object sender, EventArgs e)
        {
        }

        protected void Application_BeginRequest(object sender, EventArgs e)
        {
            Response.Cache.SetCacheability(HttpCacheability.NoCache);
        }

        protected void Session_Start(object sender, EventArgs e)
        {
            //increment number of online users by one on every session start
            //LoggedInUsers += 1;
        }

        protected void Session_End(object sender, EventArgs e)
        {
            //decrement no of online users for every session abort
            LoggedInUsers -= 1;
        }
    }
}

using Aurigo.AMP3.Common;
using System.Web.UI;
using Telerik.Web.UI;

namespace Aurigo.Brix.Platform.BusinessLayer.XmlForm_Framework.XMLControls
{
    public class XMLDateTimeControl : RadDateTimePicker, IControl
    {
        public bool RenderAsLabel
        {
            get; set;
        }

        protected override void Render(HtmlTextWriter writer)
        {
            if (RenderAsLabel)
            {
                string text = !this.SelectedDate.HasValue ? "" : this.SelectedDate.Value.ToString(AMP3ApplicationSettings.Instance.FORMAT_DATETIME);
                Helper.RenderAsLabel(this, writer, text, Width, Height, attributes: Style.Value);
                base.Style.Add("display", "none");
                base.Render(writer);
            }
            else
                base.Render(writer);
        }
    }
}

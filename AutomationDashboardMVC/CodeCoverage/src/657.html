namespace Aurigo.AMP3.ProjectDTO
{
    /// <summary>
    /// Author: Kalyan Chakradhar R.
    /// Description: Data Tranfer Object for Project Platform.
    /// </summary>
    public class Platform
    {
        public Platform(string ModuleID, string PlatformKey)
        {
            ModuleId = ModuleID;
            this.PlatformKey = PlatformKey;
        }

        public Platform()
        {
        }

        /// <summary>
        /// Gets and Sets ModuleId
        /// </summary>
        public string ModuleId { get; set; }

        /// <summary>
        /// Gets and Sets PlatformKey
        /// </summary>
        public string PlatformKey { get; set; }
    }
}

using Aurigo.Brix.Platform.BusinessLayer.AbstractModels;
using Aurigo.Brix.Platform.BusinessLayer.XMLForm;

namespace Aurigo.Brix.Platform.UI.Modules.FRMBLDR
{
    public class AddInclude : AddProcessor
    {
        public AddInclude(BrixFormModel model) : base(model)
        {
        }

        public override string ControlName => "include";

        public override string Add(Command cmd, out ControlContainer control)
        {
            control = null;
            string err = string.Empty;

            var dest = FormBuilderSupport.GetDestination(_model, cmd.destinationControl);
            if (dest != null)
            {
                if (dest is Form)
                {
                    var include = new Include
                    {
                        Name = GetUniqueName(ContainerTypes.Include),
                        Height = "100px",
                        Width = "200px"
                    };
                    dest.Containers.Add(include);
                    control = include;
                }
                else
                    err = "Can only add Include to a Form";
            }

            return err;
        }
    }
}

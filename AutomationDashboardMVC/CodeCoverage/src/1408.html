using System;
using System.Configuration;
using Aurigo.AMP3.Common;
using Aurigo.AMP3.DataAccess.Core;
using Aurigo.Brix.Core.WebUiHelper;
using Aurigo.Brix.Platform.BusinessLayer.DataAccessHelper;

namespace Aurigo.Brix.Platform.UI.Modules.ENTPRSE
{
    public partial class LinkViewer : BrixPageBase
    {
        protected void Page_Load(object sender, EventArgs e)
        {
        }

        protected override void OnInit(EventArgs e)
        {
            base.OnInit(e);
        }

        private string ProjectIDLIst
        {
            get
            {
                string projIDList = ConfigurationManager.AppSettings["SampleProjectIDs"];
                if (!string.IsNullOrEmpty(projIDList))
                    return projIDList;
                return string.Empty;
            }
        }

        protected void lbRemove_Click(object sender, EventArgs e)
        {
            try
            {
                if (!string.IsNullOrEmpty(ProjectIDLIst))
                    ComponentHelper.Instance.ExecuteNonQueryWithVariableParametersOn(
                        ConnectionHelper.GetConnectionString(), StoredProcedure.usp_RemoveSampleData, null,
                        ProjectIDLIst);
            }
            catch (Exception)
            {
            }
        }

        protected void lbRestore_Click(object sender, EventArgs e)
        {
            try
            {
                if (!string.IsNullOrEmpty(ProjectIDLIst))
                    ComponentHelper.Instance.ExecuteNonQueryWithVariableParametersOn(
                        ConnectionHelper.GetConnectionString(), StoredProcedure.usp_RestoreSampleData, null,
                        ProjectIDLIst);
            }
            catch (Exception)
            {
            }
        }

        protected void chbDefault_CheckedChanged(object sender, EventArgs e)
        {
            UIHelper.ShowHideDashbodTab("Layout_Enterprise", 1, !chbDefault.Checked);
        }
    }
}

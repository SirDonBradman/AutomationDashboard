using Aurigo.Brix.Platform.BusinessLayer.AbstractModels;
using Aurigo.Brix.Platform.BusinessLayer.XMLForm;

namespace Aurigo.Brix.Platform.UI.Modules.FRMBLDR
{
    public class AddCheckList : AddProcessor
    {
        public AddCheckList(BrixFormModel model) : base(model)
        {
        }

        public override string ControlName => "check list";

        public override string Add(Command cmd, out ControlContainer control)
        {
            string err = string.Empty;
            control = null;
            var dest = FormBuilderSupport.GetDestination(_model, cmd.destinationControl);

            if (dest is Group)
            {
                var clk = new CheckList
                {
                    Name = GetUniqueName(ContainerTypes.CheckList),
                    Description = "Checklist Description"
                };

                dest.Containers.Add(clk);
                control = clk;
            }
            else
                err = "checklist must be a child of a check list group";

            return err;
        }
    }
}

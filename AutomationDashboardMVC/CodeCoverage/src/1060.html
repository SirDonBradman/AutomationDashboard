using Aurigo.Brix.Platform.BusinessLayer.DTO;
using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using Newtonsoft.Json;
using System.Xml.Serialization;
using System.Web.Script.Serialization;

namespace Aurigo.Brix.Platform.BusinessLayer.Controller
{
    public class MobileModuleEstimateDetails
    {
        public MobileMenuDTO_InputParams Menu { get; set; }

        public string ModuleSyncDetails { get; set; }

        [XmlIgnore]
        [JsonIgnore]
        public Dictionary<string, string> ModuleSyncDetailsDictionary { get; private set; }

        //Src: https://stackoverflow.com/questions/1207731/how-can-i-deserialize-json-to-a-simple-dictionarystring-string-in-asp-net 
        //alternative Src: https://stackoverflow.com/questions/4710729/post-json-dictionary
        public void Deserialize_ModuleSyncDetailsDictionary()
        {
            JavaScriptSerializer js = new JavaScriptSerializer();
            try
            {
                if (string.IsNullOrEmpty(this.ModuleSyncDetails) || this.ModuleSyncDetails == "{}")
                    this.ModuleSyncDetailsDictionary = new Dictionary<string, string>();
                else
                    this.ModuleSyncDetailsDictionary = js.Deserialize<Dictionary<string, string>>(this.ModuleSyncDetails);
            }
            catch (Exception ex)
            {
                //DO nothing
            }
        }
    }
}

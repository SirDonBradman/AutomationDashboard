using System;
using System.Collections.Generic;
using System.ComponentModel;
using System.Linq;
using System.Web;
using System.Xml.Serialization;

namespace Aurigo.Brix.Platform.BusinessLayer.XMLForm
{
    public class xTabs : ControlContainer
    {
        public override void Render(object parent, RenderFormat format)
        {
            if (form.Renderer.ContainerRenderers[(int)ContainerTypes.Tabs] != null)
                form.Renderer.ContainerRenderers[(int)ContainerTypes.Tabs](this, parent);
        }
    }

    public class xTab : ControlContainer
    {
        [XmlAttribute("IsSelected"), DefaultValue(false)]
        public bool IsSelected = false;

        [XmlAttribute("Caption")]
        public string Caption;

        public override void Render(object parent, RenderFormat format)
        {
            if (form.Renderer.ContainerRenderers[(int)ContainerTypes.Tab] != null)
                form.Renderer.ContainerRenderers[(int)ContainerTypes.Tab](this, parent);
        }
    }
}

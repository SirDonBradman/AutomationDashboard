using System.Data;
using Aurigo.AMP3.Common;
using Aurigo.AMP3.DataAccess.Core;
using Aurigo.Brix.Platform.BusinessLayer.DataAccessHelper;

namespace Aurigo.AMP3.EstimatorUI
{
    [Context(Name = "ESTMATE")]
    public class ResourceSummaryModel : Core.AbstractModels.ResourceSummaryModel
    {
        public override DataSet GetInstanceDetails(int InstanceID)
        {
            IsInventoryDimensionsVisible = true;
            return ComponentHelper.Instance.ExecuteDataSet(StoredProcedure.usp_ESTMATEGetEstimateDetails, null,
                                                           InstanceID);
        }
    }
}

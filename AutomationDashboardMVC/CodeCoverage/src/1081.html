using System;
using System.Data;
using System.Net;
using System.Net.Http;
using System.Web.Http;
using Aurigo.AMP3.DataAccess.Core;
using Aurigo.AMP3.ProjectBL;
using Aurigo.Brix.Platform.BusinessLayer.AbstractModels;
using Aurigo.DataBinding;

namespace Aurigo.Brix.Platform.BusinessLayer.Controller
{
    public class ProjectController : ApiController
    {
        [HttpGet]
        public HttpResponseMessage Get(string jsonVersionInformation)
        {
            DataSet SyncHadlersData = new DataSet();
            SyncHadlersData =
                ComponentHelper.Instance.ExecuteDataSet(
                     "select * from projectprojectmain");

            var response = Request.CreateResponse<DataSet>(HttpStatusCode.Created, SyncHadlersData);
            return response;
        }

        [HttpGet]
        public HttpResponseMessage GetWorkingDays(int projectID, string startDateAsStr, string endDateAsStr)
        {
            int workingDays = 0;

            DateTime startDate = DateTime.Parse(startDateAsStr);
            DateTime endDate = DateTime.Parse(endDateAsStr);

            CalendarCalculationsModel calCalculationModel = CalendarCalculationsModel.GetInstance("PROJECT");
            if (calCalculationModel != null)
            {
                CalendarDetails calDetails = calCalculationModel.GetCalendarDetails(projectID);
                if (calDetails != null)
                {
                    workingDays = (int)calCalculationModel.GetWorkingDaysCount(startDate, endDate, calDetails);
                }
            }

            var response = Request.CreateResponse<int>(HttpStatusCode.Created, workingDays);
            return response;
        }
    }
}

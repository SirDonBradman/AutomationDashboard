namespace Aurigo.Brix.Platform.BusinessLayer.DTO
{
    public enum ProductType
    {
        Limited,
        Professional
    }

    public enum UserLicenseType
    {
        Named,
        Concurrent
    }

    public class CompanyDetails
    {
        private string _CompanyName;
        private string _CompanyCode;
        private ProductType _Edition;
        private int _UserLimit;
        private string _StorageLimit;
        private int _LoggedInUsers;
        private int _ProjectLimit;
        private UserLicenseType _UserLicenseType;

        public string CompanyName
        {
            get { return _CompanyName; }
            set { _CompanyName = value; }
        }

        public string CompanyCode
        {
            get { return _CompanyCode; }
            set { _CompanyCode = value; }
        }

        public ProductType Edition
        {
            get { return _Edition; }
            set { _Edition = value; }
        }

        public int UserLimit
        {
            get { return _UserLimit; }
            set { _UserLimit = value; }
        }

        public string StorageLimit
        {
            get { return _StorageLimit; }
            set { _StorageLimit = value; }
        }

        public int LoggedInUsers
        {
            get { return _LoggedInUsers; }
            set { _LoggedInUsers = value; }
        }

        public int ProjectLimit
        {
            get { return _ProjectLimit; }
            set { _ProjectLimit = value; }
        }

        public UserLicenseType UserLicenseType
        {
            get { return _UserLicenseType; }
            set { _UserLicenseType = value; }
        }
    }
}

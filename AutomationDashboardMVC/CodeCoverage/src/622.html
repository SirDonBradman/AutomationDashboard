using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;

namespace Aurigo.Brix.Construction.PlanningManagement.DTO
{
    public class ProgramCycle
    {
        private Month _startMonth;
        private Month _endMonth;
        private int _programDuration;

        public Month StartMonth
        {
            get
            {
                return _startMonth;
            }
            set
            {
                _startMonth = value;
            }
        }

        public Month EndMonth
        {
            get
            {
                return _endMonth;
            }
            set
            {
                _endMonth = value;
            }
        }

        public int ProgramDuration
        {
            get
            {
                return _programDuration;
            }
            set
            {
                _programDuration = value;
            }
        }
    }
}

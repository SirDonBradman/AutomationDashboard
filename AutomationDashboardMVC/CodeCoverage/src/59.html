using System.Data;
using Aurigo.AMP3.Core.IntegrationConnector;
using Aurigo.Brix.Platform.BusinessLayer.AbstractModels;

namespace Aurigo.Brix.Construction.ContractManager.BusinessLayer.ConcreateModels.ContractManagement
{
    public class ContractManagementEisCompanyDetails : ModuleDetails
    {
        public override string GetMappedProjectInEis(string additionalInfo, string projectId, string parentInstanceID)
        {
            string mappedCompany = additionalInfo;
            DataTable dtContractMap = IntegrationConnectorManager.Instance.GetERPObjectMapInfo(parentInstanceID,
                                                                                               AMP3Object.CONTMGT, null,
                                                                                               AMP3Object.UNKNOWN,
                                                                                               RegisteredEIS.AX);

            if (dtContractMap.Rows.Count > 0)
                mappedCompany = dtContractMap.Rows[0]["ERPCompany"].ToString();

            return mappedCompany;
        }
    }
}

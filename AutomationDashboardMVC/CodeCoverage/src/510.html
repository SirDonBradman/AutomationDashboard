using Aurigo.AMP3.Common;
using Aurigo.AMP3.Core.AbstractModels;
using System.Data;
using System.Collections.Generic;

namespace Aurigo.AMP3.LandMangement.UI
{
    [ListModelContext(Name = "LANDOWNER", Table = "LibraryLandMgmtOwnerTypes")]
    public class OwnerTypeListModel : ListModel
    {
        public override string QueryStringName
        {
            get { return "OwnerTypeID"; }
        }

        public override string ModuleId
        {
            get { return Constants.MODID_LIBRARY; }
        }

        public override void SetUIDetails()
        {
            ModifyColumnProperties("OwnerTypeID", true, null, null, null, false, null);
            ModifyColumnProperties("RowNum", true, null, null, null, false, null);

            Header = "Owner Types";
        }


         public override string NewURL
        {
            get
            {
                return ("BrixEditPage.aspx?context=LANDOWNER&mode=New");


                }


         }
        public override void HandleNew()
        {
            Response.Redirect(NewURL);
        }

        public override DataSet GetList(int pageSize, string sortOrder, string filter, ref int CurrentPage,
                                      out int count)
        {
            sortOrder = DefaultSortColumn + " DESC";
            return base.GetList(pageSize, sortOrder, filter, ref CurrentPage, out count);
        }


        public override void HandleEdit()
        {
            Response.Redirect("BrixEditPage.aspx?context=LANDOWNER&InstanceID=" + GetSelectedIds() + "&mode=Edit");
        }

        public override void HandleView()
        {
            Response.Redirect("BrixEditPage.aspx?context=LANDOWNER&InstanceID=" + GetSelectedIds() + "&mode=View");
        }
    }
}

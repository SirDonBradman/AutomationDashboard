{
    public class XMLNumericControl : WebNumericEdit, IControl
    {
        public bool RenderAsLabel { get; set; }
        protected override void Render(System.Web.UI.HtmlTextWriter writer)
        {
            if (RenderAsLabel)
            {
                //string text = this.Value == null && this.Text == null ? "" : this.Text;
                //Helper.RenderAsLabel(this, writer, text, Width, Height, attributes: Style.Value);
                //base.Style.Add("display", "none");
                //DO NOT add a label as NUMERIC fields are having computed values and needs to be aligned on the right
                //basically this should behave as a numeric control except that it must look like a label
                this.CssClass = (this.CssClass ?? string.Empty) + " NumericControl_ReadOnly";
                this.ReadOnly = true;
                this.HorizontalAlign = HorizontalAlign.Right;

                base.Render(writer);
            }
            else
                base.Render(writer);
        }
    }

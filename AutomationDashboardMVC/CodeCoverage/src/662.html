using System.Collections.Generic;

namespace Aurigo.AMP3.LibraryDTO
{
    public class LibCatalogs
    {
        internal string baseOp;
        internal KeyValuePair<string, List<string>>[] filters;
        internal string isFiltered;
        internal string name;
        internal string sortOrder;

        public LibCatalogs(string name)
        {
            this.name = name;
        }

        public LibCatalogs(string name, string sortOrder)
        {
            this.name = name;
            this.sortOrder = sortOrder;
        }

        internal LibCatalogs(string name, BaseOperators baseOp, params KeyValuePair<string, List<string>>[] cols)
        {
            this.name = name;
            this.baseOp = baseOp == BaseOperators.And ? "And" : baseOp == BaseOperators.Or ? "Or" : null;
            isFiltered = "true";
            filters = cols;
        }

        internal LibCatalogs(string name, BaseOperators baseOp, string sortOrder,
            params KeyValuePair<string, List<string>>[] cols)
        {
            this.name = name;
            this.baseOp = baseOp == BaseOperators.And ? "And" : baseOp == BaseOperators.Or ? "Or" : null;
            isFiltered = "true";
            this.sortOrder = sortOrder;
            filters = cols;
        }

        #region Nested type: BaseOperators

        internal enum BaseOperators
        {
            And = 0,
            Or = 1
        };

        #endregion
    }
}

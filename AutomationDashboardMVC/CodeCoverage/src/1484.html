using System.Collections.Generic;
using System.Xml.Serialization;

#pragma warning disable 1591

namespace Aurigo.Common
{
    [XmlRoot("Visibility")]
    public class Root
    {
        private XMLFormStageVisibility _visibility;

        public Root()
        {
        }

        [XmlElement("XMLFormStageVisibility")]
        public XMLFormStageVisibility XMLFormStageVisibility
        {
            get { return _visibility; }
            set { _visibility = value; }
        }

        public string ModuleID;
    }

    public class XMLFormStageVisibility
    {
        [XmlElement("WorkflowStage")] public List<WorkflowStage> WorkflowStages = new List<WorkflowStage>();
    }

    public class WorkflowStage
    {
        [XmlAttribute] public string Name;
        [XmlElement("ViewStakeholder")] public List<ViewStakeholder> ViewStakeholders = new List<ViewStakeholder>();
    }

    public class ViewStakeholder
    {
        [XmlAttribute] public string RoleName;
        public string HideStages;
        [XmlElement("XMLGrid")] public List<XMLGrid> Grids = new List<XMLGrid>();
    }

    public class XMLGrid
    {
        [XmlAttribute] public string Name;
        public string HideColumns;
    }
}

        /// <summary>
        /// This field is a readonly field to indicate whether a control contains DateTime value.
        /// </summary>
        [XmlIgnore]
        public bool IsTimezoneApplicable
        {
            get
            {

                return ((this.Type == ControlType.Date || this.Type == ControlType.DateTime)
                 || ((this.Type == ControlType.Hidden || this.Type == ControlType.Display || this.Type == ControlType.TextBox)
                    && ((!string.IsNullOrEmpty(this.DBType) && this.DBType.StartsWith("date", StringComparison.CurrentCultureIgnoreCase)))));
            }
        }

        [XmlIgnore]
        public List<ValidationType> ValidationTypes
        {
            get
            {
                var toReturn = new List<ValidationType>();
                if (!string.IsNullOrEmpty(Validations))
                    Validations.Split(',').ForEach(
                        str =>
                        {
                            // KLK: Fix bug where if you type incorrect validation string, software crashes
                            ValidationType vType;

                            if (Enum.TryParse(str, false, out vType))
                                toReturn.Add(vType);
                        });
                return toReturn;
            }
        }

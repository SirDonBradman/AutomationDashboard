using System;
using System.Data;
using System.Data.Linq.Mapping;
using Aurigo.Brix.Platform.CoreUtilities.Utility;

namespace Aurigo.Brix.Platform.BusinessLayer.DTO
{
    [Table(Name = "dbo.ReportDetails")]
    [Serializable]
    public class ReportDetails
    {
        public ReportDetails()
        {
            Name = string.Empty;
            Description = String.Empty;
            Context = string.Empty;
            FileName = string.Empty;
            IsActive = false;
            ModelType = string.Empty;
            Category = string.Empty;
            ModuleId = string.Empty;
            ParentContext = string.Empty;
            ParentModuleID = string.Empty;
            ReportType = string.Empty;
            Code = string.Empty;
            UploadedBy = 0;
            UploadedDate = DateTime.UtcNow;
            IsUserUploaded = false;
            SelectRecord = 10;
            QueryStringParameters = "";
        }

        public ReportDetails(DataRow dr)
        {
            Name = dr.IsNull("Name") ? string.Empty : dr["Name"].ToString2();
            ParentModuleID = dr.IsNull("ParentModuleID") ? string.Empty : dr["ParentModuleID"].ToString2();
            Description = dr.IsNull("Description") ? string.Empty : dr["Description"].ToString2();
            FileName = dr.IsNull("FileName") ? string.Empty : dr["FileName"].ToString2();
            IsActive = dr.IsNull("IsActive") ? true : dr["IsActive"].ToString2().ToBoolean2();
            ModuleId = dr.IsNull("ModuleId") ? string.Empty : dr["ModuleId"].ToString2();
            ModelType = dr.IsNull("ModelType") ? string.Empty : dr["ModelType"].ToString2();
            ParentContext = dr.IsNull("ParentContext") ? string.Empty : dr["ParentContext"].ToString2();
            Context = dr.IsNull("Context") ? string.Empty : dr["Context"].ToString2();
            Code = dr.IsNull("Code") ? string.Empty : dr["Code"].ToString2();
            Category = dr.IsNull("Category") ? string.Empty : dr["Category"].ToString2();
            ReportType = dr.IsNull("ReportType") ? string.Empty : dr["ReportType"].ToString2();
            QueryStringParameters = dr.IsNull("QueryStringParameters")
                ? string.Empty
                : dr["QueryStringParameters"].ToString2();
            GridParameters = dr.IsNull("GridParameters") ? string.Empty : dr["GridParameters"].ToString2();
            IsUserUploaded = dr.IsNull("IsUserUploaded") ? false : dr["IsUserUploaded"].ToString2().ToBoolean2();
            SelectRecord = dr.IsNull("SelectRecord")
                ? 10
                : dr["SelectRecord"].ToInt32_2();
        }

        [Column(Name = "ReportDetailID", DbType = "int", IsPrimaryKey = true, IsDbGenerated = true)]
        public int ReportDetailID { get; set; }

        [Column(Name = "Name", DbType = "varchar(500)", CanBeNull = false)]
        public string Name { get; set; }

        [Column(Name = "Description", DbType = "varchar(500)", CanBeNull = true)]
        public string Description { get; set; }

        [Column(Name = "Context", DbType = "varchar(20)", CanBeNull = false)]
        public string Context { get; set; }

        [Column(Name = "FileName", DbType = "varchar(500)")]
        public string FileName { get; set; }

        [Column(Name = "ModelType", DbType = "varchar(200)", CanBeNull = false)]
        public string ModelType { get; set; }

        [Column(Name = "Category", DbType = "varchar(50)")]
        public string Category { get; set; }

        [Column(Name = "ModuleID", DbType = "char(7)", CanBeNull = false)]
        public string ModuleId { get; set; }

        [Column(Name = "ParentContext", DbType = "varchar(20)", CanBeNull = false)]
        public string ParentContext { get; set; }

        [Column(Name = "ParentModuleID", DbType = "char(7)")]
        public string ParentModuleID { get; set; }

        [Column(Name = "ReportType", DbType = "varchar(20)", CanBeNull = false)]
        public string ReportType { get; set; }

        [Column(Name = "Code", DbType = "varchar(10)")]
        public string Code { get; set; }

        [Column(Name = "IsActive", DbType = "bit", CanBeNull = false)]
        public bool IsActive { get; set; }

        [Column(Name = "UploadedBy", DbType = "int", CanBeNull = false)]
        public int UploadedBy { get; set; }

        [Column(Name = "UploadedDate", DbType = "smalldatetime", CanBeNull = false)]
        public DateTime UploadedDate { get; set; }

        [Column(Name = "QueryStringParameters", DbType = "varchar(500)", CanBeNull = true)]
        public string QueryStringParameters { get; set; }

        [Column(Name = "GridParameters", DbType = "varchar(500)", CanBeNull = true)]
        public string GridParameters { get; set; }

        [Column(Name = "IsUserUploaded", DbType = "bit", CanBeNull = false)]
        public bool IsUserUploaded { get; set; }

        [Column(Name = "SelectRecord", DbType = "int", CanBeNull = false)]
        public int SelectRecord { get; set; }
    }

    public enum ReportHostType
    {
        dda,
        sql
    }
}

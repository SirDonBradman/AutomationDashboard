using System;
using System.Data.Linq.Mapping;

namespace Aurigo.Brix.Platform.BusinessLayer.DTO
{
    [Table(Name = "dbo.APPMGMTGridColumnSettings")]
    [Serializable]
    public class ColumnDetails
    {
        public ColumnDetails()
        {
            IsVisible = true;
        }

        [Column(Name = "ID", DbType = "int", IsPrimaryKey = true, IsDbGenerated = true)]
        public int ID { get; set; }

        [Column(Name = "Context", DbType = "varchar(50)", CanBeNull = false)]
        public string Context { get; set; }

        [Column(Name = "ColumnKey", DbType = "varchar(100)", CanBeNull = false)]
        public string ColumnKey { get; set; }

        [Column(Name = "ColumnCaption", DbType = "varchar(200)", CanBeNull = false)]
        public string ColumnCaption { get; set; }

        [Column(Name = "Scope", DbType = "varchar(50)")]
        public string Scope { get; set; }

        [Column(Name = "RoleID", DbType = "int")]
        public int RoleID { get; set; }

        [Column(Name = "UserID", DbType = "int")]
        public int UserID { get; set; }

        [Column(Name = "IsVisible", DbType = "bit")]
        public bool IsVisible { get; set; }

        [Column(Name = "Order", DbType = "int")]
        public int Order { get; set; }
    }

    [Table(Name = "dbo.APPMGMTGridPageSizeSetting")]
    [Serializable]
    public class PageSizeSetting
    {
        [Column(Name = "ID", DbType = "int", IsPrimaryKey = true, IsDbGenerated = true)]
        public int ID { get; set; }

        [Column(Name = "Context", DbType = "varchar(50)", CanBeNull = false)]
        public string Context { get; set; }

        [Column(Name = "PageSize", DbType = "int", CanBeNull = false)]
        public int PageSize { get; set; }

        [Column(Name = "Scope", DbType = "varchar(50)")]
        public string Scope { get; set; }

        [Column(Name = "RoleID", DbType = "int")]
        public int RoleID { get; set; }

        [Column(Name = "UserID", DbType = "int")]
        public int UserID { get; set; }
    }

    [Table(Name = "dbo.APPMGMTColumnResizeSettings")]
    [Serializable]
    public class ColumnResizeSettings
    {
        [Column(Name = "ID", DbType = "int", IsPrimaryKey = true, IsDbGenerated = true)]
        public int ID { get; set; }

        [Column(Name = "Context", DbType = "varchar(50)", CanBeNull = false)]
        public string Context { get; set; }

        [Column(Name = "UserID", DbType = "int")]
        public int UserID { get; set; }

        [Column(Name = "RoleID", DbType = "int")]
        public int RoleID { get; set; }

        [Column(Name = "Settings", DbType = "varchar(max)")]
        public string Settings { get; set; }
    }

    public enum ColumnSettingScope
    {
        All,
        Role,
        User
    }
}

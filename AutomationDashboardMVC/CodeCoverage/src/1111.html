using System.Web;
using System.Web.UI;
using Aurigo.AMP3.Core.AbstractModels;
using Aurigo.Brix.Platform.BusinessLayer.AbstractModels;
using Aurigo.Brix.Platform.BusinessLayer.BL;
using Aurigo.Brix.Platform.CoreUtilities.Utility;
using Telerik.Web.UI;

namespace Aurigo.Brix.Platform.BusinessLayer.ConcreteModels.Library
{
    public class LibraryProjectPhaseListModel : XMLListManagerModel
    {
        public override string GetDefaultSortColumn(ListModel model, XmlFormArgs e)
        {
            return "SequenceNo_VCRef";
        }

        public override GridSortOrder GetDefaultSortIndicator(ListModel model, XmlFormArgs e)
        {
            return GridSortOrder.Ascending;
        }

        public override void BeforeSetUIDetails(ListModel model, XmlFormArgs e)
        {
            model.DisplayReorder = true;
        }

        public override void HandleMenuItem(string eventString, string selectedIds, ListModel model, XmlFormArgs e)
        {
            if (eventString.ToLower2().Equals("reorder"))
            {
                string url = string.Format("~/Modules/LIBRARY/ReorderProjectPhase.aspx?xContext={0}&Module={1}", HttpContext.Current.Request.QueryString["xContext"], model.ModuleId);
                Page page = (HttpContext.Current.CurrentHandler as Page);
                page.Response.Redirect(url);
            }
            else
                base.HandleMenuItem(eventString, selectedIds, model, e);
        }

        public override void AfterDelete(ListModel model, string selectedIds, XmlFormArgs e)
        {
            ProjectPhaseManager.Instance.UpdateProjectPhaseLibraryOnDelete();
            base.AfterDelete(model, selectedIds, e);
        }
    }
}

using System;
using Aurigo.AMP3.Common;
using Telerik.Web.UI;

namespace Aurigo.Brix.Core.WebUiHelper
{
    [Serializable]
    public class FormStateMgmt : StateManagementBase
    {
        private int? _RowId;
        private int _contractID;

        public FormStateMgmt(string stateId, int contractID)
            : base(stateId)
        {
            _contractID = contractID;
        }

        public int ContractID
        {
            get { return _contractID; }
            set { _contractID = value; }
        }

        public int? ActiveRowID
        {
            get { return _RowId; }
            set { _RowId = value; }
        }

        public override void MaintainState(int pageIdx, string sortKey, GridSortOrder sortOrder, string filterCriterion)
        {
            ContractID = ContractID;
            base.MaintainState(pageIdx, sortKey, sortOrder, filterCriterion);
        }

        public void MaintainState(int pageIdx, string sortKey, GridSortOrder sortOrder, string filterCriterion,
            int? activeRowId)
        {
            _RowId = activeRowId;
            ContractID = ContractID;
            base.MaintainState(pageIdx, sortKey, sortOrder, filterCriterion);
        }

        public override void ClearState()
        {
            base.ClearState();
            ContractID = 0;
            ActiveRowID = null;
        }
    }
}

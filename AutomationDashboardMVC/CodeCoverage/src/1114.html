using Aurigo.AMP3.Core.UserControls;
using Aurigo.Brix.Platform.BusinessLayer.AbstractModels;
using System.Web.UI.WebControls;
namespace Aurigo.Brix.Platform.BusinessLayer.ConcreteModels.Library
{
    public class MailMergeTemplateFormManagerModel : XMLFormManagerModel
    {
        public override void OnPageLoad(BrixFormModel model, XmlFormArgs e)
        {
            if (!string.IsNullOrEmpty(model.form.page.Request["Mode"].ToString()))
            {
                string mode = model.form.page.Request["Mode"].ToString();
                if (mode.ToLower() == "edit")
                    (model.form.GetControl("DocumentTemplateName").ControlReference[0] as TextBox).Enabled = false;
            }

            AttachmentsControl mailMergeAttachmentCtrl = ((AttachmentsControl)(((System.Web.UI.Control)(model.form.GetControl("MailMergeTemplatesControl").ControlReference[0])).Controls[0]));            
            base.OnPageLoad(model, e);
        }

        public override void OnInit(BrixFormModel model, XmlFormArgs e)
        {
            AttachmentsControl mailMergeAttachmentCtrl = ((AttachmentsControl)(((System.Web.UI.Control)(model.form.GetControl("MailMergeTemplatesControl").ControlReference[0])).Controls[0]));
            mailMergeAttachmentCtrl.LinkDocumentVisible = false;

            base.OnInit(model, e);
        }

        public override void BeforeSave(BrixFormModel model, XmlFormArgs e)
        {
            AttachmentsControl mailMergeAttachmentCtrl = ((AttachmentsControl)(((System.Web.UI.Control)(model.form.GetControl("MailMergeTemplatesControl").ControlReference[0])).Controls[0]));
            if (!mailMergeAttachmentCtrl.HasAttachments())
            {
                throw new System.Exception("Upload at least one document");
            }
            base.BeforeSave(model, e);
        }
    }
}

using System.Linq;
using System.Net;
using System.Net.Http;
using System.Web;
using System.Web.Http;

namespace Aurigo.Brix.Platform.BusinessLayer.Controller
{

    [MWAuthentication]
    public class LogoutController : ApiController
    {
        [HttpPost]
        public HttpResponseMessage Post()
        {
            HttpResponseMessage resp = null;
            string token = Request.Headers.GetValues("Authorization-Token").First();
            if (!string.IsNullOrEmpty(token))
            {
                string userIP = HttpContext.Current.Request.UserHostAddress;
                MWTokenList.RemoveToken(token, userIP);
                resp = Request.CreateResponse<Status>(HttpStatusCode.OK,
                    new Status { ErrorCode = "0", ErrorMessage = "Token removed successfully." });
                return resp;
            }
            resp = Request.CreateResponse<Status>(HttpStatusCode.OK,
                new Status
                {
                    ErrorCode = "1",
                    ErrorMessage = "Authorization token is not provided or invalid authorization token."
                });
            return resp;
        }
    }
}

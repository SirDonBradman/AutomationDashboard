using System;
using System.Collections.Generic;
using Aurigo.Brix.Core.WebUiHelper;
using UC = Aurigo.AMP3.Core.UserControls;
using Aurigo.AMP3.Common;
using Aurigo.Brix.Platform.BusinessLayer;

namespace Aurigo.AMP3.ReportsUI
{
    public partial class InstantReport : BrixPageBase
    {
        protected override void OnPreInit(System.EventArgs e)
        {
            base.OnPreInit(e);
            CustomAdHocConfig.InitializeReporting();
        }

        protected override void OnLoad(System.EventArgs e)
        {
            Page.Title = "Instant Report";
        }

        protected override void OnInit(EventArgs e)
        {
            ModuleId = Constants.MODID_ENTPRSE;
            PermissionsToCheck.Add(Constants.MODFEAT_VISIBILITY);
            PermissionsToCheck.Add(Constants.MODFEAT_MYREPORTS);

            List<UC.MenuGroup> menuGroups = new List<UC.MenuGroup>();
            UC.MenuGroup generalGroup = new UC.MenuGroup("General");
            generalGroup.AddMenu(new UC.LargeButton("lnkBack", "Back", "Icn_Back.png",
                "../../Common/BrixListPage.aspx?Context=REPORTS"));
            generalGroup.AddMenu(new UC.LargeButton("lnkNew", "Design", "Icn_New2.png"));
            generalGroup.AddMenu(new UC.LargeButton("lnkView", "View", "Icn_View.png"));
            menuGroups.Add(generalGroup);
            CreateToolBarMenu(menuGroups, null);
        }

        protected override void CustomizeToolBar()
        {
            if (MainToolBar.GetMenuReference("lnkNew") != null)
            {
                MainToolBar.GetMenuReference("lnkNew").CausesValidation = false;
                MainToolBar.GetMenuReference("lnkNew").Click += lnkNew_ServerClick;
                MainToolBar.GetMenuReference("lnkNew").OnClientClick =
                    "return DesignReportRequest(CollectReportData()); return false;";
            }
            if (MainToolBar.GetMenuReference("lnkView") != null)
            {
                MainToolBar.GetMenuReference("lnkView").CausesValidation = false;
                MainToolBar.GetMenuReference("lnkView").Click += lnkView_ServerClick;
                MainToolBar.GetMenuReference("lnkView").OnClientClick =
                    "return ViewReportRequest(CollectReportData()); return false;";
            }
        }

        protected void lnkNew_ServerClick(object sender, EventArgs e)
        {
            Response.Redirect("~/Modules/REPORTS/InstantReport.aspx#design_report");
        }

        protected void lnkView_ServerClick(object sender, EventArgs e)
        {
            Response.Redirect("~/Modules/REPORTS/InstantReport.aspx#view_report");
        }
    }
}

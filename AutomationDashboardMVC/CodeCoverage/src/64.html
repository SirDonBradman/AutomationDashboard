        public Module FindModuleByModId(string modId)
        {
            IEnumerator<Module> modules = Modules.GetEnumerator();
            while (modules.MoveNext())
            {
                if (modules.Current.ModID.Equals(modId))
                    return modules.Current;
            }
            throw new Exception("Module not found.");
        }

        public bool ContainsModule(string modId)
        {
            IEnumerator<Module> modules = Modules.GetEnumerator();
            while (modules.MoveNext())
            {
                if (modules.Current.ModID.Equals(modId))
                    return true;
            }
            return false;
        }
    }

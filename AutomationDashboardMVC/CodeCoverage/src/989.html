using System;

namespace Aurigo.Brix.Platform.BusinessLayer.UserControls
{
    public partial class DocumentDashboardQuickUpload : System.Web.UI.UserControl
    {
        protected global::Aurigo.Brix.Platform.BusinessLayer.UserControls.DocumentSearchControl DocumentSearch;

        protected global::Aurigo.Brix.Platform.BusinessLayer.UserControls.RecentActivities RecentActivities1;

        protected void Page_Load(object sender, EventArgs e)
        {
            DocumentSearch.EnableAdvSearch = false;
            RecentActivities1.ReturnURL = Server.UrlEncode(this.Page.Request.Url.PathAndQuery);
            RecentActivities1.ProjectID = Convert.ToInt32(this.Page.Request.QueryString["PID"]);
        }
    }
}

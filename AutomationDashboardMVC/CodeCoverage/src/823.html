using System;
using System.ComponentModel;
using System.Data.Linq.Mapping;

namespace Aurigo.AMP3.Core.BusinessLayer.DTO
{
    [Table(Name = "dbo.CORITEMItemDetails")]
    public class Item : INotifyPropertyChanging, INotifyPropertyChanged
    {
        #region Fields

        private Decimal? _DiffQuantity = 0; //By Default, DiffQuantity represents no change in Quantity

        #endregion

        #region Properties

        /// <summary>
        /// The database automatically generates this value
        /// </summary>
        [Column(Name = "ItemID", DbType = "int", CanBeNull = false)]
        public Int32? ItemID { get; set; }

        [Column(Name = "ModuleID", DbType = "char(7)", CanBeNull = false)]
        public String ModuleID { get; set; }

        public String Unit { get; set; }

        [Column(Name = "ParentInstanceID", DbType = "int", CanBeNull = false)]
        public Int32 ParentInstanceID { get; set; }

        [Column(Name = "LineNo", DbType = "int", CanBeNull = false)]
        public Int32 LineNo { get; set; }

        /// <summary>
        /// This is not a required field
        /// </summary>
        [Column(Name = "StandardItemNo", DbType = "varchar(100)", CanBeNull = false)]
        public String StandardItemNo { get; set; }

        [Column(Name = "Description", DbType = "varchar(8000)", CanBeNull = false)]
        public String Description { get; set; }

        /// <summary>
        /// This is not a required field
        /// </summary>
        [Column(Name = "AlternateID", DbType = "int", CanBeNull = true)]
        public Int32? AlternateID { get; set; }

        public Int32 ActivityCount { get; set; }

        [Column(Name = "Quantity", DbType = "Decimal(18,4)", CanBeNull = false)]
        public Decimal Quantity { get; set; }

        [Column(Name = "UnitPrice", DbType = "Decimal(18,4)", CanBeNull = false)]
        public Decimal UnitPrice { get; set; }

        [Column(Name = "UnitID", DbType = "int", CanBeNull = false)]
        public Int32 UnitID { get; set; }

        /// <summary>
        /// This is not a required field
        /// </summary>
        [Column(Name = "AccountingCode", DbType = "varchar(500)", CanBeNull = true)]
        public String AccountingCode { get; set; }

        /// <summary>
        /// This is not a required field
        /// </summary>
        [Column(Name = "IsComplete", DbType = "bit", CanBeNull = false)]
        public Boolean? IsComplete { get; set; }

        /// <summary>
        /// This is not a required field
        /// </summary>
        [Column(Name = "IsNonContract", DbType = "bit", CanBeNull = false)]
        public Boolean? IsNonContract { get; set; }

        /// <summary>
        /// This is not a required field
        /// </summary>
        [Column(Name = "IsMustBid", DbType = "bit", CanBeNull = false)]
        public Boolean? IsMustBid { get; set; }

        [Column(Name = "CreatedBy", DbType = "int", CanBeNull = false)]
        public Int32 CreatedBy { get; set; }

        [Column(Name = "CreatedOn", DbType = "datetime", CanBeNull = false)]
        public DateTime CreatedOn { get; set; }

        /// <summary>
        /// This is not a required field
        /// </summary>
        [Column(Name = "LastUpdatedBy", DbType = "int", CanBeNull = true)]
        public Int32? LastUpdatedBy { get; set; }

        /// <summary>
        /// This is not a required field
        /// </summary>
        [Column(Name = "LastUpdatedOn", DbType = "datetime", CanBeNull = true)]
        public DateTime? LastUpdatedOn { get; set; }

        /// <summary>
        /// This is not a required field
        /// </summary>
        [Column(Name = "Notes", DbType = "varchar(2000)", CanBeNull = true)]
        public String Notes { get; set; }

        /// <summary>
        /// This is not a required field
        /// </summary>
        [Column(Name = "Margin", DbType = "Decimal(18,4)", CanBeNull = true)]
        public Decimal? Margin { get; set; }

        /// <summary>
        /// This is not a required field
        /// </summary>
        [Column(Name = "Discount", DbType = "Decimal(18,4)", CanBeNull = true)]
        public Decimal? Discount { get; set; }

        /// <summary>
        /// This is not a required field
        /// </summary>
        [Column(Name = "UnitCost", DbType = "Decimal(18,4)", CanBeNull = true)]
        public Decimal? UnitCost { get; set; }

        /// <summary>
        /// This is not a required field
        /// </summary>
        [Column(Name = "BaseUnitCost", DbType = "Decimal(18,4)", CanBeNull = true)]
        public Decimal? BaseUnitCost { get; set; }

        /// <summary>
        /// This is not a required field
        /// </summary>
        [Column(Name = "OverheadCost", DbType = "Decimal(38,4)", CanBeNull = true)]
        public Decimal? OverheadCost { get; set; }

        /// <summary>
        /// This is not a required field
        /// </summary>
        [Column(Name = "ReferenceItemID", DbType = "int", CanBeNull = true)]
        public Int32? ReferenceItemID { get; set; }

        /// <summary>
        /// This is not a required field
        /// </summary>
        [Column(Name = "ReferenceItemType", DbType = "char(1)", CanBeNull = true)]
        public String ReferenceItemType { get; set; }

        /// <summary>
        /// This is a computed field, do not assign values to it
        /// </summary>
        [Column(Name = "TimeStamp", DbType = "datetime", CanBeNull = false)]
        public Byte[] TimeStamp { get; set; }

        [Column(Name = "ContainerID", DbType = "int", CanBeNull = true)]
        public Int32? ContainerID { get; set; }

        public string ContainerName { get; set; }

        [Column(Name = "ParentID", DbType = "int", CanBeNull = true)]
        public Int32? ParentID { get; set; }

        /// <summary>
        /// This is not a required field
        /// </summary>
        public String Path { get; set; }

        [Column(Name = "GroupID", DbType = "int", CanBeNull = true)]
        public Int32 GroupID { get; set; }

        public string GroupName { get; set; }

        [Column(Name = "RuleID", DbType = "int", CanBeNull = true)]
        public Int32 RuleID { get; set; }

        public string RuleName { get; set; }

        [Column(Name = "IsReWork", DbType = "bit", CanBeNull = false)]
        public Boolean? IsReWork { get; set; }

        [Column(Name = "LinkedBudgetItem", DbType = "int", CanBeNull = true)]
        public int? LinkedBudgetItem { get; set; }

        public string LinkedBudgetItemName { get; set; }

        [Column(Name = "Priority", DbType = "int", CanBeNull = true)]
        public Int32? Priority { get; set; }

        [Column(Name = "Weightage", DbType = "Decimal(5,2)", CanBeNull = true)]
        public Decimal? Weightage { get; set; }

        public Decimal? DiffQuantity
        {
            get { return _DiffQuantity; }
            set { _DiffQuantity = value; }
        }

        [Column(Name = "ProjectID", DbType = "int", CanBeNull = true)]
        public Int32 ProjectID { get; set; }

        public string ProjectName { get; set; }

        // Inflation columns
        [Column(Name = "InflationRuleID", DbType = "int", CanBeNull = true)]
        public int? InflationRuleID { get; set; }

        [Column(Name = "InflationRuleName", DbType = "nvarchar(100)", CanBeNull = true)]
        public string InflationRuleName { get; set; }

        [Column(Name = "AnnualInflationRate", DbType = "decimal(34, 2)", CanBeNull = true)]
        public decimal? AnnualInflationRate { get; set; }

        [Column(Name = "InflationEffectiveFrom", DbType = "datetime", CanBeNull = true)]
        public DateTime? InflationEffectiveFrom { get; set; }

        [Column(Name = "Inflation", DbType = "decimal(34, 2)", CanBeNull = true)]
        public decimal Inflation { get; set; }

        [Column(Name = "AmountForResource", DbType = "decimal(38, 4)", CanBeNull = true)]
        public decimal AmountForResource { get; set; }

        [Column(Name = "TotalItemEstimate", DbType = "decimal(38, 4)", CanBeNull = true)]
        public decimal TotalItemEstimate { get; set; }

        #endregion

        private static readonly PropertyChangingEventArgs emptyChangingEventArgs =
            new PropertyChangingEventArgs(String.Empty);

        #region INotifyPropertyChanged Members

        public event PropertyChangedEventHandler PropertyChanged;

        #endregion

        #region INotifyPropertyChanging Members

        public event PropertyChangingEventHandler PropertyChanging;

        #endregion

        protected internal virtual void SendPropertyChanging()
        {
            if ((PropertyChanging != null))
            {
                PropertyChanging(this, emptyChangingEventArgs);
            }
        }

        protected internal virtual void SendPropertyChanged(String propertyName)
        {
            if ((PropertyChanged != null))
            {
                PropertyChanged(this, new PropertyChangedEventArgs(propertyName));
            }
        }
    }

    [Serializable]
    public class GlobalGroup
    {
        public int GroupID { get; set; }
        public string GroupName { get; set; }
        public string GroupDesc { get; set; }
    }
}

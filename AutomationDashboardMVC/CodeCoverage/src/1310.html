using System;
using System.Data;
using System.Web.UI;
using Aurigo.AMP3.LibraryBL;
using Aurigo.Brix.Platform.BusinessLayer.Utility;

namespace Aurigo.AMP3.EnterpriseUI
{
    public partial class LibrarySummary : UserControl
    {
        protected void Page_Load(object sender, EventArgs e)
        {
            DataTable dt = LibraryInterface.Instance.GetLibrarySummary(0);

            if (dt.Rows.Count > 0)
            {
                lblErr.Text = "";
                ChartLibSummary.Data.DataSource = dt.ToMWDateTime();
                ChartLibSummary.Data.DataBind();

                ChartLibSummary.Tooltips.FormatString = "Catalog Count: <DATA_VALUE>";
                ChartLibSummary.Visible = true;
            }
            else
            {
                lblErr.Text = "No library details available";
            }
        }
    }
}

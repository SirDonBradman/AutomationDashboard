using Aurigo.Brix.Platform.BusinessLayer.AbstractModels;
using Aurigo.Brix.Platform.BusinessLayer.XMLForm;

namespace Aurigo.Brix.Platform.UI.Modules.FRMBLDR
{
    public class AddTabs : AddProcessor
    {
        public AddTabs(BrixFormModel model) : base(model)
        {
        }

        public override string ControlName => "tabs";
        public override string Add(Command cmd, out ControlContainer control)
        {
            control = null;

            var dest = FormBuilderSupport.GetDestination(_model, cmd.destinationControl);
            if (dest == null)
                return "Could not find destination control";

            var tabs = new xTabs { Name = GetUniqueName(ContainerTypes.Tabs), Width = "700px" };
            dest.Containers.Add(tabs);
            control = tabs;

            var tab = new xTab { Name = GetUniqueName(ContainerTypes.Tab), IsSelected = true };
            tab.Caption = tab.Name;
            tabs.Containers.Add(tab);

            var s = new Section
            {
                Name = GetUniqueName(ContainerTypes.Section),
                TabName = tab.Name
            };
            s.Caption = s.Name;
            tab.Containers.Add(s);

            var ctl = new xControl
            {
                Name = GetUniqueName(ControlType.TextBox),
                Type = ControlType.TextBox,
                DBType = "nvarchar(max)"
            };
            ctl.Caption = ctl.Name;
            s.Controls.Add(ctl);

            return string.Empty;
        }
    }
}

    public class MainChecklistDTO
    {
        public int? ID { get; set; }

        public int? ChecklistID { get; set; }

        public int? StageID { get; set; }

        public int? ActivityID { get; set; }

        public int? ParcelID { get; set; }

        public bool? IsFromLib { get; set; }

        public bool? IsCompleted { get; set; }

        public override string ToString()
        {
            try
            {
                using (var sWr = new StringWriter(CultureInfo.CurrentCulture))
                {
                    using (var xWr = new XmlTextWriter(sWr))
                    {
                        xWr.WriteStartElement("CheckList");
                        if (IsFromLib.HasValue)
                            xWr.WriteElementString("FromLib", IsFromLib.ToString2());
                        if (ID.HasValue)
                            xWr.WriteElementString("ID", ID.ToString2());
                        if (ChecklistID.HasValue)
                            xWr.WriteElementString("ChecklistID", ChecklistID.ToString2());
                        if (StageID.HasValue)
                            xWr.WriteElementString("StageID", StageID.ToString2());
                        if (ActivityID.HasValue)
                            xWr.WriteElementString("ActivityID", ActivityID.ToString2());
                        if (ParcelID.HasValue)
                            xWr.WriteElementString("ParcelID", ParcelID.ToString2());
                        if (IsFromLib.HasValue)
                            xWr.WriteElementString("IsFromLib", IsFromLib.ToString2());
                        if (IsCompleted.HasValue)
                            xWr.WriteElementString("IsCompleted", IsCompleted.ToString2());
                        xWr.WriteEndElement(); // CheckList
                        xWr.Flush();
                        return sWr.ToString2();
                    }
                }
            }
            catch (Exception ex)
            {
                Logger.Log(Enumerations.LogType.Error, ex.ToString2(), LMConstants.MODULENAME);
                throw;
            }
        }
    }
}

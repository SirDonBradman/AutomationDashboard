using System;
using System.Data;
using System.Web.UI;
using Aurigo.AMP3.Common;
using Aurigo.AMP3.ProjectBL;
using Aurigo.AMP3.Resources.MessageResources;

namespace Aurigo.AMP3.EnterpriseUI
{
    public partial class ProjectSummaryByCost : UserControl
    {
        protected void Page_Load(object sender, EventArgs e)
        {
            ProjectManager projectBL = ProjectManager.Instance;
            UserDetail ud = UserDetail.GetCurrentUserDetail();
            DataTable dt = projectBL.GetProjectCountByCost("", ud.UID, ud.RID);

            if (dt.Rows.Count > 0)
            {
                lblErr.Text = "";
                ChartProjectStatusByCost.Data.DataSource = dt;
                ChartProjectStatusByCost.Data.DataBind();
                ChartProjectStatusByCost.Visible = true;
            }
            else
            {
                ChartProjectStatusByCost.Visible = false;
                lblErr.Text = MessageResourceManager.GetString("W_ENTPRSE_NO_PROJECTS_AVAILABLE",
                    Enumerations.MessageType.WarningMessage);
            }
        }
    }
}

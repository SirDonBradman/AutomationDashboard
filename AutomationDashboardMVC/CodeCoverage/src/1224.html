using System;
using System.Collections.Generic;
using System.ServiceModel;
using System.ServiceModel.Activation;
using System.Web;
using Aurigo.AMP3.Common;
using Aurigo.DataBinding;

namespace ItemGanttView.Web
{
    [ServiceContract(Namespace = "")]
    [SilverlightFaultBehavior]
    [AspNetCompatibilityRequirements(RequirementsMode = AspNetCompatibilityRequirementsMode.Allowed)]
    public class DataService
    {
        [OperationContract]
        public ScheduleDetails GetData(string Context, int parentID, int calendarID)
        {
            if (UserDetail.GetCurrentUserDetail() == null)
                return null;
            return (new SQLData()).GetData(Context, parentID, calendarID);
        }

        [OperationContract]
        public ScheduleDetails GetFilteredData(string Context, int parentID, int calendarID, string filterXml)
        {
            if (UserDetail.GetCurrentUserDetail() == null)
                return null;
            return (string.IsNullOrEmpty(filterXml)
                ? (new SQLData()).GetData(Context, parentID, calendarID)
                : (new SQLData()).GetFilteredData(Context, parentID, calendarID, filterXml));
        }

        [OperationContract]
        public bool SaveData(string Context, List<TaskInfo> xmlData, int parentID, int PID, int UID)
        {
            if (UserDetail.GetCurrentUserDetail() == null)
                return false;
            return (new SQLData()).SaveData(Context, xmlData, parentID, PID, UID);
        }

        [OperationContract]
        public TaskInfo GetYearWiseSplit(string Context, int parentID, DateTime planYear, TaskInfo data)
        {
            if (UserDetail.GetCurrentUserDetail() == null)
                return null;
            return (new SQLData()).GetYearWiseSplit(Context, parentID, planYear, data);
        }

        [OperationContract]
        public TagInfo GetTagInfo()
        {
            return new TagInfo();
        }

        [OperationContract]
        public UserDetail GetCurrentUserDetail()
        {
            return UserDetail.GetCurrentUserDetail();
        }

        [OperationContract]
        public string GetSessionValue(string key)
        {
            return HttpContext.Current.Session[key] != null ? HttpContext.Current.Session[key].ToString() : string.Empty;
        }

        [OperationContract]
        public Months GetMonthName(int monthNo)
        {
            return (Months)monthNo;
        }

        [OperationContract]
        public ScheduleDetails GetScheduleDetails()
        {
            return new ScheduleDetails();
        }
    }
}

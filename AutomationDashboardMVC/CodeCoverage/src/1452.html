using System;
using System.Collections.Generic;
using Aurigo.Brix.ReportServerHelper2005;

namespace Aurigo.Brix.ReportServerHelper
{
    public class SubscriptionDTO
    {
        public string subscriptionId { get; set; }
        public ExtensionSettings extensionSetting;
        public string desc { get; set; }
        public string status { get; set; }
        public string eventType { get; set; }
        public string matchData { get; set; }
        public ActiveState state { get; set; }
        public ParameterValue[] pars { get; set; }

        public SubscriptionDTO()
        {
            extensionSetting = new ExtensionSettings();
        }
    }

    public class EmailDeliverySetting
    {
        public string TO { get; set; } // required field *
        public string CC { get; set; }
        public string BCC { get; set; }
        public string ReplyTo { get; set; }
        public string Subject { get; set; } // required field *
        public string Comment { get; set; }
        public string Priority { get; set; } // HIGH, NORMAL, LOW
        public string IncludeReport { get; set; } // required field *
        public string RenderFormat { get; set; } // PDF, CSV, EXCEL, WORD, IMAGE , XML, HTML4.0, MHTML
        public string IncludeLink { get; set; }
    }

    public class WindowShareDeliverySetting
    {
        public string FILENAME { get; set; } //filename while saving on window share
        public string FILEEXTN { get; set; } // append file extension or not true/false
        public string PATH { get; set; } // window share path
        public string RENDER_FORMAT { get; set; } // PDF, CSV, EXCEL, WORD, IMAGE , XML, HTML4.0, MHTML
        public string WRITEMODE { get; set; } //AutoIncrement, overwrite
        public string USERNAME { get; set; } // credentials to access the shared path
        public string PASSWORD { get; set; }
    }

    public enum DeliveryType
    {
        ReportServerEmail = 0,
        WindowFileShare = 1
    }

    public enum ScheduleType
    {
        Once = 0,
        Hourly = 1,
        Daily = 2,
        DailyDOW = 3,
        DailyWeekDays = 4,
        WeeklyDOW = 5,
        MonthlyDays = 6,
        MonthlyDOW = 7
    }


    public class SCHDefination
    {
        public int Hours;
        public int Minutes;
        public int DaysInterval;
        public int WeekInterval;
        public string MonthCalendarDays;
        public Dictionary<WeekDays, bool> DayOfWeeksSelector;
        public Dictionary<Months, bool> MonthSelector;
        public string WeekOfMonth;

        public int StartTimeHours;
        public int StartTimeMinutes;

        public DateTime StartDate;
        public DateTime EndDate;

        public ScheduleType scheduleType;

        public SCHDefination()
        {
            DayOfWeeksSelector = new Dictionary<WeekDays, bool>();
            MonthSelector = new Dictionary<Months, bool>();
        }

        public DateTime StartDateTime()
        {
            return ChangeTime(StartDate, StartTimeHours, StartTimeMinutes, 0, 0);
        }

        public int MinuteInterval()
        {
            return (Hours*60) + Minutes;
        }

        private DateTime ChangeTime(DateTime dateTime, int hours, int minutes, int seconds, int milliseconds)
        {
            return new DateTime(
                dateTime.Year,
                dateTime.Month,
                dateTime.Day,
                hours,
                minutes,
                seconds,
                milliseconds,
                dateTime.Kind);
        }
    }

    public enum WeekDays
    {
        Sunday,
        Monday,
        Tuesday,
        Wednesday,
        Thursday,
        Friday,
        Saturday
    }

    public enum Months

        March,
        April,

    }
}

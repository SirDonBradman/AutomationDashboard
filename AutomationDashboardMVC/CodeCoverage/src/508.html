using System;
using System.Web;
using Aurigo.AMP3.Common;
using Aurigo.AMP3.Core.AbstractModels;
using Aurigo.Brix.Platform.CoreUtilities.Utility;

namespace Aurigo.AMP3.LandMangement.UI
{
    // Shud the table be filled?
    [ListModelContext(Name = "LANDMGT", Table = "")]
    public class LandManagementCheckListAttachmentModel : ChecklistAttachmentModel
    {
        private readonly string enterpriseLandMgmt;
        private readonly string landMgmtID;
        private readonly HttpRequest request;

        public LandManagementCheckListAttachmentModel()
        {
            landMgmtID = String.IsNullOrEmpty(request["LMID"])
                             ? String.Empty
                             : (request["LMID"].ToInt32_2().ToString2());
            enterpriseLandMgmt = String.IsNullOrEmpty(request["ENTPRSE_LM"])
                                     ? String.Empty
                                     : request["ENTPRSE_LM"].ToString2();
        }


        public override string ModuleSpecificQueryString()
        {
            string queryString = String.Empty;

            queryString = "&LMID=" + landMgmtID;
            queryString += (enterpriseLandMgmt.Length > 0) ? "&ENTPRSE_LM=Y" : String.Empty;
            queryString += "&tab=PD_TabDetails";


            return queryString;
        }
    }
}

using System.Data;
using Aurigo.AMP3.LandMangement.BusinessLayer.DTO;

namespace Aurigo.AMP3.LandMangement.BusinessLayer.BL
{
    public partial class LandMangementManager
    {
        public DataTable GetParcel(int parcelID)
        {
            return LandMangementComponent.Instance.GetParcel(parcelID);
        }

        public int SaveParcel(Mode mode, LandMangementParcelDTO dto)
        {
            return LandMangementComponent.Instance.SaveParcel(mode.ToString(), dto);
        }

        public int SaveParcelChecklist(Mode mode, LandMangementParcelChecklistDTO dto)
        {
            return LandMangementComponent.Instance.SaveParcelChecklist(mode.ToString(), dto);
        }

        public int SaveMainChecklist(string mode, int parcelID, string xmlList)
        {
            return LandMangementComponent.Instance.SaveMainChecklist(mode, parcelID, xmlList);
        }

        public DataTable GetMainCheckList(int parcelId)
        {
            return LandMangementComponent.Instance.GetMainCheckList(parcelId);
        }

        public DataTable GetReportData(int parcelId, string reportType, int projectID)
        {
            return LandMangementComponent.Instance.GetReportData(parcelId, reportType, projectID);
        }

        public DataSet CUDMediator(int parcelID, int mediatorID, string mediator, decimal conversionRate,
                                   int? conversionFactorType, string OP)
        {
            return LandMangementComponent.Instance.CUDMediator(parcelID, mediatorID, mediator, conversionRate,
                                                               conversionFactorType, OP);
        }
        public DataTable GetLandDetails(int LMID)
        {
            return LandMangementComponent.Instance.GetLandDetails(LMID);
        }
    }
}

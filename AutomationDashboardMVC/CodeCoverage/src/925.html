using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;

namespace Aurigo.Brix.Platform.BusinessLayer.XMLForm
{
    public enum AttributeType
    {
        Common = 1,
        Advanced = -1
    }

    public class FormDesignerAttributeAttribute : Attribute
    {
        public ControlType [] ReleventControls { get; set; }
        public AttributeType AttributeType { get; set; }
        public bool HideAttribute { get; set; }
        public string Caption { get; set; }
        public string ToolTip { get; set; }
        public int Priority { get; set; }

        public bool IsRelevent(ControlContainer c)
        {
            foreach(var a in ReleventControls)
            {
                if (a == ControlType.All)
                    return true;

                if (c is xControl)
                {
                    if (a == ((c as xControl).Type))
                        return true;
                }
            }

            return false;
        }

        public FormDesignerAttributeAttribute()
        {
            ReleventControls = new[] { ControlType.All };
            AttributeType = AttributeType.Common;
            HideAttribute = false;
            Caption = string.Empty;
            ToolTip = string.Empty;
            Priority = 10;
        }
    }
}

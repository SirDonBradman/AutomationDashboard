using System;
using System.Collections.Generic;
using System.Xml.Serialization;

namespace Aurigo.Brix.Platform.BusinessLayer.XMLForm
{
    public enum LinkedInstanceType
    {
        Referred,
        Chained
    }

    public class LinkedInstance
    {
        [XmlIgnore] public xControl ReferencedBy;

        public string DestinationModuleID;

        [XmlElement("Notes")] public string CustomMessage;

        public List<string> DestinationInstanceIds = new List<string>();

        public string DestinationData;
    }

    [Serializable]
    [XmlRoot("LinkedInstanceDetails")]
    public class LinkedInstanceDetails
    {
        public string SourceModuleID;

        public string SourceInstanceId;

        public string SourceData;

        public string SourceUrl;

        public LinkedInstanceType InstanceType;

        public List<LinkedInstance> LinkedDestinationInstances;
    }

    public class ReferenceModuleColumnData
    {
        public string ColumnName;

        public string ColumnValue;
    }
}

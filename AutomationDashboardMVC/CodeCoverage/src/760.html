using System.Collections.Generic;
using System.Data;
using Aurigo.AMP3.DataAccess.Core;
using Aurigo.AMP3.DocumentManagementDAC;
using Aurigo.Common.Utility;

namespace Aurigo.AMP3.DocumentManagementBL
{
    public class TransmittalManager : SingletonManagerBase<TransmittalManager>
    {
        private TransmittalManager() { }

        public DataTable GetDraftedTransmittals(int projectID, int usrID)
        {
            return TransmittalComponent.Instance.GetDraftedTransmittals(projectID, usrID);
        }

        public void TransmitDocuments(int transmittalID, int userID)
        {
            TransmittalComponent.Instance.TransmitDocuments(transmittalID, userID);
        }

        public void UpdateTransmittalNumber(int transmittalID)
        {
            TransmittalComponent.Instance.UpdateTransmittalNumber(transmittalID);
        }
    }
}

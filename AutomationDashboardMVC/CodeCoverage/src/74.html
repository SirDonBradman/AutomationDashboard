using Aurigo.AMP3.Common;
using Aurigo.AMP3.ContmgtDTO;
using Aurigo.AMP3.ContractManager.BusinessLayer;

namespace Aurigo.AMP3.ContractManagementUI
{
    [Context(Name = "CONTMGT")]
    public class ItemTotalModel : Core.AbstractModels.ItemTotalModel
    {
        public override string Header
        {
            get { return "Contract Total"; }
        }

        public override bool GetApprovalStatus(int contractID, string moduleID)
        {
            DTO dtoContract = BL.Instance.GetContract(contractID, FetchSet.Default);

            return (dtoContract.Locked == "Y") ? true : false;
            //return false;
        }
    }
}

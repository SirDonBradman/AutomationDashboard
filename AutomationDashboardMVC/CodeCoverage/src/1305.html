using Aurigo.AMP3.Core.UserControls;
using Aurigo.Brix.Core.WebUiHelper;

namespace Aurigo.AMP3.EnterpriseUI
{
    public partial class Enterprise1 : BrixPageBase
    {
        protected Dashboard DBControl;

        protected override void OnPreRender(EventArgs e)
        {
            UIHelper.CreateDashboadTabs(1, PageTabBar, DBControl.Layout.pages, DBControl.CurrentPage);
            base.OnPreRender(e);
        }

        protected override void OnInit(EventArgs e)
        {
            PermissionsToCheck.Add(Constants.MODFEAT_VISIBILITY);

            DBControl.SettingName = "Layout_Enterprise";
            DBControl.IsMultiplePages = true;
            DBControl.CurrentPage = 2;
            base.OnInit(e);
        }
    }

using System;
using System.Data.Linq.Mapping;

namespace Aurigo.Brix.Platform.CoreUtilities.DTO
{
    [Table(Name = "Workflows")]
    public class Workflows
    {
        [Column(Name = "WorkflowID", DbType = "Int", IsDbGenerated = true, IsPrimaryKey = true)]
        public int WorkflowID { get; set; }

        [Column(Name = "InstanceID", DbType = "Int")]
        public int InstanceID { get; set; }

        [Column(Name = "State", DbType = "Varchar(15)")]
        public string State { get; set; }

        [Column(Name = "WorkflowInstanceID", DbType = "uniqueidentifier")]
        public Guid WorkflowInstanceID { get; set; }

        [Column(Name = "ModuleID", DbType = "Varchar(15)")]
        public string ModuleID { get; set; }
    }
}

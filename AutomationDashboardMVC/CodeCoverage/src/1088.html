using System;
using System.Collections.Generic;
using System.Linq;
using System.Net;
using System.Net.Http;
using System.Web;
using System.Web.Http;

namespace Aurigo.Brix.Platform.BusinessLayer.Controller
{
    public class CommonController : ApiController
    {
        /// <summary>
        /// This method is used for getting Guid to use in Client Side
        /// </summary>
        /// <returns></returns>
        //[HttpGet]
        public HttpResponseMessage GetServerGuidAsText()
        {
            var response = new HttpResponseMessage(HttpStatusCode.OK);
            response.Content = new StringContent(Guid.NewGuid().ToString(), System.Text.Encoding.UTF8, "text/plain");
            return response;
        }
    }
}

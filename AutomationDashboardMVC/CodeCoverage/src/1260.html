using System.Collections.Generic;
using Aurigo.Brix.Core.WebUiHelper;
using UC = Aurigo.AMP3.Core.UserControls;
using Aurigo.AMP3.Common;
using Aurigo.Brix.Platform.BusinessLayer;

namespace Aurigo.AMP3.ReportsUI
{
    public partial class Settings : BrixPageBase
    {
        protected override void OnPreInit(System.EventArgs e)
        {
            base.OnPreInit(e);
            CustomAdHocConfig.InitializeReporting();
        }

        protected override void OnLoad(System.EventArgs e)
        {
            Page.Title = "Report Settings";
        }

        protected override void OnInit(System.EventArgs e)
        {
            ModuleId = Constants.MODID_ENTPRSE;
            PermissionsToCheck.Add(Constants.MODFEAT_VISIBILITY);
            PermissionsToCheck.Add(Constants.MODFEAT_MYREPORTS);

            List<UC.MenuGroup> menuGroups = new List<UC.MenuGroup>();
            UC.MenuGroup generalGroup = new UC.MenuGroup("General");
            generalGroup.AddMenu(new UC.LargeButton("lnkNew", "New", "Icn_New.png"));
            generalGroup.AddMenu(new UC.LargeButton("lnkReportList", "ReportList", "Icn_Details.png"));
            generalGroup.AddMenu(new UC.LargeButton("lnkSave", "Save", "Icn_Save.png"));
            generalGroup.AddMenu(new UC.LargeButton("lnkPreview", "Preview", "Icn_View.png"));
            generalGroup.AddMenu(new UC.TextIcon("lnkUpdate", "Update Schema", "Icn_Sql_16.png"));
            generalGroup.AddMenu(new UC.TextIcon("lnkGenCode", "Generate C# Code", "Icn_xml_16.png"));
            menuGroups.Add(generalGroup);
            CreateToolBarMenu(menuGroups, null);
        }

        protected override void CustomizeToolBar()
        {
            if (MainToolBar.GetMenuReference("lnkNew") != null)
            {
                MainToolBar.GetMenuReference("lnkNew").CausesValidation = false;
                MainToolBar.GetMenuReference("lnkNew").OnClientClick =
                    "  ShowPopup('ReportDesignerButton'); return false;";
            }
            if (MainToolBar.GetMenuReference("lnkReportList") != null)
            {
                MainToolBar.GetMenuReference("lnkReportList").CausesValidation = false;
                MainToolBar.GetMenuReference("lnkReportList").OnClientClick =
                    "  ShowPopup('ReportListButton'); return false;";
            }
            if (MainToolBar.GetMenuReference("lnkSave") != null)
            {
                MainToolBar.GetMenuReference("lnkSave").CausesValidation = false;
                MainToolBar.GetMenuReference("lnkSave").OnClientClick = "  ShowPopup('SaveIzendaConfig'); return false;";
            }
            if (MainToolBar.GetMenuReference("lnkPreview") != null)
            {
                MainToolBar.GetMenuReference("lnkPreview").CausesValidation = false;
                MainToolBar.GetMenuReference("lnkPreview").OnClientClick = "  ShowPopup('Preview'); return false;";
            }
            if (MainToolBar.GetMenuReference("lnkUpdate") != null)
            {
                MainToolBar.GetMenuReference("lnkUpdate").CausesValidation = false;
                MainToolBar.GetMenuReference("lnkUpdate").OnClientClick = "  ShowPopup('UpdateSchema'); return false;";
            }
            if (MainToolBar.GetMenuReference("lnkGenCode") != null)
            {
                MainToolBar.GetMenuReference("lnkGenCode").CausesValidation = false;
                MainToolBar.GetMenuReference("lnkGenCode").OnClientClick =
                    "  ShowPopup('GenerateCshCode'); return false;";
            }
        }
    }
}

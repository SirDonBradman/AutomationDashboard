using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using Aurigo.Brix.Construction.BudgetManager.BusinessLayer.BL;
using Aurigo.Brix.Platform.BusinessLayer.DataResouces;
using Aurigo.Common;

namespace Aurigo.Brix.Construction.PlanningManagement.BL
{
    [Serializable]
    [CustomResourceType("Forms", "BDGTREV")]
    public class WFAUpdateBudgetRevisionStatusToCompleted : FormsCustomResource
    {
        public WFAUpdateBudgetRevisionStatusToCompleted()
        {
            _Namespace = "Forms";
            _Path = "Forms.PLANING.WFActivities.WFAUpdateBudgetRevisionStatusToCompleted";
            _Name = "Update Status to Completed";
            _Desc = "Updates the status to completed";
            //_InParameters = new[] { "System.String" };
            _OutParameters = new[] { "IsSuccessful,System.Boolean", "Message,System.String" };
        }

        public override void ExecuteDerived()
        {
            BudgetManagementManager.Instance.UpdateRevisionStatus(int.Parse(InstanceId), "Completed");

            SetOutParam("IsSuccessful", true, "System.Boolean");
            SetOutParam("Message", string.Empty, "System.String");
        }
    }
}

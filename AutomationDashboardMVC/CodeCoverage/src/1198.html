using System;
using System.Collections.Generic;
using System.Web.UI;
using Aurigo.AMP3.Common;

namespace Aurigo.Brix.Platform.BusinessLayer.AbstractModels
{
    public abstract class AfterSaveRecordModel
    {
        private static readonly Dictionary<string, Type> contextDictionary;

        static AfterSaveRecordModel()
        {
            if (contextDictionary == null)
                contextDictionary = ModelHelper.GetModelObjects(typeof (AfterSaveRecordModel));
        }

        public static AfterSaveRecordModel GetInstance(string context)
        {
            if (!contextDictionary.ContainsKey(context) || contextDictionary[context] == null)
                return null;
            return (AfterSaveRecordModel)Activator.CreateInstance(contextDictionary[context]);
        }

        public static Dictionary<string, Type> GetModelObjects()
        {
            return contextDictionary;
        }

        public abstract void SaveDetails(object recordDetails, Page page, string mode, string callerContext,
            object additionalParameters = null);
    }
}

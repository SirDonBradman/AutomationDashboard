using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using Aurigo.Brix.Construction.BudgetManager.BusinessLayer.BL;
using Aurigo.Brix.Platform.BusinessLayer.DataResouces;
using Aurigo.Common;

namespace Aurigo.Brix.Construction.BudgetManager.BusinessLayer.WFActivities
{
    [Serializable]
    [CustomResourceType("Forms", "BDGTEST")]
    public class WFABudgetEstimateUpdateCurrentBudget : FormsCustomResource
    {
        public WFABudgetEstimateUpdateCurrentBudget()
        {
            _Namespace = "Forms";
            _Path = "Forms.PLANING.WFActivities.WFAUpdateBudgetEstimateStatusToApproved";
            _Name = "Update Current Budget for Budget Estimate";
            _Desc = "Update Current Budget for Budget Estimate";
            //_InParameters = new[] { "Complete Action ID,System.String", "Submit Action ID,System.String", "Approve Action ID,System.String" };
            _OutParameters = new[] { "IsSuccessful,System.Boolean", "Message,System.String" };
        }
        public override void ExecuteDerived()
        {
            try
            {
                string message = string.Empty;
                int currentuserid = int.Parse(GetInputParam("_currentuserid").ToString());

                int res = BudgetManagementManager.Instance.CopyBudgetEstimateToCurrentBudget(int.Parse(InstanceId), currentuserid, out message);

                if (res < 0 && !string.IsNullOrEmpty(message))
                    throw new Exception(message);

                SetOutParam("IsSuccessful", true, "System.Boolean");
                SetOutParam("Message", string.Empty, "System.String");
            }
            catch(Exception ex)
            {
                SetOutParam("IsSuccessful", false, "System.Boolean");
                SetOutParam("Message", ex.Message, "System.String");
            }
        }
    }
}

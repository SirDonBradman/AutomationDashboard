using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using System.Xml.Serialization;

namespace Aurigo.Brix.Platform.BusinessLayer.XMLForm
{
    /// <summary>
    /// QR code Generator Container class holds the properties of QRGenerator container in the XML
    /// </summary>
    public class QRGenerator
    {
        [XmlElement("QRCodeControl")]
        public List<QRCodeControl> QRControl = new List<QRCodeControl>();
       
        [XmlIgnore]
        public string Prefix { get; set; }

        [XmlIgnore]
        public int NoOfColumns { get; set; }
        public string QRColumns
        {
            get
            {
                List<string> colList = new List<string>();
                foreach (var item in QRControl)
                {
                    string[] qrCol = item.QRColumns.Split(',');
                    foreach (var col in qrCol)
                    {
                        if (!colList.Contains(col))
                            colList.Add(col);
                    }

                }
                foreach (var item in QRControl)
                {
                    string[] qrCol = item.QRLabels.Split(',');
                    foreach (var col in qrCol)
                    {
                        if (!colList.Contains(col))
                            colList.Add(col);
                    }

                }
                return String.Join(",", colList);
            }
        }
        public QRGenerator()
        {
            NoOfColumns = 2;
            Prefix = "AUR_QR_";
        }
    }
}

using System;
using System.Collections.Generic;
using System.ComponentModel;
using System.Linq;
using System.Web;
using System.Xml.Serialization;

namespace Aurigo.Brix.Platform.BusinessLayer.XMLForm
{
    /// <summary>
    /// This class is just for logical separation of the Form - these stages may be useful while defining the workflow.
    /// </summary>
    public class Stage : ControlContainer
    {
        [XmlAttribute("SetPermission"), DefaultValue(false)]
        public bool SetPermission { get; set; }

        public override void Render(object parent, RenderFormat format)
        {
            // get the delegate that actuall renders the control from the registered renderers.  For now we hard
            // code htmlrender since reports dont work through this model yet.

            if (form.Renderer.ContainerRenderers[(int)ContainerTypes.Stage] != null)
                form.Renderer.ContainerRenderers[(int)ContainerTypes.Stage](this, parent);
        }
    }
}

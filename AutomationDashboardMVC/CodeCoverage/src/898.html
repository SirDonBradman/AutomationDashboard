using System;
using System.Xml.Serialization;

namespace Aurigo.Brix.Platform.BusinessLayer.XmlForm_Framework
{
    [Serializable]
    public class DataPicker
    {
        [XmlAttribute("Name")]
        public string Name { get; set; }

        [XmlAttribute("DataSourceType")]
        public DataSourceType DataSourceType { get; set; }

        [XmlAttribute("DataSource")]
        public string DataSource { get; set; }

        [XmlAttribute("PageSize")]
        public string PageSize { get; set; }

        [XmlAttribute("IdColumn")]
        public string IdColumn { get; set; }

        [XmlAttribute("DisplayColumn")]
        public string DisplayColumn { get; set; }

        [XmlElement("PickerColumn")]
        public PickerColumn[] PickerColumn { get; set; }
    }

    [Serializable]
    public class PickerColumn
    {
        [XmlAttribute("Name")]
        public string Name { get; set; }

        [XmlAttribute("Hide")]
        public bool Hide { get; set; }

        [XmlAttribute("ColumnHeader")]
        public string ColumnHeader { get; set; }

        [XmlAttribute("Filter")]
        public bool Filter { get; set; }

        [XmlAttribute("Width")]
        public string Width { get; set; }

        [XmlAttribute("FilterDataSourceType")]
        public DataSourceType FilterDataSourceType { get; set; }

        [XmlAttribute("FilterDataSource")]
        public string FilterDataSource { get; set; }

        [XmlAttribute("FilterType")]
        public FilterType FilterType { get; set; }
    }

    public enum DataSourceType
    {
        Table,
        View,
        Query,
        StoreProcedure,
        Expression
    }

    public enum FilterType
    {
        TextBox,
        DropDownList,
        CheckBox,
        RadioButton,
        Hidden
    }
}

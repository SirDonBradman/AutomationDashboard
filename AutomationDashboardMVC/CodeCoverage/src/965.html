using System.Collections.Generic;
using System.IO;
using System.Xml.Serialization;
using Aurigo.AMP3.DataAccess.Core;
using Aurigo.Common.Utility;

namespace Aurigo.Brix.Platform.BusinessLayer.ChartBuilder
{
    public class ChartModel : SingletonManagerBase<ChartModel>
    {
        private ChartModel()
        {
        }

        public FileStream ReadXML(string path)
        {
            return File.OpenRead(path);
        }

        public Chart GetChartModel(string path)
        {
            FileStream fs = File.OpenRead(path);
            return GetChartModel(fs);
        }

        public Chart GetChartModel(FileStream xmlFileStream)
        {
            if (xmlFileStream != null)
            {
                XmlSerializer serializer = new XmlSerializer(typeof (Chart));
                return ((Chart)serializer.Deserialize(xmlFileStream));
            }
            else return null;
        }
    }
}

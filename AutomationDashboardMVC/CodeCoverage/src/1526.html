using Aurigo.Common;

namespace Aurigo.Workflow
{
    public abstract class WFCustomResource : BaseCustomResource
    {
        protected IDataSource _ParentDataSource;

        public override IDataSource ParentDataSource
        {
            get { return _ParentDataSource; }
        }

        public override IResourceInstance GetNewResourceInstance()
        {
            var ri = new WFResourceInstance();
            ri.InitNew(this, true);
            ri.DataSourceId = _TemplateId + "_" + _InstanceId + "_" + _Namespace + "_" + _Path;
            return ri;
        }
    }
}

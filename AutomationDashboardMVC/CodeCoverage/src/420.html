using System.Collections.Generic;
using System.Web;
using Aurigo.AMP3.Common;
using Aurigo.AMP3.Core.AbstractModels;
using Aurigo.AMP3.Core.UserControls;
using Aurigo.AMP3.ModuleManagementBL;
using Aurigo.Brix.Platform.CoreUtilities.Utility;

namespace Aurigo.AMP3.EstimatorUI
{
    [Context(Name = "ESTMATE")]
    public class EstimateChecklistModel : ChecklistModel
    {
        //Need not override CustomMenu() as the default definition
        //is required for this model.

        private int instanceID;

        public override List<string> CustomMainToolBar(ToolBar MainToolBar, HttpContext Context, HttpRequest Request)
        {
            return ModuleManager.Instance.GetPermissionByUserOrRole(Constants.MODID_ESTMATE, UserDetail.GetCurrentUserDetail().UID,
                                                              UserDetail.GetCurrentUserDetail().RID, Request.QueryString["PID"].ToInt32_2());
        }

        public override int GetProjectIdFromInstanceId(int parentInstanceID)
        {
            int pid = 0;
            if (!string.IsNullOrEmpty(Request.QueryString["PID"]))
                pid = int.Parse(Request.QueryString["PID"]);

            return pid;
        }

        public override int GetInstanceID()
        {
            instanceID = Request["ParentID"].ToInt32_2();
            return instanceID;
        }

        public override string ModuleSpecificQueryString()
        {
            // By default return empty string, the derived
            // classes in construction vertical will have
            // concrete imp for this method.
            return ("&ParentID=" + instanceID.ToString2());
        }
    }
}

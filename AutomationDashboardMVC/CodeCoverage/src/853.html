using System.Collections.Generic;
using System.Web.UI.WebControls;
using System.Xml.Serialization;
using Aurigo.Brix.Core.WebUiHelper;

namespace Aurigo.AMP3.Core.AbstractModels
{
    public class Column
    {
        [XmlAttribute("Caption")] public string Caption;
        [XmlElement("Column")] public List<Column> ChildColumns = new List<Column>();
        [XmlAttribute("ClientSupport")] public string ClientSupport;
        [XmlAttribute("DataType")] public string DataType;
        [XmlElement("EditProperties")] public PropertyList EditProperties = new PropertyList();
        [XmlElement("ListProperties")] public PropertyList ListProperties = new PropertyList();
        [XmlAttribute("Name")] public string Name;
        [XmlAttribute("RepeatDirection")] public RepeatDirection RepeatDirection;
        [XmlAttribute("UserControlSrc")] public string Src;
        [XmlAttribute("Style")] public string Style;
        [XmlAttribute("Target")] public string Target;
        [XmlAttribute("Type")] public ControlType Type;
        [XmlAttribute("Validation")] public string Validation;

        [XmlElement("Validator")] public List<Validator> Validators = new List<Validator>();
    }

    public enum ControlType
    {
        Label,
        Picture,
        TextBox,
        TextArea,
        Button,
        Combobox,
        Listbox,
        Grid,
        Radiobox,
        Checkbox,
        Date,
        HierarchicalSelector,
        UserControl,
        WebNumeric,
        HiddenField,
        ColumnGroup,
        LinkButton,
        InsertHide,
        TimeInput,
        Integer,
        Column,
        FileUpload,
        None
    }

    public enum ModeTypes
    {
        New,
        Edit,
        View
    };
}

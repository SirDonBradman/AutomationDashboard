using System;
using System.Globalization;
using System.Xml;
using Aurigo.AMP3.Common;
using Aurigo.Brix.Platform.CoreUtilities.Utility;
using Aurigo.Brix.Platform.BusinessLayer.Utility;

namespace Aurigo.AMP3.ContmgtDTO
{
    public class GenericFunctions
    {
        public static DateTime? ToDateTime(XmlNode node)
        {
            return isNull(node) ? (DateTime?) null : node.InnerText.ToDateTime_MWCulture();
        }

        public static Decimal? ToDecimal(XmlNode node)
        {
            return isNull(node) ? (Decimal?) null : node.InnerText.ToDecimal2();
        }

        public static int? ToInt(XmlNode node)
        {
            return isNull(node) ? (int?) null : node.InnerText.ToInt32_2();
        }

        public static string ToString(XmlNode node)
        {
            return isNull(node) ? null : UIHelper.RevertXMLCharEntities(node.InnerText);
        }

        public static char ToChar(XmlNode node)
        {
            return Convert.ToChar(node.InnerText, CultureInfo.InvariantCulture);
        }

        public static bool isNull(XmlNode node)
        {
            return node == null ? true : String.IsNullOrEmpty(node.InnerText) ? true : false;
        }
    }
}

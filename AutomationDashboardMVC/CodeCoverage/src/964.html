    [Serializable]
    public class Chart : ControlContainer
    {
        [XmlAttribute("BackColor")]
        public string BackColor { get; set; }

        [XmlAttribute("Title")]
        public string Title { get; set; }

        [XmlAttribute("Palette")]
        public ChartColorPalette Palette { get; set; }

        [XmlAttribute("DataSource")]
        public string DataSource { get; set; }

        [XmlAttribute("DataSourceType")]
        public DataSourceType DataSourceType { get; set; }

        [XmlAttribute("Parameters")]
        public string Parameters { get; set; }

        [XmlElement("ChartArea")] public List<ChartArea> ChartAreas = new List<ChartArea>();

        public override void Render(object parent, RenderFormat format)
        {
            if (form.Renderer.ContainerRenderers[(int)ContainerTypes.Chart] != null)
                form.Renderer.ContainerRenderers[(int)ContainerTypes.Chart](this, parent);
        }
    }

    [Serializable]
    public class ChartArea : ControlContainer
    {
        //[XmlAttribute("Name")]
        //public string Name { get; set; }
        [XmlAttribute("Enable3D")]
        public bool Enable3D { get; set; }

        [XmlAttribute("Visible")] public bool Visible = true;

        [XmlAttribute("BackColor")]
        public string BackColor { get; set; }

        [XmlAttribute("BackGradientStyle")]
        public GradientStyle BackGradientStyle { get; set; }

        [XmlAttribute("BorderColor")]
        public string BorderColor { get; set; }

        [XmlAttribute("BorderStyle")]
        public ChartDashStyle BorderStyle { get; set; }

        [XmlAttribute("BorderWith")]
        public int BorderWith { get; set; }

        [XmlAttribute("ShadowOffset")]
        public int ShadowOffset { get; set; }

        [XmlElement("Series")] public List<Series> Series = new List<Series>();
        [XmlElement("XAxis")] public XAxis XAxis;
        [XmlElement("YAxis")] public YAxis YAxis;
        [XmlElement("Legend")] public List<Legends> Legends = new List<Legends>();
    }

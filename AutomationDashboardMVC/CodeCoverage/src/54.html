using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using Aurigo.Brix.Platform.BusinessLayer.AbstractModels;
using Aurigo.DataBinding;
using Aurigo.AMP3.Common;
using Aurigo.Brix.Platform.BusinessLayer.ConcreteModels.Utility;

namespace Aurigo.Brix.Construction.ContractManager.BusinessLayer.ConcreateModels
{
    [ContextAttribute(Name = "CONTMGT")]
    public class ContractCalendarCalculationsModel : CalendarCalculationsModel
    {
        public override decimal GetYearWiseValue(CalendarDetails calendar, TaskInfo task, Decimal value, decimal totDays, DateTime planYear)
        {
            return new SQLData().GetYearWiseValue(calendar, task, value, totDays, planYear);
        }

        public override CalendarDetails GetCalendarDetails(int parentID)
        {
            return AMP3.ContractManager.BusinessLayer.BL.Instance.GetCalendarDetailsFromContract(parentID);
        }
    }
}

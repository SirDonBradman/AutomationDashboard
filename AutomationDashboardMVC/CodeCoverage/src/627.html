using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using Aurigo.Brix.Construction.BudgetManager.BusinessLayer.BL;
using Aurigo.Brix.Platform.BusinessLayer.DataResouces;
using Aurigo.Common;

namespace Aurigo.Brix.Construction.PlanningManagement.BL
{
    [Serializable]
    [CustomResourceType("Forms", "BDGTREV")]
    public class WFAUpdateBudgetRevisionStatusToDraft : FormsCustomResource
    {
        public WFAUpdateBudgetRevisionStatusToDraft()
        {
            _Namespace = "Forms";
            _Path = "Forms.PLANING.WFActivities.WFAUpdateBudgetRevisionStatusToDraft";
            _Name = "Update Status to Draft";
            _Desc = "Updates the status to draft";
            //_InParameters = new[] { "System.String" };
            _OutParameters = new[] { "IsSuccessful,System.Boolean", "Message,System.String" };
        }
        public override void ExecuteDerived()
        {
            BudgetManagementManager.Instance.UpdateRevisionStatus(int.Parse(InstanceId), "Draft");

            SetOutParam("IsSuccessful", true, "System.Boolean");
            SetOutParam("Message", string.Empty, "System.String");
        }
    }
}

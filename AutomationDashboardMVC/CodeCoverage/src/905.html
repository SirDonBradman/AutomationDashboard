using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Web.UI.WebControls;
using Infragistics.WebUI.WebSchedule;
using Aurigo.Brix.Platform.CoreUtilities.Utility;
using Aurigo.AMP3.Common;
using Infragistics.WebUI.WebDataInput;
using Aurigo.Brix.Platform.BusinessLayer.CustomControls;
using System.Web.UI;

namespace Aurigo.Brix.Platform.BusinessLayer.XmlForm_Framework
{
    public class XMLDropDownList : DropDownList, IControl
    {
        public bool RenderAsLabel { get; set; }

        protected override void Render(System.Web.UI.HtmlTextWriter writer)
        {
            if (RenderAsLabel)
            {
                string text = this.SelectedItem == null ? "" : this.SelectedItem.Text;
                Helper.RenderAsLabel(this, writer, text, Width, Height, attributes: Style.Value);
                string value = this.SelectedValue == null ? "" : this.SelectedValue;
                Helper.RenderAsLabel(this, writer, value, Width, Height, hide: true, attributes: Style.Value);
                base.Style.Add("display", "none");
                base.Render(writer);
            }
            else
            {
                base.Render(writer);
            }
        }
    }
}

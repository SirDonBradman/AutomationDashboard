using System;
using System.Collections.Generic;
using Aurigo.AMP3.Core.UserControls;
using Aurigo.Brix.Platform.BusinessLayer.DTO;

namespace Aurigo.AMP3.ModuleManagementDTO
{
    public class ModuleDetails
    {
        //added for search
        // added recently to form the tree
        //Added for Uploadable Reports

        public ModuleDetails()
        {
            ModuleID = string.Empty;
            Name = string.Empty;
            Description = string.Empty;
            Version = string.Empty;
            NavigateUrl = string.Empty;
            IsActive = true;
            UploadedDate = DateTime.UtcNow;
            UploadedBy = string.Empty;
            IsCoreModule = false;
            ModuleType = string.Empty;
            ParentModule = string.Empty;
            ParentModuleID = 0;
            SiteMapXml = string.Empty;
            dllFiles = new List<string>();
            Features = new Dictionary<string, string>();
            InstanceUrl = string.Empty;
            IsSearchable = false;
            MainDll = string.Empty;
            Folder = string.Empty;
            ModuleGroup = string.Empty;
            CoreVersion = string.Empty;
            ConfigUrl = string.Empty;
            ReportLogo = string.Empty;
            BLClassName = string.Empty;
            IsSingleInstance = false;
            IsMapEnabled = false;
            Reports = new List<ReportDetails>();
        }

        public string ModuleID { get; set; }

        public string Name { get; set; }

        public string Description { get; set; }

        public string Version { get; set; }

        public string NavigateUrl { get; set; }

        public bool IsActive { get; set; }

        public DateTime UploadedDate { get; set; }

        public string UploadedBy { get; set; }

        public bool IsCoreModule { get; set; }

        public string ModuleType { get; set; }

        public string ModuleGroup { get; set; }

        public string ParentModule { get; set; }

        public long ParentModuleID { get; set; }

        public string SiteMapXml { get; set; }

        public List<string> dllFiles { get; set; }

        public Dictionary<string, string> Features { get; set; }

        public string InstanceUrl { get; set; }

        public bool IsSearchable { get; set; }

        public string MainDll { get; set; }

        public string Folder { get; set; }

        public bool HasSubTree { get; set; }

        public string CoreVersion { get; set; }

        public string ConfigUrl { get; set; }

        public string ReportLogo { get; set; }

        public string BLClassName { get; set; }

        public List<ReportDetails> Reports { get; set; }

        public string AssociatedModuleID { get; set; }

        public string AssociatedContext { get; set; }

        public bool IsSingleInstance { get; set; }

        public bool IsMapEnabled { get; set; }

        public List<BrixContextMenu> ContextMenus { get; set; }
    }

    public enum ModuleGroup
    {
        Forms = 0,
        Reports = 1,
        Unknown = 9
    }
}

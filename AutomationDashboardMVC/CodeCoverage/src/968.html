using System.Collections.Generic;
using System.IO;

namespace Aurigo.Brix.Platform.BusinessLayer.FileParser
{
    public class DWGParseResult
    {
        public DWGParseResult()
        {
            Xrefs = new List<AutoCadDrawingXRef>();
        }

        public string DocumentName
        {
            get { return Path.GetFileName(DocumentPath); }
        }

        public string DocumentPath { get; set; }

        public List<AutoCadDrawingXRef> Xrefs { get; set; }

        public int DocumentID { get; set; }

        public void AddXref(string documentPath)
        {
            AutoCadDrawingXRef xref = new AutoCadDrawingXRef();
            xref.DocumentPath = documentPath;
            Xrefs.Add(xref);
        }
    }

    public class AutoCadDrawingXRef
    {
        public string DocumentName
        {
            get { return Path.GetFileName(DocumentPath); }
        }

        public string DocumentPath { get; set; }

        public bool IsRelativePath
        {
            get { return !Path.IsPathRooted(DocumentPath); }
        }
    }
}

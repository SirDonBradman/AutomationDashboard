using System;
using System.Collections.Generic;
using Aurigo.AMP3.Common;

namespace Aurigo.AMP3.Core.AbstractModels
{
    public abstract class MobileCientSyncModel
    {
        private static readonly Dictionary<string, Type> contextDictionary;

        static MobileCientSyncModel()
        {
            if (contextDictionary == null)
                contextDictionary = ModelHelper.GetModelObjects(typeof (MobileCientSyncModel));
        }

        public static MobileCientSyncModel GetInstance(string context)
        {
            if (!contextDictionary.ContainsKey(context) || contextDictionary[context] == null)
                return null;
            return (MobileCientSyncModel)Activator.CreateInstance(contextDictionary[context]);
        }

        public static Dictionary<string, Type> GetModelObjects()
        {
            return contextDictionary;
        }

        public abstract SyncStatus SubmitXmlData(string xmlData);
    }
}

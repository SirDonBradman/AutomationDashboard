using System.Web;
using Aurigo.Common.Utility;

#pragma warning disable 1591

namespace Aurigo.Common
{
    public class BrixErrorInfo : AppErrorInfo
    {
        public BrixErrorInfo(string moduleID, string message, string currentUser) : base(moduleID, message, currentUser)
        {
            if (!string.IsNullOrEmpty(this.currentUser)) return;
            var userDetails = (IUserDetails)HttpContext.Current.Session["UserDetail"];
            if (null == userDetails) return;
            this.currentUser = userDetails.GetUserName();
        }
    }
}

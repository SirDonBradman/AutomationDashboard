using System;
using System.Data;
using System.Web.UI;
using Aurigo.AMP3.UserManagementBL;

namespace Aurigo.AMP3.EnterpriseUI
{
    public partial class UsersSummary : UserControl
    {
        protected void Page_Load(object sender, EventArgs e)
        {
            DataTable dt = UserManager.Instance.GetUserSummary();

            if (dt.Rows.Count > 0)
            {
                lblErr.Text = "";
                ChartUserSummary.Data.DataSource = dt;
                ChartUserSummary.Data.DataBind();

                ChartUserSummary.Tooltips.FormatString = "User Count: <DATA_VALUE>";

                ChartUserSummary.Visible = true;
            }
            else
            {
                lblErr.Text = "No users available.";
                ChartUserSummary.Visible = false;
            }
        }
    }
}

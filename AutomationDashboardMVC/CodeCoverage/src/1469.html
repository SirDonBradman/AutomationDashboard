using System;

#pragma warning disable 1591

namespace Aurigo.Common
{
    [Serializable]
    public class CustomResourceType : Attribute
    {
        private readonly string _NameSpace;
        private readonly string _Template;
        private readonly bool _IsCreatable;

        public CustomResourceType(string nameSpace = "Workflow", string template = "All", bool isCreatable = true)
        {
            if (null == template) template = "All";
            _NameSpace = nameSpace;
            _Template = "," + template.Trim(',') + ",";
            _IsCreatable = isCreatable;
        }

        public string NameSpace
        {
            get { return _NameSpace; }
        }

        public string Template
        {
            get { return _Template; }
        }

        public bool IsCreatable
        {
            get { return _IsCreatable; }
        }
    }
}

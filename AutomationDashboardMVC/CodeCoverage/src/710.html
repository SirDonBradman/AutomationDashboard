using System.Web.UI.WebControls;
using Aurigo.AMP3.DataAccess.Core;
using Aurigo.Brix.Platform.BusinessLayer.AbstractModels;

namespace Aurigo.AMP3.LibraryUI
{
    public class CalendarXmlFormManager : XMLFormManagerModel
    {
        public override void BeforeSave(BrixFormModel model, XmlFormArgs e)
        {
            Panel p = model.form.GetControl("IsDefault").ControlReference[0] as Panel;
            if ((p.Controls[0] as CheckBox).Checked)
                ComponentHelper.Instance.ExecuteNonQueryWithVariableParameters(
                    Aurigo.Brix.Platform.BusinessLayer.DataAccessHelper.StoredProcedure.usp_LIBRARYCalendarSetIsDefault,
                    null);
            e.IsValid = true;
        }
    }
}

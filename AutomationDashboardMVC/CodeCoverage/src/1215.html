using System;

namespace Brix62
{
    [BrixWidget("Widget2")]
    public partial class Widget2 : Widget
    {
        [Persistable]
        public string guyName
        {
            get { return tbName.Text; }
            set { tbName.Text = value; }
        }

        protected override void OnPreRender(EventArgs e)
        {
            // render the control based on the mode
            ltrName.Text = guyName;
            salesDisplaydiv.Visible = SalesFullMode.Visible = salesSettings.Visible = false;
            switch (Mode)
            {
                case WidgetMode.Settings:
                    salesSettings.Visible = true;
                    break;
                case WidgetMode.Full:
                    SalesFullMode.Visible = true;
                    break;
                case WidgetMode.Catalog:
                    // Render catalog ; hide everything else
                    break;
                default:
                    salesDisplaydiv.Visible = true;
                    break;
            }

            base.OnPreRender(e);
        }
    }
}

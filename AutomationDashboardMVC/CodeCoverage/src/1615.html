using System.Collections.Generic;
using Aurigo.AMP3.ContractManager.BusinessLayer;
using Aurigo.AMP3.ContractManager.BusinessLayer.DTO;
using Aurigo.AMP3.UserManagementBL;
using Aurigo.AMP3.UserManagementDTO;
using Aurigo.Brix.Core.WebUiHelper.EmailServiceFramework;
using Aurigo.Brix.Platform.BusinessLayer.EmailServiceFramework;
using Aurigo.Brix.Platform.CoreUtilities.Utility;

namespace Aurigo.BRIX.ContractManager.MilestoneUI
{
    public class MilestoneSubstitutionHandler : ISubstitute
    {
        #region ISubstitute Members

        public void Substitute(EmailTemplate MilestoneTemplate, Dictionary<string, object> parameters)
        {
            MILSTONDTO milestone = MILSTONBL.Instance.GetMilestoneDto(parameters["MilestoneID"].ToString().ToInt32_2(),
                                                                      parameters["ContractID"].ToString().ToInt32_2());
            User admin = UserManager.Instance.GetAUser(UserManager.Instance.GetUID("Administrator"));
            var dtos = new object[] {milestone, admin};
            EmailManager.EmailTemplateSubstitution(dtos, MilestoneTemplate);
        }

        #endregion
    }
}

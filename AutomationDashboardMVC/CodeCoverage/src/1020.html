using System;
using System.Collections.Generic;
using System.Linq;
using System.Reflection;
using System.Web;

namespace Aurigo.Brix.Platform.BusinessLayer.DTO
{
    public class MobileModuleConfigDTO_Extended : MobileModuleConfigDTO
    {
        public MobileModuleConfigDTO_Extended(MobileModuleConfigDTO config, List<string> permissions)
        {
            foreach (PropertyInfo baseProp in config.GetType().GetProperties())
            {
                PropertyInfo derivedProp = config.GetType().GetProperty(baseProp.Name);
                derivedProp.SetValue(this, baseProp.GetValue(config, null), null);
            }

            Permissions = permissions;
            if (config.NodeType == MobileModuleNodeType.Module)
            {
                if (!config.RequestParameters.ContainsKey("pid"))
                    config.RequestParameters.Add("PID", 0);
                if (!config.RequestParameters.ContainsKey("parentid"))
                    config.RequestParameters.Add("ParentID", 0);
                if (!config.RequestParameters.ContainsKey("context"))
                    config.RequestParameters.Add("Context", config.ModuleId);
            }
        }
        public List<string> Permissions { get; set; }
    }
}

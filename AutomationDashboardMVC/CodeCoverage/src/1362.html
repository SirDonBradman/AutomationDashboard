using Aurigo.AMP3.Common;
using System;
using System.Collections.Generic;
using System.IdentityModel.Services;
using System.Linq;
using System.Web;
using System.Web.UI;
using System.Web.UI.WebControls;

namespace Aurigo.Brix.Platform.UI
{
    public partial class AdfsLogout : System.Web.UI.Page
    {
        protected void Page_Load(object sender, EventArgs e)
        {
            /*
            * This page has access to the user in case of only ADFS has been enabled or
            * reauthentication from this page also needs to be redirected.
            */
            if (!AMP3ApplicationSettings.Instance.AdfsOnlyEnabled || 
                (Context.User != null && Context.User.Identity != null 
                    && Context.User.Identity.AuthenticationType == "Federation"))
            {
                Response.Redirect("~/Default.aspx");
            }
        }

        protected void btnLogin_Click(object sender, EventArgs e)
        {
            Request.Cookies.Set(new HttpCookie("mode", "RedirectToADFSLogin"));
            FederatedAuthentication.WSFederationAuthenticationModule.SignIn("Masterworks");
        }
    }
}

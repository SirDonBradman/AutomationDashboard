using System;
using System.Web.UI;
using Aurigo.AMP3.Common;
using Aurigo.AMP3.EstimatorBL;
using Aurigo.AMP3.Logging;
using Aurigo.AMP3.Resources.MessageResources;
using Aurigo.Brix.Platform.CoreUtilities.Utility;

namespace Aurigo.AMP3.EstimatorUI
{
    public partial class ReportFrame : Page
    {
        protected void Page_Load(object sender, EventArgs e)
        {
            try
            {
                int reportType = 0;
                string incEngEstimate = "";
                string rpxFile = "";

                if (!String.IsNullOrEmpty(Request.QueryString.Get("Type")))
                    reportType = Request.QueryString.Get("Type").ToInt32_2();

                if (!String.IsNullOrEmpty(Request.QueryString.Get("in")))
                    incEngEstimate = Request.QueryString.Get("in");

                string path = Server.MapPath(EstimatorConstants.PERIOD).ToString2();
                //MemoryStream strm;
                switch ((RPTName) reportType.ToByte2())
                {
                    case RPTName.EngEstimate:
                        rpxFile = @"\EngineerEstimate.rpx";
                        break;
                    case RPTName.BidSheet:
                        rpxFile = @"\rptBidsheet.rpx";
                        break;
                    case RPTName.BidResponse:
                        rpxFile = @"\rptBidResponse.rpx";
                        break;
                    case RPTName.BidDetails:
                        if (incEngEstimate == "" || incEngEstimate == "0")
                            rpxFile = @"\rptBidCheck_Exclude_EngEst.rpx";
                        else
                            rpxFile = @"\rptBidCheck.rpx";
                        break;
                    case RPTName.BidSummary:
                        if (incEngEstimate == "" || incEngEstimate == "0")
                            rpxFile = @"\rptBidSummary.rpx";
                        else
                            rpxFile = @"\rptBidSummary_Exclude_EngEst.rpx";
                        break;
                    case RPTName.CostSummaryReport:
                        rpxFile = @"\rptCostSummaryReport.rpx";
                        break;
                    case RPTName.Bidders:
                        rpxFile = @"\rptItemBidders.rpx";
                        break;
                    case RPTName.Addendums:
                        rpxFile = @"\rptAddendums.rpx";
                        break;
                    case RPTName.BidAdvertisement:
                        rpxFile = @"\rptBidLettingInfo.rpx";
                        break;

                    case RPTName.CostSheet:
                        rpxFile = @"\rptCostSheet.rpx";
                        break;

                    case RPTName.Submittals:
                        rpxFile = @"\rptEstimateSubmittals.rpx";
                        break;
                }

                int AddendumID = -1;
                if (!String.IsNullOrEmpty(Request["AddendumID"]))
                    AddendumID = Request["AddendumID"].ToInt32_2();

                path = path + rpxFile;
                string logoFilePath = Server.MapPath(AMP3ApplicationSettings.Instance.AppLogo);
            }
            catch (Exception ex)
            {
                Logger.Log(Enumerations.LogType.Error, ex.Message, EstimatorConstants.ESTIMATOR);
            }
        }

        #region Nested type: RPTName

        private enum RPTName : byte
        {
            EngEstimate,
            BidSheet,
            BidDetails,
            BidSummary,
            CostSummaryReport,
            Bidders,
            Addendums,
            BidAdvertisement,
            CostSheet,
            BidResponse,
            Submittals
        }

        #endregion
    }
}

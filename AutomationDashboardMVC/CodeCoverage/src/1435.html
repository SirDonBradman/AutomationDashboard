using Aurigo.Brix.Platform.BusinessLayer.AbstractModels;
using Aurigo.Brix.Platform.BusinessLayer.XMLForm;

namespace Aurigo.Brix.Platform.UI.Modules.FRMBLDR
{
    public class AddStage : AddProcessor
    {
        public AddStage(BrixFormModel model) : base(model)
        {
        }

        public override string ControlName => "stage";

        public override string Add(Command cmd, out ControlContainer control)
        {
            control = null;
            var err = string.Empty;

            var dest = FormBuilderSupport.GetDestination(_model, cmd.destinationControl);
            if (dest != null)
            {
                if (dest is Form)
                {
                    var stage = new Stage
                    {
                        Name = GetUniqueName(ContainerTypes.Stage),
                        Width = "800px"
                    };

                    dest.Containers.Add(stage);
                    control = stage;
                }
                else
                    err = "Stage must be dropped on Form";
            }

            return err;
        }
    }
}

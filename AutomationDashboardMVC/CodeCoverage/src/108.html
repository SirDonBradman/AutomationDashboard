using Aurigo.Brix.Platform.BusinessLayer.AbstractModels;
using Aurigo.Brix.Platform.BusinessLayer.XMLForm;
using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;

namespace Aurigo.Brix.Construction.AddonModules.ContractManager.ChangeOrderUI.ConcreateModels
{
    public class RFIFormManager : XMLFormManagerModel
    {
        public override void BeforeSave(BrixFormModel model, XmlFormArgs e)
        {
            DateTime submittedDate, targetResponseDate;

            xControl submittedDateControl = model.form.GetControl("SubmitDate");
            xControl responseDateControl = model.form.GetControl("ResDate");

            DateTime.TryParse(submittedDateControl.GetServerValue(), out submittedDate);
            DateTime.TryParse(responseDateControl.GetServerValue(), out targetResponseDate);

            if (DateTime.Compare(targetResponseDate.Date, submittedDate.Date) < 0)
            {
                e.IsValid = false;
                e.ErrorMessage = "Target response date can not be before submitted date.";
            }
            else
            {
                e.IsValid = true;
            }
        }
    }
}

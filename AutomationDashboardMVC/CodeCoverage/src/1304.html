using System;
using Aurigo.AMP3.Common;
using Aurigo.AMP3.Core.UserControls;
using Aurigo.Brix.Core.WebUiHelper;

namespace Aurigo.AMP3.EnterpriseUI
{
    public partial class Enterprise : BrixPageBase
    {
        protected NewDashboard DBControl;

        protected override void OnLoad(EventArgs e)
        {
            //If bidder is logged, redirect to Contractor Bidding Page...
            if (AMP3ApplicationSettings.Instance.IsContractorBidderUser)
                Response.Redirect("~/Common/BrixListPage.aspx?Context=CONTBID", true);

            UpdateTree = true;

            PageTitle = "Enterprise Dashboard";

            base.OnLoad(e);
        }

        protected override void OnInit(EventArgs e)
        {
            if (!AMP3ApplicationSettings.Instance.IsContractorBidderUser)
                PermissionsToCheck.Add(Constants.MODFEAT_VISIBILITY);

            DBControl.ParentPagePath = "~/MODULES/ENTPRSE/Enterprise.aspx";

            ModuleId = AMP3ApplicationSettings.Instance.IsContractorBidderUser ? null : ModuleId;

            base.OnInit(e);

            ValidateContext();
        }

        private void ValidateContext()
        {
            if (Request["Context"].ToLower() != "entprse")
            {
                OnAccessDeniedRedirect();
            }
        }
    }
}

using System;
using Aurigo.Brix.Platform.CoreUtilities.Utility;

namespace Aurigo.AMP3.ExceptionManagement
{
    public class AMP3Exception : Exception
    {
        public AMP3Exception(string message)
        {
            //Logger.LogError(message, "");
        }

        public AMP3Exception(string message, Exception innerException)
        {
            string error = "Message : {0}, Source : {1}, StackTrace : {2}".Format2(message, innerException.Source,
                innerException.StackTrace);
            //Logger.LogError( error, "");
        }

        public AMP3Exception(Exception ex)
        {
            string error = "Message : {0}, Source : {1}, StackTrace : {2}".Format2(ex.Message, ex.Source, ex.StackTrace);
            //Logger.LogError(error, "");
        }

        public AMP3Exception(string message, string userID, int severity)
        {
            string error = "Message : {0}, User : {1}, Severity : {2}".Format2(message, userID, severity);
            //Logger.LogError(message, userID);
        }

        public AMP3Exception(string message, string userID, int severity, Exception innerException)
        {
            string error = "Message : {0}, UserId : {1}, Severity : {2}, Source : {3}, StackTrace : {4}".Format2(
                message, userID, severity, innerException.Source, innerException.StackTrace);
            //Logger.LogError(error, userID);
        }

        public AMP3Exception(Exception ex, string userID, int severity)
        {
            string error =
                "Message : {0}, User : {1}, Severity : {2}, Source : {3}, StackTrace : {4}".Format2(ex.Message, userID,
                    severity, ex.Source,
                    ex.StackTrace);
            //Logger.LogError(error, userID);
        }
    }

    public class AMP3SuccessMessage : Exception
    {
        public AMP3SuccessMessage(string message) : base(message)
        {
        }
    }
}

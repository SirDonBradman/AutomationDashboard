using System.Collections.Generic;
using Aurigo.AMP3.DataAccess.Core;
using Aurigo.AMP3.UserManagementBL;
using Aurigo.Brix.Platform.BusinessLayer.DTO;
using Aurigo.Common.Utility;

namespace Aurigo.Brix.Platform.BusinessLayer.BL
{
    public class SignUpManager : SingletonManagerBase<SignUpManager>
    {
        private SignUpManager()
        {
        }

        public int CUDOrganizationDetails(SignUpDTO dto)
        {
            // Check for all business logic here
            return SignUpComponent.Instance.CUDOrganizationDetails(dto);
        }

        public SignUpDTO GetOrganizationDetails(int id)
        {
            return SignUpComponent.Instance.GetOrganizationDetails(id);
        }

        public UserMapping GetUserMapping(string emailId)
        {
            return SignUpComponent.Instance.GetUserMapping(emailId);
        }

        public int CUDUserMapping(UserMapping dto)
        {
            return SignUpComponent.Instance.CUDUserMapping(dto);
        }

        public CompanyDetails GetCompanyDetails(string code)
        {
            return SignUpComponent.Instance.GetCompanyDetails(code);
        }

        public void CUOrganizationDetail(SignUpDetailDTO dto)
        {
            SignUpComponent.Instance.CUOrganizationDetail(dto);
        }

        public void CURequestPassword(ForgotPasswrd dto)
        {
            SignUpComponent.Instance.CURequestPassword(dto);
        }

        public ForgotPasswrd GetRequestPassword(string requestGuid)
        {
            return SignUpComponent.Instance.GetRequestPassword(requestGuid);
        }

        public void ResetPassword(string email, string password, string dbPerfix)
        {
            password = UserManager.Instance.GetSHA256Password(password);
            SignUpComponent.Instance.ResetPassword(email, password, dbPerfix);
        }
    }
}

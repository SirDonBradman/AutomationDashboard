using Aurigo.AMP3.Core.BusinessLayer.BL;
using System;
using System.Collections.Generic;
using System.Linq;
using System.Net;
using System.Net.Http;
using System.Web;
using System.Web.Http;

namespace Aurigo.Brix.Platform.BusinessLayer.Controller
{
    public class ChecklistController : ApiController
    {
        [HttpGet]
        public HttpResponseMessage IsCheckListCompleted(string moduleId, int checklistId, int formInstanceId)
        {
            bool isCompleted = ChecklistManager.Instance.IsCheckListCompleted(moduleId, checklistId, formInstanceId);
            return Request.CreateResponse<Boolean>(HttpStatusCode.OK, isCompleted);
        }
    }
}

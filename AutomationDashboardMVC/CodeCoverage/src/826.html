using System;
using System.Collections.Generic;
using System.Xml.Serialization;

namespace Aurigo.AMP3.Core.BusinessLayer.DTO
{
    [Serializable]
    [XmlRoot("SectionDetail")]
    public class SectionDetail
    {
        #region Fields

        private String _ModuleID;
        private Int32 _ParentInstanceID;
        private List<Section> _Sections = new List<Section>();

        #endregion

        #region Properties

        public String ModuleID
        {
            get { return _ModuleID; }

            set { _ModuleID = value; }
        }

        public Int32 ParentInstanceID
        {
            get { return _ParentInstanceID; }

            set { _ParentInstanceID = value; }
        }

        public List<Section> Sections
        {
            get { return _Sections; }

            set { _Sections = value; }
        }

        #endregion
    }

    public class Section
    {
        #region Fields

        #endregion

        #region Properties

        /// <summary>
        /// The database automatically generates this value
        /// </summary>
        public Int32 SectionID { get; set; }

        public String SectionName { get; set; }

        public String SectionDesc { get; set; }

        public String SectionNotes { get; set; }

        /// <summary>
        /// This is not a required field
        /// </summary>
        public Int64? OrderID { get; set; }

        #endregion
    }
}

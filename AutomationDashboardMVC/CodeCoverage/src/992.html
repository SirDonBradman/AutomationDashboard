using System;
using System.Data;
using Aurigo.AMP3.Common;
using Aurigo.AMP3.DocumentManagementBL;
using Aurigo.Brix.Platform.CoreUtilities.Utility;

namespace Aurigo.Brix.Platform.BusinessLayer.UserControls
{
    public partial class DraftedTransmittals : System.Web.UI.UserControl
    {
        public int ProjectID
        {
            get
            {
                if (!string.IsNullOrEmpty(Request.QueryString.Get("PID")))
                    return Request.QueryString.Get("PID").ToInt32_2();
                else
                    return 0;
            }
        }

        public int UserID
        {
            get
            {
                UserDetail ud = UserDetail.GetCurrentUserDetail();
                return ud.UID;
            }
        }

        protected void btnLoadData_Click(object sender, EventArgs e)
        {
            try
            {
                DataTable dtTransmittals = TransmittalManager.Instance.GetDraftedTransmittals(ProjectID, UserID);
                ddlDraftedTransmittals.DataSource = dtTransmittals;
                ddlDraftedTransmittals.DataTextField = dtTransmittals.Columns[0].ColumnName;
                ddlDraftedTransmittals.DataValueField = dtTransmittals.Columns[1].ColumnName;
                ddlDraftedTransmittals.DataBind();
            }
            catch (Exception ex)
            {
                throw ex;
            }
        }

        protected void Page_Load(object sender, EventArgs e)
        {
        }
    }
}

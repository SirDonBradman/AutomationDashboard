using Aurigo.Common.Utility;
using System;
using System.Configuration;

namespace Aurigo.Brix.Platform.BusinessLayer.BL
{
    public class PDFNetManager : SingletonManagerBase<PDFNetManager>
    {
        private PDFNetManager()
        {
            try
            {
                string pdfNetSDKLicenseKey = ConfigurationManager.AppSettings["PDFNetSDKLicenseKey"];
                pdftron.PDFNet.Initialize(pdfNetSDKLicenseKey);
            }
            catch (Exception ex)
            {
            }
        }
        public void Initiallize()
        {

        }
    }
    public class PDFNetLoader: SingletonManagerBase<PDFNetLoader>
    {
        private pdftron.PDFNetLoader pdfnet = null;
        /// <summary>
        /// Load PDFNetManager for runtime PDFNet loading.
        /// Loads PDFNet.dll's from 'PDFNet/[x86|x64]'
        /// folders in current working directory.
        /// </summary>
        /// <returns></returns>
        private PDFNetLoader()
        {
            try
            {
                pdfnet = pdftron.PDFNetLoader.Instance();
            }
            catch(Exception ex)
            {

            }
        }

        public void Initiallize()
        {

        }
    }
}

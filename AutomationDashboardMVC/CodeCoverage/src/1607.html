using System.Collections.Generic;
using System.Data;
using Aurigo.AMP3.Common;
using Aurigo.AMP3.DataAccess.Core;
using Aurigo.Brix.Construction.ContractManager.BusinessLayer.BL;

namespace Aurigo.AMP3.ContractManager.MATRISSUI
{
    [ReportModelContext(Name = "MATRISSSR", ModuleId = "MATRISS")]
    public class MIStockReconciliationReportModel : MIMaterialsReportModel
    {
        public override string ReportPath
        {
            get { return "../Modules/CONTMGT/Reports/rptStockReconciliation.rpx"; }
        }

        public override string Header
        {
            get { return "Material Stock Reconciliation Report"; }
        }

        public override bool HasStaticFilter
        {
            get { return false; }
        }

        public override DataTable GetDataTable(Dictionary<string, string> dicFilter)
        {
            return
                ComponentHelper.Instance.ExecuteDataSet(
                    MaterialIssueStoredProcedure.usp_MATRISSGetMaterialsBySiteEngineer, null, 0, ParentID, null, null,
                    null).Tables[0];
        }
    }
}

    public class CustomDBException : Exception
    {
        /// <summary>
        /// This variable is used for purely for debugging purpose.
        /// Must not be set as inner exception of this class
        /// </summary>
        public Exception ActualException { get; private set; }

        /// <summary>
        /// Only if this is set then SqlException_Number, SqlException_State and SqlException_Class will have values in them
        /// </summary>
        public bool IsDBException { get; private set; }

        /// <summary>
        /// Summary:
        ///     Gets a number that identifies the type of error.
        /// 
        /// Returns:
        ///     The number that identifies the type of error.
        /// </summary>
        public int? SqlException_Number { get; private set; }

        /// <summary>
        /// Summary:
        ///     Gets a numeric error code from SQL Server that represents an error, warning or
        ///     "no data found" message.For more information about how to decode these values,
        ///     see SQL Server Books Online.
        ///
        /// Returns:
        ///     The number representing the error code.
        /// </summary>
        public byte? SqlException_State { get; private set; }

        /// <summary>
        ///  Summary:
        ///      Gets the severity level of the error returned from the .NET Framework Data Provider
        ///      for SQL Server.
        /// 
        ///  Returns:
        ///      A value from 1 to 25 that indicates the severity level of the error.
        /// </summary>
        public byte? SqlException_Class { get; private set; }

        public CustomDBException(string wrapperMessage, Exception ex) : base(wrapperMessage)
        {
            if (ex is SqlException)
            {
                var sqlEx = ex as SqlException;

                this.SqlException_Number = sqlEx.Number;
                this.SqlException_State = sqlEx.State;
                this.SqlException_Class = sqlEx.Class;
            }
        }
    }

        private static void ThrowCustomDBException(Exception ex)
        {
            if (ex is CustomDBException)
                throw ex;

            int logId = LogInDB("ComponentHelper", "UnknownUser", ex);
            throw new CustomDBException("Oops! There was a database error. We have logged it for you. Error LogID: " + logId.ToString() + ". Please Contact your Administrator for more details.", ex);
        }

        private static void ThrowCustomDBException(Exception ex, object dto, params object[] parameters)
        {
            if (ex is CustomDBException)
                throw ex;

            StringBuilder sbr_StackTrace = new StringBuilder();
            StringBuilder errorMsg = new StringBuilder();
            errorMsg.AppendLine("Database exception");

            if (ex != null)
                errorMsg.AppendLine(ex.Message);

            if (dto != null)
            {
                foreach (PropertyInfo prop in dto.GetType().GetProperties())
                {
                    var propVal = prop.GetValue(dto);
                    if (propVal != null && prop.PropertyType == typeof(string))
                        errorMsg.AppendFormat("Parameter: {0}, Value: {1}", prop.Name, string.IsNullOrEmpty(propVal.ToString()) ? "null" : propVal);
                    else
                        errorMsg.AppendFormat("Parameter: {0}, Value: {1}", prop.Name, (propVal == null || propVal.Equals(DBNull.Value)) ? "null" : propVal);
                    errorMsg.AppendLine();
                }
            }
            foreach (object obj in parameters)
            {
                if (obj != null)
                {
                    if (obj is BaseStoredProcedure)
                    {
                        BaseStoredProcedure db = (obj as BaseStoredProcedure);
                        errorMsg.AppendFormat("Stored Procedure Name: {0}, Input: {1}, Out: {2}, InOut: {3}", db.Name, db.Input, db.Out, db.InOut);
                    }
                    else if (obj is Dictionary<string, object>)
                    {
                        foreach (KeyValuePair<string, object> item in (obj as Dictionary<string, object>))
                        {
                            errorMsg.AppendFormat("Key: {0}, Value: {1}", item.Key, item.Value);
                            errorMsg.AppendLine();
                        }
                    }
                    else if (obj is object[])
                    {
                        foreach (object item in (obj as object[]))
                        {
                            errorMsg.AppendFormat("Parameter Value: {0}", item);
                            errorMsg.AppendLine();
                        }
                    }
                    else
                        errorMsg.AppendFormat("Parameter Value: {0}", obj);
                    errorMsg.AppendLine();
                }
            }

            if (ex != null)
                sbr_StackTrace.AppendLine(ex.StackTrace);

            int logId = LogInDB(0, "0", DateTime.UtcNow, "ComponentHelper", 0, "UnknownUser", errorMsg.ToString(), sbr_StackTrace.ToString());

            throw new CustomDBException("Oops! There was a database error. We have logged it for you. Error LogID: " + logId.ToString() + ". Please Contact your Administrator for more details.", ex);
        }

        /// <summary>
        /// 
        /// </summary>
        /// <param name="source"></param>
        /// <param name="username"></param>
        /// <param name="ex"></param>
        /// <returns></returns>
        public static int LogInDB(string source, string username, Exception ex)
        {
            StringBuilder sbr_StackTrace = new StringBuilder();
            string errorMsg = "Unknown exception";

            if (ex != null)
            {
                errorMsg = ex.Message;
                sbr_StackTrace.AppendLine(ex.StackTrace);
            }

            return LogInDB(0, "0", DateTime.UtcNow, source, 0, username, errorMsg, sbr_StackTrace.ToString());
        }
        
        /// <summary>
        /// Log exception to table APPMGMTLog and returns reference id
        /// </summary>
        /// <param name="category"></param>
        /// <param name="type"></param>
        /// <param name="timestamp"></param>
        /// <param name="source"></param>
        /// <param name="eventid"></param>
        /// <param name="username"></param>
        /// <param name="message"></param>
        /// <param name="stackTrace"></param>
        /// <returns></returns>
        public static int LogInDB(int category, string type, DateTime timestamp, string source, int eventid, string username, string message, string stackTrace)
        {
            int logId = 0;

            using (var conn = new ConnHolder(ConnectionHelper.GetConnectionString()))
            {
                DbCommand cmd = conn.DB.GetStoredProcCommand("usp_APPMGMTWriteLog");

                conn.DB.AddInParameter(cmd, "TYPE", DbType.String, type);
                conn.DB.AddInParameter(cmd, "DATETIME", DbType.DateTime, timestamp);
                conn.DB.AddInParameter(cmd, "SOURCE", DbType.String, source);
                conn.DB.AddInParameter(cmd, "CATEGORY", DbType.Int32, category);
                conn.DB.AddInParameter(cmd, "EVENT", DbType.Int32, eventid);
                conn.DB.AddInParameter(cmd, "USERNAME", DbType.String, username);
                conn.DB.AddInParameter(cmd, "MESSAGE", DbType.String, message);
                conn.DB.AddInParameter(cmd, "STACKTRACE", DbType.String, stackTrace);
                conn.DB.AddOutParameter(cmd, "STATUS", DbType.Int32, 4);
                

                conn.DB.ExecuteScalar(cmd);

                logId = cmd.Parameters["@STATUS"].Value.ToInt32_2();
            }

            return logId;
        }

        public DataSet ExecuteDataSetOn(string connectionString, BaseStoredProcedure sp, Dictionary<string, object> dic,
            params object[] parameters)
        {
            try
            {
                using (var conn = new ConnHolder(connectionString))
                {
                    DbCommand cmd = conn.DB.GetStoredProcCommand(sp.Name);
                    int timeOut = cmd.CommandTimeout;
                    if (!String.IsNullOrEmpty(ConfigurationManager.AppSettings["SQLCommandTimeOut"]) &&
                        Int32.TryParse(ConfigurationManager.AppSettings["SQLCommandTimeOut"], out timeOut))
                        cmd.CommandTimeout = timeOut;
                    string[] strs = SetInputOutParameters(conn.DB, sp, dic, parameters, cmd);

                    DataSet ds = conn.DB.ExecuteDataSet(cmd);
                    SetOutValues(conn.DB, dic, cmd, strs);
                    if (LogManager.CanLogDebug)
                        LogManager.LogDebug(new AppErrorInfo("ComponentHelper", "DBQUERY: " + sp.Name, "System"),
                            MethodBase.GetCurrentMethod(), LogManager.LogFileForDBOps);
                    return ds;
                }
            }
            catch (Exception ex)
            {
                ThrowCustomDBException(ex, null, sp, dic, parameters);
            }
            return null;
        }

        private string[] SetInputOutParameters<T>(Database db, BaseStoredProcedure sp, Dictionary<string, object> dic,
            T dto, DbCommand cmd) where T : class
        {
            string[] strs = null;

            try
            {
                SetInputParametersFromDto(db, dto, cmd, sp);
                strs = SetOutPatameters(db, cmd, sp, strs);

                if (dic != null && dic.Count != 0)
                {
                    strs = SetInOutParameters(db, sp, dic, cmd, strs);
                }
            }
            catch (Exception ex)
            {
                ThrowCustomDBException(ex, dto, sp, dic);
            }

            return strs;
        }

        private string[] SetInputOutParametersWithNulls<T>(Database db, BaseStoredProcedure sp,
            Dictionary<string, object> dic,
            T dto, DbCommand cmd) where T : class
        {
            string[] strs = null;

            try
            {

                strs = SetOutPatameters(db, cmd, sp, strs);

                {

        }

        private string[] SetInputOutParameters(Database db, BaseStoredProcedure sp, Dictionary<string, object> dic,
            object[] parameters, DbCommand cmd)
        {
            string[] strs = null;

            try
            {
                if (parameters != null)
                {
                    SetInputParameters(db, parameters, cmd, sp);
                    strs = SetOutPatameters(db, cmd, sp, strs);
                }

                if (dic != null && dic.Count != 0)
                {
                    strs = SetInOutParameters(db, sp, dic, cmd, strs);
                }
            }
            catch (Exception ex)
            {
                ThrowCustomDBException(ex, null, sp, dic, parameters);
            }
            return strs;
        }

        private void SetInputParameters(Database db, object[] values, DbCommand cmd, BaseStoredProcedure sp)
        {
            if (sp.Input != null && !sp.Input.Equals(string.Empty))
            {
                string[] inputs = sp.Input.Split(",".ToCharArray());
                int i = 0;
                foreach (string input in inputs)
                {
                    if (values[i] != null)
                    {
                        db.AddInParameter(cmd, input.Trim(), dbTypeDictionary[values[i].GetType()], values[i]);
                    }
                    i++;
                }
            }
        }

        private void SetInputParametersFromDto<T>(Database db, T dto, DbCommand cmd, BaseStoredProcedure sp)
            where T : class
        {
            if (dto != null)
            {
                if (sp.Input != null && !sp.Input.Equals(string.Empty))
                {
                    string[] inputs = sp.Input.Split(",".ToCharArray());
                    foreach (string input in inputs)
                    {
                        PropertyInfo prop = dto.GetType().GetProperty(input);
                        if (prop != null)
                        {
                            object value = prop.GetValue(dto, null);
                            if (value != null)
                            {
                                db.AddInParameter(cmd, prop.Name, dbTypeDictionary[prop.PropertyType], value);
                            }
                        }
                    }
                }
            }
        }

        private void SetInputParametersFromDtoWithNulls<T>(Database db, T dto, DbCommand cmd, BaseStoredProcedure sp)
            where T : class
        {
            if (dto == null) return;
            if (sp.Input == null || string.IsNullOrEmpty(sp.Input)) return;
            string[] inputs = sp.Input.Split(",".ToCharArray());
            foreach (string input in inputs)
            {
                PropertyInfo prop = dto.GetType().GetProperty(input);
                if (prop == null) continue;
                object value = prop.GetValue(dto, null);
                db.AddInParameter(cmd, prop.Name, dbTypeDictionary[prop.PropertyType], value);
            }
        }

        private string[] SetOutPatameters(Database db, DbCommand cmd, BaseStoredProcedure sp, string[] strs)
        {
            if (sp.Out != null && !sp.Out.Equals(string.Empty))
            {
                var list = new List<string>();
                string[] outputs = sp.Out.Split(";".ToCharArray());
                foreach (string output in outputs)
                {
                    string[] outputValue = output.Split(",".ToCharArray());
                    int dbType = 0;
                    int.TryParse(outputValue[1].Trim(), out dbType);
                    int size = 0;
                    int.TryParse(outputValue[2].Trim(), out size);

                    db.AddOutParameter(cmd, outputValue[0].Trim(), (DbType)dbType, size);
                    list.Add(outputValue[0]);
                }
                strs = list.ToArray();
            }
            return strs;
        }

        private string[] SetInOutParameters(Database db, BaseStoredProcedure sp, Dictionary<string, object> dic,
            DbCommand cmd,
            string[] strs)
        {
            if (sp.InOut != null && !sp.InOut.Equals(string.Empty))
            {
                var list = new List<string>();
                string[] inouts = sp.InOut.Split(";".ToCharArray());
                foreach (string inout in inouts)
                {
                    string[] inoutValue = inout.Split(",".ToCharArray());
                    int dbType = 0;
                    int.TryParse(inoutValue[1].Trim(), out dbType);
                    int size = 0;
                    if (inoutValue.Length > 2)
                        int.TryParse(inoutValue[2].Trim(), out size);
                    byte precision = 0;
                    if (inoutValue.Length > 3)
                        byte.TryParse(inoutValue[3].Trim(), out precision);
                    byte scale = 0;
                    if (inoutValue.Length > 4)
                        byte.TryParse(inoutValue[4].Trim(), out scale);
                    db.AddParameter(cmd, inoutValue[0].Trim(), (DbType)dbType, size, ParameterDirection.InputOutput,
                        false, precision, scale, null,
                        DataRowVersion.Default, dic[inoutValue[0]]);

                    list.Add(inoutValue[0]);
                }
                if (strs != null)
                    list.AddRange(strs);
                strs = list.ToArray();
            }
            return strs;
        }

        private void SetOutValues(Database db, Dictionary<string, object> dic, DbCommand cmd, string[] strs)
        {
            if (dic != null && strs != null)
            {
                foreach (string str in strs)
                {
                    dic[str] = db.GetParameterValue(cmd, str);
                }
            }
        }

        public static void CopyDataReaderToDto<T>(IDataReader dr, T dto) where T : class
        {
            //NOTE: This will work for one level
            //PLEASE call dr.Read() before passing the reader
            DataTable table = dr.GetSchemaTable();
            foreach (DataRow row in table.Rows)
            {
                PropertyInfo prop = dto.GetType().GetProperty(row["ColumnName"].ToString2());
                if (prop != null)
                {
                    SetValue(dr[prop.Name], dto, prop);
                }
            }
        }

        public static void CopyDataTableToDto<T>(DataRow row, DataColumnCollection columns, T dto) where T : class
        {
            //NOTE: This will work for one level
            foreach (DataColumn column in columns)
            {
                PropertyInfo prop = dto.GetType().GetProperty(column.Caption);
                if (prop != null)
                {
                    SetValue(row[prop.Name], dto, prop);
                }
            }
        }

        private static void SetValue<T>(object obj, T dto, PropertyInfo prop) where T : class
        {
            if (prop.PropertyType == obj.GetType())
            {
                prop.SetValue(dto, obj, null);
            }
            else
            {
                if (prop.PropertyType == typeof(string))
                {
                    prop.SetValue(dto, obj.ToString2(), null);
                }
                else if (!prop.PropertyType.IsPrimitive)
                {
                    if ((prop.PropertyType).FullName.Contains("System.Int64"))
                    {
                        if (!obj.Equals(DBNull.Value))
                            prop.SetValue(dto, obj.ToInt64_2(), null);
                    }
                    else
                        prop.SetValue(dto, obj.Equals(DBNull.Value) ? null : obj, null);
                }
            }
        }

        private void GenerateSQLParametersAndNamesFromObjectArray(object[] parameters, out SqlParameter[] paramsArray, out object[] names)
        {
            paramsArray = null;
            names = null;

            if (parameters != null && parameters.Length > 0)
            {
                paramsArray = new SqlParameter[parameters.Length];
                names = new string[parameters.Length];

                string pName = string.Empty;
                for (int i = 0; i < parameters.Length; i++)
                {
                    pName = "p" + i.ToString();
                    names[i] = "@" + pName;
                    paramsArray[i] = new SqlParameter(pName, parameters[i]);
                }
            }
        }

    public class ELDBPool : SafePool<Database>
    {
        private int _PoolSize;

        public ELDBPool(string connect)
        {
            ConnectionStr = connect;
        }

        public string ConnectionStr { get; set; }

        public override int MaxObjectsInPool
        {
            get
            {
                if (_PoolSize > 0 && _PoolSize < 21) return _PoolSize;
                try
                {
                    _PoolSize = int.Parse((string)ConfigurationManager.AppSettings["DBPoolSize"]);
                    return _PoolSize;
                }
                catch
                {
                }

                _PoolSize = 20;
                return _PoolSize;
            }
        }

        protected override Database GetNewItemInstance()
        {
            if (string.IsNullOrEmpty(ConnectionStr))
                return DatabaseFactory.CreateDatabase();
            //return DatabaseFactory.CreateDatabase(ConnectionStr);
            return new Microsoft.Practices.EnterpriseLibrary.Data.Sql.SqlDatabase(ConnectionStr);
        }
    }

    public class ELibConnectionPool
    {
        private static volatile ELibConnectionPool _Inst = null;
        private static readonly object _oSync = new object();

        private ELibConnectionPool()
        {
            _ConnectionBasedPools = new Dictionary<string, ELDBPool>();
        }

        private Dictionary<string, ELDBPool> _ConnectionBasedPools;

        public static ELibConnectionPool Instance()
        {
            if (_Inst == null) lock (_oSync) if (_Inst == null) _Inst = new ELibConnectionPool();
            return _Inst;
        }

        public SafePool<Database>.PoolItem<Database> GetFromPool(string connString)
        {
            if (!_ConnectionBasedPools.ContainsKey(connString))
            {
                lock (_oSync)
                {
                    if (!_ConnectionBasedPools.ContainsKey(connString))
                        _ConnectionBasedPools.Add(connString, new ELDBPool(connString));
                }
            }

            return _ConnectionBasedPools[connString].GetFromPool();
        }
    }

    public class ConnHolder : IDisposable
    {
        SafePool<Database>.PoolItem<Database> _db;

        public ConnHolder(string connString)
        {
            _db = ELibConnectionPool.Instance().GetFromPool(connString);
        }

        public void Dispose()
        {
            _db.ReturnToPool();
        }

        public Database DB
        {
            get { return _db.Value; }
        }
    }

    public delegate void ReturnToQ(string key);

    public abstract class SafePool<T>
    {
        private Queue<PoolItem<T>> _availableQueue;
        private Dictionary<string, PoolItem<T>> _objectsInUse;
        private AutoResetEvent _evt;
        private int _NextKey;
        private static readonly object _poolLock = new object();

        public SafePool()
        {
            _availableQueue = new Queue<PoolItem<T>>();
            _objectsInUse = new Dictionary<string, PoolItem<T>>();
            _evt = new AutoResetEvent(true);
            _NextKey = 1;
        }

        protected abstract T GetNewItemInstance();

        public int CountSize
        {
            get { return _availableQueue.Count + _objectsInUse.Count; }
        }

        public int CountAvailable
        {
            get { return _availableQueue.Count; }
        }

        public int CountInUse
        {
            get { return _objectsInUse.Count; }
        }

        public void ReturnToQueue(PoolItem<T> item)
        {
            ReturnToQueue(item.Key);
        }

        public PoolItem<T> GetFromPool()
        {
            PoolItem<T> obj = null;
            EnsureObjectsInQueue();
            if (_availableQueue.Count < 1)
            {
                if (MaxWait < 1) _evt.WaitOne();
                else _evt.WaitOne(MaxWait);
            }
            lock (_poolLock)
            {
                if (_availableQueue.Count > 0) obj = (PoolItem<T>)_availableQueue.Dequeue();
                _objectsInUse.Add(obj.Key, obj);
            }
            return obj;
        }

        private void EnsureObjectsInQueue()
        {
            if (MaxObjectsInPool <= CountSize) return;
            lock (_poolLock)
            {
                while (MaxObjectsInPool > CountSize)
                {
                    PoolItem<T> obj = new PoolItem<T>(new ReturnToQ(ReturnToQueue), _NextKey.ToString(),
                        GetNewItemInstance());
                    _NextKey++;
                    _availableQueue.Enqueue(obj);
                }
            }
        }

        private void ReturnToQueue(string key)
        {
            lock (_poolLock)
            {
                PoolItem<T> obj = (PoolItem<T>)_objectsInUse[key];
                _objectsInUse.Remove(key);
                _availableQueue.Enqueue(obj);
            }
            if (_availableQueue.Count > 0) _evt.Set();
        }

        public abstract int MaxObjectsInPool { get; }

        public virtual int MaxWait
        {
            get { return 0; }
        }

        public class PoolItem<Tp>
        {
            event ReturnToQ _delg;
            string _Key;
            Tp _Val;

            public PoolItem(ReturnToQ delg, string key, Tp val)
            {
                _delg = delg;
                _Key = key;
                _Val = val;
            }

            public string Key
            {
                get { return _Key; }
            }

            public Tp Value
            {
                get { return _Val; }
            }

            public void ReturnToPool()
            {
                if (null != _delg) _delg(_Key);
            }
        }
    }

using System.Collections.Generic;
using System.Web;
using Aurigo.AMP3.Resources.MessageResources;
using Aurigo.AMP3.Resources.TerminologyResources;

namespace Aurigo.AMP3.Core.AbstractModels
{
    public delegate void RegisterScriptHandler(string script, System.Web.UI.Control registerForControl = null, string key = "", string targetKey = "", bool afterTarget = false);
    public class BrixModel
    {
        public delegate void ShowNotificationHandler(string message);

        public event ShowNotificationHandler OnShowSuccess;
        public event ShowNotificationHandler OnShowError;
        public event ShowNotificationHandler OnShowWarning;
        public event ShowNotificationHandler OnShowInfo;

        public event RegisterScriptHandler OnRegisterStartupScript;
        public event RegisterScriptHandler OnRegisterClientScriptBlock;
        public event RegisterScriptHandler OnRegisterClientScriptInclude;
        
        public HttpRequest Request;
        public HttpResponse Response;

        public Dictionary<string, object> RequestParams { get; set; }

        public string ItemNameAbbr
        {
            get { return LocalizationManager.GetString(KeyConstants.LOC_ITEM_ABBR); }
        }

        public string PayEstimatesName
        {
            get { return LocalizationManager.GetString(KeyConstants.LOC_PAY_ESTIMATES); }
        }

        public string PayEstimateName
        {
            get { return LocalizationManager.GetString(KeyConstants.LOC_PAY_ESTIMATE); }
        }

        public string PrePaymentName
        {
            get { return LocalizationManager.GetString(KeyConstants.LOC_MATERIAL_ON_HAND); }
        }

        public string Group
        {
            get { return LocalizationManager.GetString(KeyConstants.Group); }
        }

        public string UserRejectMessage
        {
            get
            {
                return MessageResourceManager.GetString("E_USRMGMT_SELECTED_USERS_HAVE_BEEN_REJECTED",
                    Enumerations.MessageType.ErrorMessage);
            }
        }

        public string UserApproveMessage
        {
            get
            {
                return MessageResourceManager.GetString("M_USRMGMT_SELECTED_USERS_HAVE_BEEN_APPROVED",
                    Enumerations.MessageType.InfoMessage);
            }
        }

        protected string SectionName
        {
            get { return LocalizationManager.GetString(KeyConstants.LOC_SECTION); }
        }

        protected string PayItemNoAbbr
        {
            get { return LocalizationManager.GetString(KeyConstants.LOC_PAY_ITEM_NO_ABBR); }
        }

        protected string ItemName
        {
            get { return LocalizationManager.GetString(KeyConstants.LOC_ITEMS); }
        }

        protected string PayItemNo
        {
            get { return LocalizationManager.GetString(KeyConstants.LOC_PAY_ITEM_NO); }
        }

        protected string Retainage
        {
            get { return LocalizationManager.GetString(KeyConstants.LOC_RETAINAGE); }
        }

        protected string PayEstimates
        {
            get { return LocalizationManager.GetString(KeyConstants.LOC_PAY_ESTIMATES); }
        }

        protected string PayEstimate
        {
            get { return LocalizationManager.GetString(KeyConstants.LOC_PAY_ESTIMATE); }
        }

        protected string CurrencySymbol
        {
            get { return LocalizationManager.GetString(KeyConstants.LOC_CURRENCY_SYMBOL); }
        }

        protected void ShowSuccess(string message)
        {
            if (OnShowSuccess != null)
                OnShowSuccess(message);
        }

        protected void ShowWarning(string message)
        {
            if (OnShowWarning != null)
                OnShowWarning(message);
        }

        protected void ShowError(string message)
        {
            if (OnShowError != null)
                OnShowError(message);
        }

        protected void ShowInfo(string message)
        {
            if (OnShowInfo != null)
                OnShowInfo(message);
        }

        protected void RegisterStartupScript(string script, System.Web.UI.Control registerForControl = null, string key = "", string targetKey = "", bool afterTarget = false)
        {
            if (OnRegisterStartupScript != null)
                OnRegisterStartupScript(script, registerForControl, key, targetKey, afterTarget);
        }

        protected void RegisterClientScriptBlock(string script, System.Web.UI.Control registerForControl = null, string key = "", string targetKey = "", bool afterTarget = false)
        {
            if (OnRegisterClientScriptBlock != null)
                OnRegisterClientScriptBlock(script, registerForControl, key, targetKey, afterTarget);
        }

        protected void RegisterClientScriptInclude(string fileUrl, System.Web.UI.Control registerForControl = null, string key = "", string targetKey = "", bool afterTarget = false)
        {
            if (OnRegisterClientScriptInclude != null)
                OnRegisterClientScriptInclude(fileUrl, registerForControl, key, targetKey, afterTarget);
        }
    }
}

using System.Collections.Generic;
using Aurigo.AMP3.Core.IntegrationConnector;
using Aurigo.Brix.Platform.CoreUtilities.Utility;

namespace Aurigo.AMP3.TabInjectionDAC
{
    public enum PageMode
    {
        Create,
        View,
        Edit
    }

    public enum TabEnabledPages
    {
        CreateProject,
        CreateContract,
        ProjectDetails,
        ContractItems,
        EstimateItems,
        NewEstimateItem,
        EditEstimateItem,
        CreateContractItem,
        CreateStandardItem,
        StandardItems
    }

    public enum SaveStatus
    {
        Unknown,
        Add,
        Delete
    }


    public interface ITab
    {
        PageMode Mode { get; set; }
        TabEnabledPages AMP3PageName { get; set; }
        AMP3Object AMP3ParentObjectType { set; }
        string AMP3ParentObjectID { set; }
        AMP3Object AMP3ObjectType { get; set; }
        List<TabData> ERPPageData { get; set; }
        List<TabData> ERPTabPageData { get; set; }
        int AMP3ObjectID { get; set; }

        void MasterSaveAction(AMP3Object amp3ObjectType, int amp3ObjectId, AMP3Object amp3ParentObjectType,
            string amp3ParentObjectId, List<TabData> mainPageData);

        TabPreSaveCheckStatus PreSaveCheck();
        void MasterReset();
        void FillData(AMP3Object amp3ObjectType, int amp3ObjectId);
        //void FillData(object sender, EventArgs e);
    }

    public class TabData
    {
        private readonly object objValue;
        private readonly string strDisplayText;
        private readonly string strName;

        public TabData(string name, string displayText, object value)
        {
            strName = name;
            strDisplayText = displayText;
            objValue = value;
        }

        public TabData(string name, object value) : this(name, value.ToString2(), value)
        {
        }

        public string Name
        {
            get { return strName; }
        }

        public string DisplayText
        {
            get { return strDisplayText; }
        }

        public object Value
        {
            get { return objValue; }
        }
    }

    public class TabPreSaveCheckStatus
    {
        private readonly bool bIsSuccess;
        private readonly string strTabMessage;

        public TabPreSaveCheckStatus(string TabMessage, bool success)
        {
            bIsSuccess = success;
            strTabMessage = TabMessage;
        }

        public string TabMessage
        {
            get { return strTabMessage; }
        }

        public bool IsSuccess
        {
            get { return bIsSuccess; }
        }
    }


    public class TabObjectMapInfo
    {
        private readonly AMP3Object enmAMP3ObjectType;
        private readonly AMP3Object enmAMP3ParentObjectType;
        private readonly int nAMP3ObjectId;
        private readonly Dictionary<string, object> objTabDataCollection;
        private readonly string strAMP3ParentId;

        public TabObjectMapInfo(AMP3Object amp3ObjectType, int amp3ObjectId, AMP3Object AMP3ParentObjectType,
            string AMP3ParentObectId, Dictionary<string, object> tabData, SaveStatus saveStatus)
        {
            enmAMP3ObjectType = amp3ObjectType;
            nAMP3ObjectId = amp3ObjectId;
            objTabDataCollection = tabData;
            SaveStatus = saveStatus;
            enmAMP3ParentObjectType = AMP3ParentObjectType;
            strAMP3ParentId = AMP3ParentObectId;
        }

        public AMP3Object AMP3ObjectType
        {
            get { return enmAMP3ObjectType; }
        }

        public int AMP3ObjectId
        {
            get { return nAMP3ObjectId; }
        }

        public AMP3Object AMP3ParentObjectType
        {
            get { return enmAMP3ParentObjectType; }
        }

        public string AMP3ParentObjectId
        {
            get { return strAMP3ParentId; }
        }


        //public AMP3Attribute AMP3Attribute
        //{
        //    get { return this.enmAMP3Attribute; }
        //}

        public Dictionary<string, object> TabData
        {
            get { return objTabDataCollection; }
        }

        public NonQueryStatus NonQueryStatus { get; set; }

        public SaveStatus SaveStatus { get; set; }
    }
}

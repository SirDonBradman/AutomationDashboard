using System;

namespace Aurigo.AMP3.LinksDTO
{
    public class LinkDetails
    {
        #region variables

        #endregion

        #region Constructor

        public LinkDetails()
        {
            LinkID = 0;
            LinkName = string.Empty;
            LinkURL = string.Empty;
            DestName = string.Empty;
            SrcID = string.Empty;
            SrcType = string.Empty;
            DestID = string.Empty;
            DestType = string.Empty;
            CreatedBy = string.Empty;
            CreatedDate = DateTime.UtcNow;
            Description = string.Empty;
            VersionNumber = string.Empty;
            DocSize = string.Empty;
            SignedDocID = 0;
        }

        #endregion

        #region Properties

        /// <summary>
        /// LinkID Property
        /// </summary>
        public int LinkID { get; set; }

        /// <summary>
        /// LinkName Property
        /// </summary>
        public string LinkName { get; set; }

        /// <summary>
        /// DestName Property
        /// </summary>
        public string DestName { get; set; }

        /// <summary>
        /// SrcID Property
        /// </summary>
        public string SrcID { get; set; }

        /// <summary>
        /// SrcType Property
        /// </summary>
        public string SrcType { get; set; }

        /// <summary>
        /// CreatedDate Property
        /// </summary>
        public DateTime CreatedDate { get; set; }

        /// <summary>
        /// DestID Property
        /// </summary>
        public string DestID { get; set; }

        /// <summary>
        /// DestType Property
        /// </summary>
        public string DestType { get; set; }

        /// <summary>
        /// CreatedBy Property
        /// </summary>
        public string CreatedBy { get; set; }

        /// <summary>
        /// Description Property
        /// </summary>
        public string Description { get; set; }


        public string VersionNumber { get; set; }
        public string LinkURL { get;  set; }

        ///<summary>
        /// Document Size Property
        /// </summary>
        public string DocSize;

        ///<summary>
        /// Id of Signed Document
        /// </summary>
        public int SignedDocID;
        #endregion
    }
}

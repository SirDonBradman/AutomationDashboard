using System;
using System.Data;
using System.Web.UI;
using Aurigo.AMP3.Common;
using Aurigo.AMP3.ProjectBL;
using Aurigo.AMP3.Resources.MessageResources;

namespace Aurigo.AMP3.EnterpriseUI
{
    public partial class ProjectSummaryByStatus : UserControl
    {
        protected void Page_Load(object sender, EventArgs e)
        {
            ProjectManager projectBL = ProjectManager.Instance;
            UserDetail ud = UserDetail.GetCurrentUserDetail();
            DataTable dt = projectBL.GetProjectCountByStatus("", ud.UID, ud.RID);

            if (dt.Rows.Count > 0)
            {
                lblErr.Text = "";
                ChartProjectStatus.Data.DataSource = dt;
                ChartProjectStatus.Data.DataBind();
                ChartProjectStatus.Visible = true;
                //ChartProjectStatus.Tooltips.Format = Infragistics.UltraChart.Shared.Styles.TooltipStyle.Custom;
                //ChartProjectStatus.Tooltips.FormatString = "Project Count: <DATA_VALUE>";
            }
            else
            {
                ChartProjectStatus.Visible = false;
                lblErr.Text = MessageResourceManager.GetString("W_ENTPRSE_NO_PROJECTS_AVAILABLE",
                    Enumerations.MessageType.WarningMessage);
            }
        }
    }
}

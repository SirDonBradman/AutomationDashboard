using System;
using System.Collections.Generic;
using System.ComponentModel;
using System.Linq;
using System.Web;
using System.Xml.Serialization;

namespace Aurigo.Brix.Platform.BusinessLayer.XMLForm
{
    public class Section : ControlContainer
    {
        [XmlAttribute("Caption")]
        [FormDesignerAttribute(Priority = 10001)]
        public string Caption;

        [XmlIgnore]
        public bool IsSectionDisabled;
        [XmlAttribute("Orientation"), DefaultValue(0)]
        public Orientation Orientation;


        [XmlAttribute("Stage"), DefaultValue("")]
        public string Stage;

        [XmlAttribute("StageName"), DefaultValue("")]
        public string StageName;

        [XmlIgnore]
        public int level;


        public override void Render(object parent, RenderFormat format)
        {
            // get the delegate that actuall renders the control from the registered renderers.  For now we hard
            // code htmlrender since reports dont work through this model yet.

            if (form.Renderer.ContainerRenderers[(int)ContainerTypes.Section] != null)
                form.Renderer.ContainerRenderers[(int)ContainerTypes.Section](this, parent);
        }
    }

}

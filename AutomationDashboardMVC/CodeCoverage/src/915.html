using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Web.UI.WebControls;
using Infragistics.WebUI.WebSchedule;
using Aurigo.Brix.Platform.CoreUtilities.Utility;
using Aurigo.AMP3.Common;
using Infragistics.WebUI.WebDataInput;
using Aurigo.Brix.Platform.BusinessLayer.CustomControls;
using System.Web.UI;

namespace Aurigo.Brix.Platform.BusinessLayer.XmlForm_Framework
{
    static class Helper
    {
        public static void RenderAsLabel(WebControl wc, System.Web.UI.HtmlTextWriter writer, string text,
            Unit width, Unit height, bool hide = false, string attributes = "")
        {
            Label lbl = new Label();
            lbl.ID = wc.ID + (hide ? "_Value" : "_Text");

            // To Fix line break Issue:: Replacing "&lt;br/&gt;" to "<br/>" after encoding
            lbl.Text = System.Web.HttpUtility.HtmlEncode(text).Replace("&lt;br/&gt;", "<br/>");
            if (hide)
                lbl.Style.Add(System.Web.UI.HtmlTextWriterStyle.Display, "none");
            lbl.Width = width;
            if (!string.IsNullOrEmpty(attributes))
                lbl.Attributes.Add("style", attributes);
            lbl.Style.Add("word-wrap", "break-word");
            lbl.Style.Add("overflow", "auto");
            lbl.Style.Add("text-align", "left");
            lbl.Style.Add("vertical-align", "middle");
            lbl.RenderControl(writer);
        }
    }
}

using System;
using System.Collections.Generic;
using System.Xml.Serialization;

namespace Aurigo.Brix.Platform.BusinessLayer.EmailServiceFramework
{
    [Serializable]
    [XmlRoot("EmailTemplates", IsNullable = false)]
    public class EmailTemplates
    {
        [XmlElement("Port")] public string Port;
        [XmlElement("SmtpHost")] public string SmtpHost;

        [XmlElement("EmailTemplate")] public List<EmailTemplate> emailTemplates = new List<EmailTemplate>();
    }

    [Serializable]
    public class EmailTemplate
    {
        [XmlElement("Body")] public string Body;
        [XmlElement("From")] public string From;
        [XmlElement("Inputs")] public Inputs InputClasses;
        [XmlAttribute("Name")] public string Name;
        [XmlElement("Subject")] public string Subject;
        [XmlElement("SubstitutionHandler")] public SubstitutionHandler SubstitutionHandlerClass;
    }

    [Serializable]
    public class SubstitutionHandler
    {
        [XmlAttribute("ClassName")] public string Name;
    }

    [Serializable]
    public class Inputs
    {
        [XmlElement("Input")] public List<Input> InputList;
    }

    [Serializable]
    public class Input
    {
        [XmlAttribute("ClassName")] public string ClassName;
        [XmlAttribute("Name")] public string Name;
    }

    [Serializable]
    [XmlRoot("Esculations", IsNullable = false)]
    public class Esculations
    {
        [XmlElement("Escalation")] public List<Escalation> escalations = new List<Escalation>();
    }

    [Serializable]
    public class Escalation
    {
        [XmlAttribute("DateColumn")] public string DateColumn;
        [XmlAttribute("StatusValue")] public string IsFromProject;
        [XmlAttribute("ModuleID")] public string ModuleID;
        [XmlAttribute("Name")] public string Name;
        [XmlAttribute("PLName")] public string PLName;
        [XmlAttribute("PrimaryKeyName")] public string PrimaryKeyName;
        [XmlAttribute("SPName")] public string SPName;
        [XmlAttribute("StatusColumn")] public string StatusColumn;
        [XmlAttribute("IsFromProject")] public string StatusValue;
        [XmlAttribute("TableName")] public string TableName;
        [XmlElement("Filter")] public List<Filter> filters = new List<Filter>();
        [XmlElement("Step")] public List<Step> steps = new List<Step>();
    }

    [Serializable]
    public class Filter
    {
        [XmlAttribute("Column")] public string Column;
        [XmlAttribute("Value")] public string Value;
    }

    [Serializable]
    public class Step
    {
        [XmlAttribute("AgedDays")] public int AgedDays;
        [XmlAttribute("EmailTemplate")] public string EmailTemplate;
        [XmlAttribute("UserId")] public string UserId;
    }
}

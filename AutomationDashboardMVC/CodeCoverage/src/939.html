using System;
using System.Collections.Generic;
using System.ComponentModel;
using System.Linq;
using System.Web;
using System.Xml.Serialization;

namespace Aurigo.Brix.Platform.BusinessLayer.XMLForm
{
    public class Picker : ControlContainer
    {
        [XmlAttribute("DataSource"), DefaultValue("")]
        public string DataSource;

        [XmlAttribute("PageSize"), DefaultValue("")]
        public string PageSize;

        [XmlAttribute("KeyColumn"), DefaultValue("")]
        public string KeyColumn;

        [XmlAttribute("DisplayColumn"), DefaultValue("")]
        public string DisplayColumn;

        [XmlAttribute("Caption"), DefaultValue("")]
        public string Caption;

        [XmlAttribute("MobileFilterKey"), DefaultValue("")]
        public string MobileFilterKey;

        [XmlAttribute("DataSourceURL"), DefaultValue("/api/GenericXMLPicker/")]
        public string DataSourceURL =
            "/api/GenericXMLPicker/";

        [XmlAttribute("DataSourceMethodType"), DefaultValue("GET")]
        public string DataSourceMethodType = "GET";

        public override void Render(object parent, RenderFormat format)
        {
            if (form.Renderer.ContainerRenderers[(int)ContainerTypes.Picker] != null)
                form.Renderer.ContainerRenderers[(int)ContainerTypes.Picker](this, parent);
        }
    }

}

using System;
using System.Collections.Generic;
using System.ComponentModel;
using System.Linq;
using System.Web;
using System.Xml.Serialization;

namespace Aurigo.Brix.Platform.BusinessLayer.XMLForm
{
    public class xColumnGroup
    {
        /// <summary>
        /// Name of the column group, which will be associated to the columns
        /// </summary>
        [XmlAttribute("Name"), DefaultValue("")]
        public string Name { get; set; }

        /// <summary>
        /// Header text of the column group
        /// </summary>
        [XmlAttribute("Caption"), DefaultValue("")]
        public string Caption { get; set; }

        /// <summary>
        /// Name of the parent column group
        /// </summary>
        [XmlAttribute("ParentColumnGroup"), DefaultValue("")]
        public string ParentColumnGroup { get; set; }

        [XmlAttribute("Align"), DefaultValue(0)]
        public GridHeaderAlignment Align { get; set; }
    }

    public class xColumnGroups
    {
        [XmlElement("ColumnGroup")]
        public List<xColumnGroup> ColumnGroup = new List<xColumnGroup>();
    }
}

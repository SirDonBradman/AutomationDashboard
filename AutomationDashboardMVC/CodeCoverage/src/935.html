using System;
using System.Collections.Generic;
using System.ComponentModel;
using System.Linq;
using System.Web;
using System.Xml.Serialization;

namespace Aurigo.Brix.Platform.BusinessLayer.XMLForm
{
    public class Include : ControlContainer
    {
        [XmlAttribute("FormName"), DefaultValue("")]
        public string FormName;
        [XmlAttribute("NamePrefix"), DefaultValue("")]
        public string NamePrefix;

        /// <summary>
        /// Comma seperated keys which can be sent to controls in the shared xml 
        /// Eg: Module id of the attachment control - ModuleID="{_CUSTOM:[0]}" - first key in additional key array
        /// Eg: Value of the control - Value="{_CUSTOM:[1]}" - second key in the additional key array
        /// </summary>
        [XmlAttribute("AdditionalKeys"), DefaultValue("")]
        public string AdditionalKeys;

        public override void Render(object parent, RenderFormat format)
        {
            base.Render(parent, format);
        }
    }
}

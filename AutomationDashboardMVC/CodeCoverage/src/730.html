using System.Collections.Generic;
using Aurigo.AMP3.DataAccess.Core;
using Aurigo.AMP3.ModuleManagementDAC;
using Aurigo.Brix.Platform.BusinessLayer.DTO;
using Aurigo.Common.Utility;

namespace Aurigo.AMP3.ModuleManagementBL
{
    public class ModuleUploadManager : SingletonManagerBase<ModuleUploadManager>
    {
        private ModuleUploadManager()
        {
        }

        public void LogModuleUploadMaster(ModuleUploadMaster dto)
        {
            ModuleManagementComponent.Instance.LogModuleUploadMaster(dto);
        }

        public void LogModuleUploadDetails(ModuleUploadDetails dto)
        {
            ModuleManagementComponent.Instance.LogModuleUploadDetails(dto);
        }

        public void ClearModuleUploadLogs()
        {
            ModuleManagementComponent.Instance.ClearModuleUploadLogs();
        }

        public List<ModuleUploadDetails> GetModuleUploadLogs(int? logID = null)
        {
            return ModuleManagementComponent.Instance.GetModuleUploadLogs(logID);
        }
    }
}

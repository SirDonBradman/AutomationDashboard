using Aurigo.Brix.Platform.CoreUtilities.Utility;

namespace Aurigo.Brix.Platform.BusinessLayer.BL
{
    public class SizeConverter
    {
        public static double ConvertToGB(string number)
        {
            if (number.EndsWith("MB"))
            {
                double no = number.ToString().TrimEnd("MB".ToCharArray()).ToDouble2();
                return ConvertMegabytesToGigabytes(no);
            }
            else if (number.EndsWith("GB"))
            {
                return number.ToString().TrimEnd("GB".ToCharArray()).ToDouble2();
            }
            else if (number.EndsWith("TB"))
            {
                double no = number.ToString().TrimEnd("TB".ToCharArray()).ToDouble2();
                return ConvertTerabytesToGigabytes(no);
            }
            else
                return 0;
        }

        private static double ConvertMegabytesToGigabytes(double megabytes)
        {
            // 1024 megabyte in a terabyte
            return megabytes/1024.0;
        }

        private static double ConvertTerabytesToGigabytes(double terabytes)
        {
            // 1024 gigabytes in a terabyte
            return terabytes*1024.0;
        }
    }
}

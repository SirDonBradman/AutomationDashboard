using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using Aurigo.Brix.Platform.BusinessLayer.AbstractModels;
using Aurigo.Brix.Platform.CoreUtilities.Utility;
using Aurigo.AMP3.Common;
using System.Data;
using Aurigo.AMP3.DataAccess.Core;
using Aurigo.Masterworks.Construction.ContractManager.PurchaseOrder.BL;

namespace Aurigo.Masterworks.BYU.Models
{
    [ContextAttribute(Name = "BYUBFLK")]
    public class BeforeContractLockModel : AfterSaveRecordModel
    {
        public override void SaveDetails(object recordDetails, System.Web.UI.Page page, string mode, string callerContext, object additionalParameters = null)
        {

            if (callerContext == "AFTER_CONTRACT_LOCK")
            {
                //Insert into Activity dates table
                int contractId = recordDetails.ToString().ToInt32_2();
                PurchaseOrderManager.Instance.CreateActivityDatesForContractItem(contractId);
            }
            else if (callerContext == "AFTER_CONTRACT_UNLOCK")
            {
                int contractId = recordDetails.ToString().ToInt32_2();
                PurchaseOrderManager.Instance.DeleteActivityDatesForContractItem(contractId);
                
            }
        }


    }
}

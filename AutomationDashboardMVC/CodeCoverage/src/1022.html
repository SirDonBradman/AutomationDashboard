using System;
using System.Collections.Generic;
using System.Linq;
using System.Reflection;
using System.Web;

namespace Aurigo.Brix.Platform.BusinessLayer.DTO
{
    public class MobileMenuDTO_Extended : MobileMenuDTO
    {
        public MobileMenuDTO_Extended()
        {
        }
        public MobileMenuDTO_Extended(MobileMenuDTO menu)
        {
            foreach (PropertyInfo baseProp in menu.GetType().GetProperties())
            {
                PropertyInfo derivedProp = menu.GetType().GetProperty(baseProp.Name);
                derivedProp.SetValue(this, baseProp.GetValue(menu, null), null);
            }
        }
        public string ClassAttribute { get; set; }
        public bool HasError { get; set; }
        public string ErrorMessage { get; set; }
    }
}

using System;
using System.Collections.Generic;

namespace Aurigo.AMP3.ProjectDTO
{
    /// <summary>
    /// Author: Kalyan Chakradhar R.
    /// Description: Data Tranfer Object for Project.
    /// </summary>
    public class Project
    {
        private List<AttributeClass> projectAttributes = new List<AttributeClass>();

        public Project()
        {
            ProjectAttributes = new Dictionary<string, string>();
            ProjectModules = new List<Platform>();
        }

        /// <summary>
        /// Gets and Sets ERP Company
        /// </summary>
        public string AXCompanyName { get; set; }

        /// <summary>
        /// Gets and Sets ERP Currency
        /// </summary>
        public string ERPCurrency { get; set; }


        /// <summary>
        /// Gets and Sets Project Mode
        /// </summary>
        public char ProjectMode { get; set; }


        /// <summary>
        /// Gets and Sets Project ID
        /// </summary>
        public int ProjectID { get; set; }

        /// <summary>
        /// Gets and Sets Project Code
        /// </summary>
        public string ProjectCode { get; set; }

        /// <summary>
        /// Gets and Sets Project Status Name
        /// </summary>
        public string ProjectStatusName { get; set; }

        /// <summary>
        /// Gets and Sets Project Name
        /// </summary>
        public string ProjectName { get; set; }

        /// <summary>
        /// Gets and Sets Project Description
        /// </summary>
        public string ProjectDescription { get; set; }

        /// <summary>
        /// Gets and Sets Project Created Date
        /// </summary>
        public DateTime ProjectCreatedDate { get; set; }

        /// <summary>
        /// Gets and Sets Project Owner
        /// </summary>
        public string ProjectOwner { get; set; }

        /// <summary>
        /// Gets and Sets Program Type
        /// </summary>
        public string ProgramYear { get; set; }

        /// <summary>
        /// Gets and Sets Program Type
        /// </summary>
        public string ProjectClass { get; set; }

        /// <summary>
        /// Gets and Sets Program Type
        /// </summary>
        public string ProjectClassName { get; set; }

        /// <summary>
        /// Gets and Sets ExpenditureCurveID
        /// </summary>
        public int ExpenditureCurveID { get; set; }

        /// <summary>
        /// Gets and Sets CalendarType
        /// </summary>
        public int CalendarType { get; set; }

        /// <summary>
        /// Gets and Sets CalendarDays
        /// </summary>
        public int CalendarDays { get; set; }

        /// <summary>
        /// Gets and Sets Project Start Date
        /// </summary>
        public DateTime ProjectStartDate { get; set; }

        /// <summary>
        /// Gets and Sets Project End Date
        /// </summary>
        public DateTime ProjectEndDate { get; set; }

        /// <summary>
        /// Gets and Sets user defined Project Status
        /// </summary>
        public int ProjectStatus { get; set; }

        /// <summary>
        /// Gets and Sets user defined Project Value
        /// </summary>
        public decimal ProjectValue { get; set; }

        /// <summary>
        /// Gets and Sets Project Active Status
        /// </summary>
        public bool ProjectIsActive { get; set; }

        /// <summary>
        /// Gets and Sets Project Template ID
        /// </summary>
        public int ProjectTemplateID { get; set; }
        
        /// <summary>
        /// Gets and Sets the IsPublished flag
        /// </summary>
        public bool IsPublished { get; set; }

        /// <summary>
        /// Gets and Sets Project Modules
        /// </summary>
        public List<Platform> ProjectModules { get; set; }

        /// <summary>
        /// Gets and Sets Modified By
        /// </summary>
        public string ModifiedBy { get; set; }

        /// <summary>
        /// Gets and Sets Modified On
        /// </summary>
        public DateTime? ModifiedOn { get; set; }

        public int LandMasterID { get; set; }

        public string LandMasterName { get; set; }

        public int? DocumentTemplateID { get; set; }

        public string DocumentTemplateName { get; set; }

        public int? DocumentMetadataID { get; set; }

        public string DocumentMetadataName { get; set; }

        public bool? IsDeleted { get; set; }

        public int? ProjectCalendarID { get; set; }

        public string ProjectCalendarName { get; set; }

        /// <summary>
        /// Gets and Sets Project Attributes
        /// </summary>
        public Dictionary<string, string> ProjectAttributes { get; set; }

        public List<AttributeClass> ProjectAttributesList
        {
            get { return projectAttributes; }
            set { projectAttributes = value; }
        }

        public string DocumentListID { get; set; }

        /// <summary>
        /// Gets and Sets user defined Project Score
        /// </summary>
        public double ProjectScore { get; set; }

        /// <summary>
        /// This has to be nullable for now to support backward compatibility with
        /// existing projects
        /// </summary>
        public int? BusinessUnitID { get; set; }

        /// <summary>
        /// full path of the Business Unit
        /// </summary>
        public string BusinessUnitPath { get; set; }

        public string Timestamp { get; set; }

        #region Nested type: AttributeClass

        public struct AttributeClass
        {
            public int AttributeId;
            public string AttributeName;
            public string AttributeNotes;
            public string AttributeType;
            public string AttributeValue;
        }

        #endregion
    }

    [Serializable]
    public struct ReportParams
    {
        public bool? GetActiveProjects;
        public int? ProjectID;
        public string ReportLogoPath;
        public string ReportPath;
        public int? UserID;
    }

    public enum RPTName
    {
        ProjectStatusReport,
        ActiveProjectsList,
        InActiveProjectsList,
        ProjectSubmittalStatus
    }
}

using System;
using System.Data.Linq.Mapping;

namespace Aurigo.Brix.Platform.CoreUtilities.DTO
{
    [Table(Name = "WorkflowUsers")]
    [Serializable]
    public class WorkflowUser
    {
        [Column(Name = "ID", DbType = "Int", IsPrimaryKey = true, IsDbGenerated = true)]
        public string ID { get; set; }

        [Column(Name = "UserID", DbType = "varchar(100)")]
        public string UserID { get; set; }

        [Column(Name = "RoleName", DbType = "varchar(100)")]
        public string RoleName { get; set; }

        [Column(Name = "State", DbType = "varchar(100)")]
        public string State { get; set; }

        [Column(Name = "WorkflowID", DbType = "int")]
        public int WorkflowID { get; set; }

        [Column(Name = "Completed", DbType = "bit")]
        public bool Completed { get; set; }

        [Column(Name = "Expression", DbType = "varchar(500)")]
        public string Expression { get; set; }
    }
}

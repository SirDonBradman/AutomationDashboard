using Aurigo.AMP3.Common;
using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using System.Web.UI;
using System.Web.UI.WebControls;

namespace Aurigo.Brix.Platform.UI
{
    public partial class AccessDenied : System.Web.UI.Page
    {
        protected void Page_Load(object sender, EventArgs e)
        {
            /*
            * This page has access to the user in case of only ADFS has been enabled.
            */
            if (!AMP3ApplicationSettings.Instance.AdfsOnlyEnabled)
            {
                Response.Redirect("~/Default.aspx");
            }

            divAccessDenied.Visible = !IsLicenseExceeded;
            divLicense.Visible = IsLicenseExceeded;
        }

        public bool IsLicenseExceeded
        {
            get
            {
                var exceeded = Request.QueryString["le"];

                return !string.IsNullOrEmpty(exceeded) && exceeded == "1";
            }

        }
    }
}

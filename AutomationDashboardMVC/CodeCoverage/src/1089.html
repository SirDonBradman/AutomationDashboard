using Aurigo.Brix.Platform.BusinessLayer.Utility;
using System.Net;
using System.Net.Http;
using System.Web.Http;

namespace Aurigo.Brix.Platform.BusinessLayer.Controller
{
    public class DocumentConverterController : ApiController
    {
        [HttpGet]
        public HttpResponseMessage ConvertToPDF(string inputFilePath, string outputFilePath)
        {
            if (string.IsNullOrWhiteSpace(inputFilePath) || string.IsNullOrWhiteSpace(outputFilePath))
                return Request.CreateResponse(HttpStatusCode.PreconditionFailed, "Document File path missing");

            var client = new LeadHttpClient();

            //Call the LeadTool API
            HttpResponseMessage response = client.GetResponse(string.Format("/api/Converter/ConvertToPDF?inputFilePath={0}&outputFilePath={1}", inputFilePath, outputFilePath));

            return response;
        }
    }
}

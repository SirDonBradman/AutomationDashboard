namespace Aurigo.AMP3.DocumentManagementDTO
{
    public enum FolderType
    {
        None = 0,
        UserFolder = 1,
        SystemFolder = 2,
        ArchiveFolder = 3
    }

    public class Folder
    {
        //Properties
        public int FolderId { get; set; }

        public string FolderName { get; set; }

        public string FolderDesc { get; set; }

        public int ParentId { get; set; }

        public string ModuleId { get; set; }

        public string ParentModuleId { get; set; }

        public int? ParentInstanceId { get; set; }

        public string InstanceId { get; set; }

        public string ListID { get; set; }

        public string ListPath { get; set; }

        public bool CreateSharePointFolder { get; set; }

        public int RootModuleFolderId { get; set; }

        public int? PermissionFolderID { get; set; }

        public FolderType FolderType { get; set; }

        public Folder()
        {
            CreateSharePointFolder = true;
        }
    }
}

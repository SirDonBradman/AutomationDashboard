using System.Collections.Generic;
using System.Data;
using Aurigo.AMP3.DataAccess.Core;
using Aurigo.Common.Utility;

namespace Aurigo.Brix.Platform.BusinessLayer.BL
{
    public class ProcurementAdjustments : SingletonManagerBase<ProcurementAdjustments>
    {
        private ProcurementAdjustments()
        {
        }
        
        public DataTable GetAdjustmentsFromLibrary(string filterids, string filterText, string sortBy, int rowStartIdx,
            int rowEndIdx)
        {
            DataSet ds = ProcurementComponent.Instance.GetAdjustmentsFromLibrary(filterids, filterText, sortBy,
                rowStartIdx, rowEndIdx);
            if (ds != null && ds.Tables.Count > 0)
                return ds.Tables[0];
            else
                return null;
        }
    }
}

using Aurigo.AMP3.Common;
using Aurigo.AMP3.Core.UserControls;
using Aurigo.Brix.Core.WebUiHelper;
using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using System.Web.UI;
using System.Web.UI.WebControls;

namespace Aurigo.Construction.MainModules.ResourceManagement
{
    public partial class ResourceGantt : BrixPageBase
    {
        protected override void OnPreInit(EventArgs e)
        {
            ModuleId = "ENTPRSE";
            ParentModuleID = "ENTPRSE";
            base.OnPreInit(e);
        }
        protected override void OnLoad(EventArgs e)
        {
            Page.ClientScript.RegisterClientScriptBlock(this.GetType(), "QueryStringParam", string.Format("var pid = '{0}'; var context='{1}'; var parentid='{2}';", UIHelper.JavascriptEncode(Request.QueryString["PID"]), UIHelper.JavascriptEncode(Request.QueryString["Context"]), UIHelper.JavascriptEncode(Request.QueryString["ParentID"])), true);
            PageTitle = "Resource Gantt";
            base.OnLoad(e);
        }
    }
}

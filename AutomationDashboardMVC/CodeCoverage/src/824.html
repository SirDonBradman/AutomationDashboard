using System;
using System.ComponentModel;
using System.Data.Linq.Mapping;

namespace Aurigo.AMP3.Core.BusinessLayer.DTO
{
    [Table(Name = "dbo.CORITEMResourceDetails")]
    public class Resource : INotifyPropertyChanging, INotifyPropertyChanged
    {
        #region Fields

        private Decimal? _DiffQuantity = 0; //By Default, DiffQuantity represents no change in Quantity

        #endregion

        #region Properties

        /// <summary>
        /// The database automatically generates this value
        /// </summary>
        [Column(Name = "ResourceDetailID", DbType = "int", IsDbGenerated = true, IsPrimaryKey = true, CanBeNull = false)
        ]
        public Int32? ResourceDetailID { get; set; }

        [Column(Name = "ModuleID", DbType = "char(7)", CanBeNull = false)]
        public String ModuleID { get; set; }

        [Column(Name = "ParentInstanceID", DbType = "int", CanBeNull = false)]
        public Int32 ParentInstanceID { get; set; }

        public string Unit { get; set; }

        [Column(Name = "ItemID", DbType = "int", CanBeNull = false)]
        public Int32 ItemID { get; set; }

        /// <summary>
        /// This is not a required field
        /// </summary>
        [Column(Name = "ResourceTypeID", DbType = "smallint", CanBeNull = true)]
        public Int16? ResourceTypeID { get; set; }

        /// <summary>
        /// This is not a required field
        /// </summary>
        [Column(Name = "Description", DbType = "varchar(8000)", CanBeNull = true)]
        public String Description { get; set; }

        public String ResourceTypeName { get; set; }
        
        /// <summary>
        /// This is not a required field
        /// </summary>
        [Column(Name = "UnitID", DbType = "int", CanBeNull = true)]
        public Int64? UnitID { get; set; }

        /// <summary>
        /// This is not a required field
        /// </summary>
        [Column(Name = "Rate", DbType = "decimal(18,4)", CanBeNull = true)]
        public Decimal? Rate { get; set; }

        /// <summary>
        /// This is not a required field
        /// </summary>
        [Column(Name = "WastagePercent", DbType = "decimal(18,4)", CanBeNull = true)]
        public Decimal? WastagePercent { get; set; }

        [Column(Name = "WastageQty", DbType = "decimal(18,4)", CanBeNull = true)]
        public Decimal? WastageQty { get; set; }

        [Column(Name = "ReworkQty", DbType = "decimal(18,4)", CanBeNull = true)]
        public Decimal? ReworkQty { get; set; }

        [Column(Name = "RegularQty", DbType = "decimal(18,4)", CanBeNull = true)]
        public Decimal? RegularQty { get; set; }

        /// <summary>
        /// This is not a required field
        /// </summary>
        [Column(Name = "Quantity", DbType = "decimal(18,4)", CanBeNull = true)]
        public Decimal? Quantity { get; set; }

        /// <summary>
        /// This is not a required field
        /// </summary>
        [Column(Name = "Cost", DbType = "decimal(18,4)", CanBeNull = true)]
        public Decimal? Cost { get; set; }

        /// <summary>
        /// This is not a required field
        /// </summary>
        [Column(Name = "Notes", DbType = "varchar(8000)", CanBeNull = true)]
        public String Notes { get; set; }

        /// <summary>
        /// This is not a required field
        /// </summary>
        [Column(Name = "ResItemID", DbType = "varchar(50)", CanBeNull = true)]
        public String ResItemID { get; set; }

        [Column(Name = "ParentResItemID", DbType = "varchar(250)", CanBeNull = true)]
        public String ParentResItemID { get; set; }

        public String Prorated { get; set; }

        [Column(Name = "IsCritical", DbType = "bit", CanBeNull = false)]
        public Boolean IsCritical { get; set; }

        [Column(Name = "MaterialIssueNo", DbType = "varchar(100)", CanBeNull = true)]
        public String MaterialIssueNo { get; set; }

        [Column(Name = "AccountNumber", DbType = "char(10)", CanBeNull = true)]
        public String AccountNumber { get; set; }

        [Column(Name = "Configuration", DbType = "char(10)", CanBeNull = true)]
        public String Configuration { get; set; }

        [Column(Name = "Size", DbType = "char(10)", CanBeNull = true)]
        public String Size { get; set; }

        [Column(Name = "Color", DbType = "char(10)", CanBeNull = true)]
        public String Color { get; set; }

        [Column(Name = "TradeAgmtCurrency", DbType = "char(10)", CanBeNull = true)]
        public String Currency { get; set; }

        [Column(Name = "CreatedDate", DbType = "datetime", CanBeNull = true)]
        public DateTime CreatedDate { get; set; }

        public Decimal? DiffQuantity
        {
            get { return _DiffQuantity; }
            set { _DiffQuantity = value; }
        }

        /// <summary>
        /// This field will be populated only for Personnel. Can be Internal, Contractor or Consultant. Other cases it will be NA.
        /// </summary>
        [Column(Name = "ResourceType", DbType = "varchar(100)", CanBeNull = true)]
        public string ResourceType { get; set; }

        /// <summary>
        /// This field will be populated only for Personnel. Can be Hour, Day, Month or Year. Other cases it will be null.
        /// </summary>
        [Column(Name = "UnitForPersonnel", DbType = "varchar(100)", CanBeNull = true)]
        public string UnitForPersonnel { get; set; }

        [Column(Name = "VariablePrice", DbType = "decimal(18,4)", CanBeNull = true)]
        public decimal VariablePrice { get; set; }
        
        [Column(Name = "UtilizationPercentage", DbType = "decimal(18,4)", CanBeNull = true)]
        public decimal UtilizationPercentage { get; set; }

        [Column(Name = "Count", DbType = "decimal(18,4)", CanBeNull = true)]
        public decimal Count { get; set; }

        [Column(Name = "AddedManually", DbType = "varchar(3)", CanBeNull = true)]
        public string AddedManually { get; set; }

        #endregion

        private static readonly PropertyChangingEventArgs emptyChangingEventArgs =
            new PropertyChangingEventArgs(String.Empty);

        #region INotifyPropertyChanged Members

        public event PropertyChangedEventHandler PropertyChanged;

        #endregion

        #region INotifyPropertyChanging Members

        public event PropertyChangingEventHandler PropertyChanging;

        #endregion

        protected internal virtual void SendPropertyChanging()
        {
            if ((PropertyChanging != null))
            {
                PropertyChanging(this, emptyChangingEventArgs);
            }
        }

        protected internal virtual void SendPropertyChanged(String propertyName)
        {
            if ((PropertyChanged != null))
            {
                PropertyChanged(this, new PropertyChangedEventArgs(propertyName));
            }
        }
    }
}

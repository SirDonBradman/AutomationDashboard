using System;
using Aurigo.AMP3.Common;
using Aurigo.AMP3.Core.UserControls;
using Aurigo.Brix.Core.WebUiHelper;
using Aurigo.AMP3.Resources.TerminologyResources;
using System.Collections.Generic;

namespace Aurigo.Brix.Construction.PlanningManagement
{
    public partial class PlanningDashboard : BrixPageBase
    {
        protected NewDashboard DBControl;

        protected override void OnLoad(EventArgs e)
        {
            UpdateTree = true;

            List<WidgetPage> temp = new List<WidgetPage>();
            temp.Add(new WidgetPage() { Caption = "Dashboard", PageNo = 1 });

            UIHelper.CreateDashboadTabs(1, PageTabBar, temp, 1, "Dashboard");

            if (PageTabBar.Tabs.Count == 0)
                PageTabBar.Tabs.Add(new Tab("Planning Dashboard", "#"));

            base.OnLoad(e);
        }

        protected override void OnInit(EventArgs e)
        {
            PermissionsToCheck.Add(Constants.MODFEAT_VISIBILITY);

            PageTitle = LocalizationManager.GetString("Planning") + " Dashboard";
            
            DBControl.ParentPagePath = "~/Modules/PLANING/Dashboard.aspx";

            base.OnInit(e);

            ValidateContext();
        }

        private void ValidateContext()
        {
            if (Request["Context"].ToLower() != "planing")
            {
                OnAccessDeniedRedirect();
            }
        }
    }
}

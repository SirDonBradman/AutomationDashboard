using Aurigo.Brix.Platform.BusinessLayer.XMLForm.FieldParsers;
using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;

namespace Aurigo.Brix.Platform.BusinessLayer.XMLForm
{
    public class ReferenceColumnDetails
    {
        public string OriginalFieldText_WithAlias { get; set; }
        public string FieldName { get; set; }
        public string AliasText { get; set; }

        public ReferenceColumnDetails() { }
        public ReferenceColumnDetails(ListColumnsTagParser.ColumnsAttributeItem columnDetails)
        {
            this.OriginalFieldText_WithAlias = columnDetails.OriginalFieldText_WithAlias;
            this.FieldName = columnDetails.FieldName;
            this.AliasText = columnDetails.AliasText;
        }
    }
}

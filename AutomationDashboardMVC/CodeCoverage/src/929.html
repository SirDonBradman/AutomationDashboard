using System;
using System.Collections.Generic;
using System.ComponentModel;
using System.Linq;
using System.Web;
using System.Xml.Serialization;

namespace Aurigo.Brix.Platform.BusinessLayer.XMLForm
{
    public class DataSourceObject
    {
        [XmlElement("Row")]
        public List<DataSourceRow> DataSourceRow = new List<DataSourceRow>();
    }

    public class DataSourceRow
    {
        [XmlElement("Column")]
        public List<xDataSourceColumn> DataSourceColumn = new List<xDataSourceColumn>();
    }

    public class xDataSourceColumn
    {
        /// <summary>
        /// Name of the column 
        /// </summary>
        [XmlAttribute("Name"), DefaultValue("")]
        public string Name { get; set; }

        /// <summary>
        /// Header text of the column group
        /// </summary>
        [XmlAttribute("Value"), DefaultValue("")]
        public string Value { get; set; }
    }
}

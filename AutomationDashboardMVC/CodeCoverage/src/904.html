using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Web.UI.WebControls;
using Infragistics.WebUI.WebSchedule;
using Aurigo.Brix.Platform.CoreUtilities.Utility;
using Aurigo.AMP3.Common;
using Infragistics.WebUI.WebDataInput;
using Aurigo.Brix.Platform.BusinessLayer.CustomControls;
using System.Web.UI;

namespace Aurigo.Brix.Platform.BusinessLayer.XmlForm_Framework
{
    public class XMLDateControl : WebDateChooser, IControl
    {
        public const string CONST_JS_FUNCTION_PATCH_DATEPICKER = "WebDateChooser_CalendarRenderDay";
        public bool RenderAsLabel { get; set; }
        protected override void Render(System.Web.UI.HtmlTextWriter writer)
        {
            if (RenderAsLabel)
            {
                string text = this.Text == null ? "" : this.Text;
                Helper.RenderAsLabel(this, writer, text, Width, Height, attributes: Style.Value);
                base.Style.Add("display", "none");
                base.Render(writer);
            }
            else
                base.Render(writer);
        }

        public XMLDateControl() : base()
        {
            ApplyPatch_For_DatePickerStyle(this);
        }

        public static void ApplyPatch_For_DatePickerStyle(WebDateChooser datePickerControl)
        {
            if (datePickerControl != null && datePickerControl.ClientSideEvents != null && datePickerControl.ClientSideEvents.CalendarRenderDay != CONST_JS_FUNCTION_PATCH_DATEPICKER)
            {
                datePickerControl.ClientSideEvents.CalendarRenderDay = CONST_JS_FUNCTION_PATCH_DATEPICKER;
                datePickerControl.CalendarLayout.ChangeMonthToDateClicked = true;
            }
        }

    }
}

using System.Collections.Generic;
using System.Data;
using Aurigo.AMP3.DataAccess.Core;
using Aurigo.Brix.Platform.BusinessLayer.DataAccessHelper;
using Aurigo.Common.Utility;

namespace Aurigo.AMP3.ConfigHandler
{
    public class Configuration : SingletonManagerBase<Configuration>
    {
        private Configuration()
        {
        }

        public DataTable GetConfig()
        {
            DataSet ds = ComponentHelper.Instance.ExecuteDataSet(StoredProcedure.usp_APPMGMTGetMenuNavigationTabs, null,
                null);
            if (ds.Tables.Count > 0)
                return ds.Tables[0];
            else
                return null;
        }
    }
}

using System;
using System.ComponentModel;
using System.Web.UI;
using System.Web.UI.WebControls;

namespace Aurigo.Brix.Platform.BusinessLayer.CustomControls
{
    [DefaultProperty("Text")]
    [ToolboxData("<{0}:IntegerBox runat=server></{0}:IntegerBox>")]
    public class IntegerBox : TextBox
    {
        public int? MinValue { get; set; }
        public int? MaxValue { get; set; }

        protected override void OnPreRender(EventArgs e)
        {
            Attributes.Add("onkeypress",
                "return IsValueInteger(event)");
            Attributes.Add("onblur",
                "if(!IsValueIntegerInRange(this.value,'" + (MinValue.HasValue ? MinValue : null) + "','" +
                (MaxValue.HasValue ? MaxValue : null) + "')) this.value=''");

            Style.Add("text-align", "right");
            CssClass = "tbC";
            base.OnPreRender(e);
        }
    }
}

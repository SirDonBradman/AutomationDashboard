using System;
using System.Collections.Generic;
using System.Web.UI;
using Aurigo.AMP3.Common;
using Aurigo.AMP3.Core.UserControls;
using Infragistics.WebUI.UltraWebGrid;

namespace Aurigo.Brix.Platform.BusinessLayer.AbstractModels
{
    public abstract class CustomizePageModel
    {
        private static readonly Dictionary<string, Type> contextDictionary;

        static CustomizePageModel()
        {
            if (contextDictionary == null)
                contextDictionary = ModelHelper.GetModelObjects(typeof (CustomizePageModel));
        }

        public static CustomizePageModel GetInstance(string context)
        {
            if (!contextDictionary.ContainsKey(context) || contextDictionary[context] == null)
                return null;
            return (CustomizePageModel)Activator.CreateInstance(contextDictionary[context]);
        }

        public static Dictionary<string, Type> GetModelObjects()
        {
            return contextDictionary;
        }

        public abstract void CustomizePage(Page page, ToolBar toolBar, string mode, string callerContext,
            object additionalParameters = null);

        public abstract void CustomizePageOnPreInit(Page page, ToolBar toolBar, string mode, string callerContext,
            object additionalParameters = null);

        public abstract void CustomizePageOnInit(Page page, ToolBar toolBar, string mode, string callerContext,
            object additionalParameters = null);

        public abstract void CustomizePageOnPreRender(Page page, ToolBar toolBar, string mode, string callerContext,
            object additionalParameters = null);

        public abstract void CustomizePageOnPageLoad(Page page, ToolBar toolBar, string mode, string callerContext,
            object additionalParameters = null);

        public abstract void CustomizePageAfterCustomizeToolBar(Page page, ToolBar toolBar, string mode,
            string callerContext, object additionalParameters = null);

        public abstract void CustomizeGrid(Page page, UltraWebGrid grid, string mode, string callerContext,
            object additionalParameters = null);

        public abstract Dictionary<string, object> GetData(Page page, string mode, string callerContext,
            object additionalParameters = null);
    }
}

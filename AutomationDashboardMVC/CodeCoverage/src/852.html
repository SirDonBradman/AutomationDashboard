using System;
using System.Collections.Generic;
using System.Web;
using Aurigo.AMP3.Common;

namespace Aurigo.AMP3.Core.AbstractModels
{
    public abstract class ItemTotalModel : BrixModel
    {
        private static readonly Dictionary<string, Type> contextDictionary;
        public string AdditionalInfo;

        static ItemTotalModel()
        {
            if (contextDictionary == null)
            {
                contextDictionary = ModelHelper.GetModelObjects(typeof (ItemTotalModel));
            }
        }

        public virtual bool IsPriceRelatedFieldsHidden
        {
            get { return false; }
        }

        public abstract string Header { get; }

        public static ItemTotalModel GetInstance(HttpRequest request, HttpResponse response, string additionalInfo)
        {
            string context = request["Context"];

            if (!contextDictionary.ContainsKey(context)
                || contextDictionary[context] == null)
            {
                return null;
            }

            var brixItemTotalModel = (ItemTotalModel)Activator.CreateInstance(contextDictionary[context]);

            brixItemTotalModel.Request = request;
            brixItemTotalModel.Response = response;
            brixItemTotalModel.AdditionalInfo = additionalInfo;

            return brixItemTotalModel;
        }

        public abstract bool GetApprovalStatus(int projectEstimateID, string moduleID);
    }
}

using System;
using System.Collections.Generic;
using System.Data;

namespace Aurigo.AMP3.Core.IntegrationConnector.DTO
{
    public enum ProjectTypeItems
    {
        Cost = 0,
        FixedPrice,
        Investment,
        TimeMaterial,
        Internal,
        Time
    };

    public enum ProjectStatus
    {
        Created,
        Inprocess,
        Finished,
        Estimated,
        Scheduled,
        Completed
    };

    public class ProjGroup
    {
        private readonly List<string> _projGroup;
        private readonly ProjectTypeItems _projType;

        public ProjGroup(ProjectTypeItems type, List<string> groups)
        {
            _projGroup = groups;
            _projType = type;
        }

        public ProjGroup(ProjectTypeItems type, string[] groups)
        {
            _projGroup.AddRange(groups);
            _projType = type;
        }

        public List<string> Group
        {
            get { return new List<string>(_projGroup.ToArray()); }
        }

        public ProjectTypeItems Type
        {
            get { return _projType; }
        }
    }

    public class Project : EISDTObjectBase
    {
        private readonly ProjGroup _projGroup;
        private readonly string _projID;

        //Constructor for the project class
        public Project(string projectId, ProjGroup projectGroup)
        {
            type = EISObjectType.Project;
            keyAttribute = "ProjId";
            checkExistMethod = "exist";
            _projGroup = projectGroup;
            _projID = projectId;
        }

        //Exposing properties for AX Project
        public string ProjectID
        {
            get { return _projID; }
        }

        public ProjGroup ProjectGroup
        {
            get { return _projGroup; }
        }

        public string Name { get; set; }

        public string ProjectType { get; set; }

        public string CreatedDate { get; set; }

        public string StartDate { get; set; }

        public string EndDate { get; set; }

        public ProjectStatus Status { get; set; }

        //Implemetation of Base class
        public override EISObjectState ObjectState
        {
            get { return state; }
        }

        public override EISObjectType ObjectType
        {
            get { return type; }
        }

        public override string KeyAttribute
        {
            get { return keyAttribute; }
        }

        public override string CheckExistMethodName
        {
            get { return checkExistMethod; }
        }

        public override EISOperationStatus LoadObject(DataRow dataRow)
        {
            throw new NotImplementedException();
        }

        public override string[] GetDefaultAttributeList(EISDTObjectBase dtObject)
        {
            return new[] { "ProjId", "Name", "Created", "EISOperationStatus" };
        }

        #region Nested type: MandatoryFields

        public class MandatoryFields
        {
            public const string Code = "ProjID";
            public const string Name = "Name";
            public const string GroupId = "ProjGroupID";
            public const string Type = "Type";
            public const string InvoiceProject = "ProjInvoiceProjId";
            public const string ParentId = "ParentId";
            public const string Format = "Format";
            public const string Status = "Status";
            public const string CustAccount = "CustAccount";
            public const string InventLocId = "InventLocationId";
            public const string InventSiteId = "InventSiteId";
        }

        #endregion
    }

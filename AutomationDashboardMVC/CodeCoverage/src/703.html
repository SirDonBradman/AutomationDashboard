using System;
using System.Collections.Generic;

namespace Aurigo.AMP3.DocumentManagementDTO
{
    public class Document
    {
        //Methods

        ////constructor
        public Document()
        {
            DocName = string.Empty;
            DocSize = string.Empty;
            Status = false;
            DocDesc = string.Empty;
            AuthorName = string.Empty;
            CreationDate = null;
            UpdationDate = null;
            StorageId = string.Empty;
            XMLInfo = string.Empty;
            CheckedByName = string.Empty;
            IsTemplate = false;
            DocumentType = DocumentType.Document;
        }

        //Properties
        public int DocId { get; set; }

        public int ParentId { get; set; }

        public string DocName { get; set; }

        public string DocSize { get; set; }

        public int DocType { get; set; }

        public string DocTypeDesc { get; set; }

        public bool Status { get; set; }

        public string DocDesc { get; set; }

        public int Author { get; set; }

        public string AuthorName { get; set; }

        public DateTime? CreationDate { get; set; }

        public DateTime? UpdationDate { get; set; }

        public int FolderId { get; set; }

        public string StorageId { get; set; }

        public int VersionNo { get; set; }

        public string XMLInfo { get; set; }

        public int CheckedBy { get; set; }

        public string CheckedByName { get; set; }

        public List<Document> DTOList { get; set; }

        public string SrcModuleID { get; set; }

        public Folder FolderDetails { get; set; }

        public string SPName { get; set; }
        public bool IsTemplate { get; set; }
        public int UpdatedBy { get; set; }

        public string SrcModuleCode { get; set; }
        public string SrcID { get; set; }
        public string DisplayDocName { get; set; }
        public string ChildAssociationType { get; set; }
        public int ParentFolderID { get; set; }
        public string DocumentOperationType { get; set; } //used C- for Copy and T - for Transmit
        public DocumentType DocumentType { get; set; }
        public DateTime? SignLockedOn { get; set; }
        public string SignLockedBy { get; set; }
        public int SignedDocID { get; set; }
    }

    public class DocumentVersion
    {
        public string VersionLabel { get; set; }
        public DateTime Created { get; set; }
        public string CreatedBy { get; set; }
        public string CheckInComment { get; set; }
        public string Url { get; set; }
        public int Major { get; set; }
    }

    public enum DocumentType
    {
        Document,
        SignedDocument
    }
}

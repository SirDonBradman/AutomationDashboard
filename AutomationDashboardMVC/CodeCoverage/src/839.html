using System;
using System.Collections.Generic;
using Aurigo.AMP3.Common;

namespace Aurigo.AMP3.Core.AbstractModels
{
    public abstract class ChecklistAttachmentModel : BrixModel
    {
        #region virtual methods

        public virtual string ModuleSpecificQueryString()
        {
            return String.Empty;
        }

        #endregion

        private static readonly Dictionary<string, Type> contextDictionary;

        static ChecklistAttachmentModel()
        {
            if (contextDictionary == null)
            {
                contextDictionary = ModelHelper.GetModelObjects(typeof (ChecklistAttachmentModel));
            }
        }
    }
}

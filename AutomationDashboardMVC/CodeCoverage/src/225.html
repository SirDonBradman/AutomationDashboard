        public override DataDynamics.ActiveReports.ActiveReport GetActiveReport(Report repObject, DataDynamics.ActiveReports.ActiveReport rpt)
        {
            rpt.FetchData += new DataDynamics.ActiveReports.ActiveReport.FetchEventHandler(rpt_FetchData);
            DataDynamics.ActiveReports.ActiveReport report = base.GetActiveReport(repObject, rpt);
            return report;
        }

        void rpt_FetchData(object sender, DataDynamics.ActiveReports.ActiveReport.FetchEventArgs eArgs)
        {
            DataDynamics.ActiveReports.ActiveReport report = (DataDynamics.ActiveReports.ActiveReport)sender;
            List<string> components = ModuleManager.Instance.GetModuleComponenets(Constants.MODID_CONTMGT);
            if (!components.Contains("MaterialOnHand"))// For Hiding MOH Section
            {
                report.Sections["GrpMOH"].Visible = false;
            }
            if (!HasAdditions)
                report.Sections["grpAdditions"].Visible = false;
            if (!HasDeductions)
                report.Sections["grpDeductionDetails"].Visible = false;

        }

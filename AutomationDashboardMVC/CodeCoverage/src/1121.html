using Aurigo.Brix.Platform.BusinessLayer.AbstractModels;
using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using System.Web.UI;
using Aurigo.AMP3.Core.AbstractModels;
using Aurigo.AMP3.Common;

namespace Aurigo.Brix.Platform.BusinessLayer.ConcreteModels.Favorite
{
    public class FavoriteListManagerModel: XMLListManagerModel
    {
        public override void AfterDelete(ListModel model, string selectedIds, XmlFormArgs e)
        {
            (HttpContext.Current.CurrentHandler as Page).ClientScript.RegisterStartupScript(
                HttpContext.Current.CurrentHandler.GetType(),"FavDeletePopup",
                "parent.location.reload();", true);
            base.AfterDelete(model, selectedIds, e);
        }
    }
}

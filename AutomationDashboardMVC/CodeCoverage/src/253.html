using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using Aurigo.Brix.Platform.BusinessLayer.BL;
using Aurigo.AMP3.Common;
using Aurigo.AMP3.Core.UserControls;
using Aurigo.AMP3.Resources.TerminologyResources;

namespace Aurigo.Brix.Construction.ContractManager.PayEstimatesUI.ConcreateModels
{
    [ContextElement(Name = "Contract List")]
    public class ContractEstimateContextMenuImplementation : ContextElementCallBack
    {
        #region ContextElementCallBack Members

        public List<BrixContextMenu> GetActionList()
        {
            var lst = new List<BrixContextMenu>();
            var item = new BrixContextMenu();
            item.Name = item.Description = LocalizationManager.GetString("LOC_PAY_ESTIMATE");
            item.GridParameters = "ContractID";
            item.IconImage = "IcnPage.gif";
            lst.Add(item);
            return lst;
        }

        public void DoCallBack(string Name, CIDictionary Parameters)
        {
            if(Name.ToUpper()==LocalizationManager.GetString("LOC_PAY_ESTIMATE").ToUpper())
                HttpContext.Current.Response.Redirect(
                        string.Format(
                            @"~/Common/BrixListPage.aspx?Context=PROCMGT&PID={0}&ContractID={1}&ParentID={1}&POType=CONTMGT&POTypeInstanceID={1}",
                            HttpContext.Current.Request["PID"], Parameters["ContractID"]), true);           
        }

        #endregion
    }
}

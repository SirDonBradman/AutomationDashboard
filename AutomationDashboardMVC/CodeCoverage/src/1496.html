using System;
using System.IO;
using System.Net;
using System.Web;

namespace Aurigo.Common.Utility
{
    public static class SMSManager
    {
        public static string SendSMS(string toMobileNumbers, string message)
        {
            string conString =
                "http://s2.freesmsapi.com/messages/send?skey=93f6f95be9844c76894f0a3875dbbf79&message={0}&senderid=Brix6.0&recipient={1}";
            try
            {
                WebRequest myRequest =
                    WebRequest.Create(string.Format(conString, HttpUtility.UrlEncode(message), toMobileNumbers));
                myRequest.Credentials = CredentialCache.DefaultCredentials;
                return new StreamReader(myRequest.GetResponse().GetResponseStream()).ReadLine();
            }
            catch (Exception ex)
            {
                return ex.Message;
            }
        }
    }
}

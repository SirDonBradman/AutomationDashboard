        public override string UploadableReportReturnPath()

            string PID = Request.QueryString.Get("PID");
            string FID = Request.QueryString["FID"];
            string IID = Request.QueryString["IID"];
            string MID = Request.QueryString["MID"];
            return string.Format("~/Modules/DOCMGMT/Documents.aspx?FID={0}&PID={1}&IID={2}&MID={3}&nt=1",
                FID, PID, IID, MID);
        }

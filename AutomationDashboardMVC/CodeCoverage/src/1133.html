using System;
using System.Collections.Generic;
using System.Data;
using System.IO;
using System.Reflection;
using Aurigo.AMP3.DataAccess.Core;
using Aurigo.Brix.Platform.BusinessLayer.DataAccessHelper;
using Aurigo.Brix.Platform.CoreUtilities.Utility;
using Aurigo.Common.Utility;
using System.Web;

namespace Aurigo.Brix.Platform.BusinessLayer.BL
{
    public class AttachmentStreamManager : SingletonManagerBase<AttachmentStreamManager>
    {
        private AttachmentStreamComponent DAL { get; set; } = new AttachmentStreamComponent();

        public Guid GetNewGroupId_ForAttachements()
        {
            return Guid.NewGuid();
        }
        
        /// <summary>
        /// Save new file to DB and returns the id of the new record
        /// </summary>
        /// <param name="groupGuid"></param>
        /// <param name="actualFileName"></param>
        /// <param name="attachmentStream"></param>
        /// <returns>returns the id of the new record</returns>
        public long AddAttachment(Guid groupGuid, string actualFileName, Stream attachmentStream)
        {
            string sessionID = HttpContext.Current.Session.SessionID.ToString();

            return DAL.AddAttachment(groupGuid, actualFileName, sessionID, attachmentStream);
        }

        /// <summary>
        /// Remove records
        /// </summary>
        /// <param name="idOfNewRecord"></param>
        /// <param name="groupGuid"></param>
        public void RemoveAttachment(long idOfNewRecord, Guid groupGuid)
        {
            DAL.RemoveAttachment(idOfNewRecord, groupGuid);
        }

        /// <summary>
        /// Deletes all attachments with respect to the current group
        /// </summary>
        /// <param name="groupGuid"></param>
        public void Remove_All_Attachment(Guid groupGuid)
        {
            DAL.Remove_All_Attachment(groupGuid);
        }

        /// <summary>
        /// Get attachment content 
        /// </summary>
        /// <param name="idOfNewRecord"></param>
        /// <param name="groupGuid"></param>
        /// <returns></returns>
        public AttachmentStreamDto GetAttachmentFileContent(long idOfNewRecord, Guid groupGuid)
        {
            return DAL.GetAttachmentFileContent(idOfNewRecord, groupGuid); 
        }
    }
}

    [Serializable]
    public class DTOPE
    {
        public Decimal? AdditionsTotal;
        public List<PEAdjustment> Adjustments;
        public string Comments;
        public Decimal? ContractAdvRecoveryPercent;
        public int? ContractorID;
        public string ContractorType;

        public List<PEPrepayments> Prepayments;

        internal DateTime? RetentionReleaseDate;
        public DateTime? ToDt;
        public int? WOID;
        public string RetentionMode;
        public decimal RetentionPercent;
        public decimal RetReleaseAmt;

        public DTOPE()
        {
            IDs = new List<int>();
            Postings = new List<PEPostings>();
            MBooks = new List<PEMBooks>();
            Prepayments = new List<PEPrepayments>();
            MaterialDeductions = new List<PEMaterialDeduction>();
            Adjustments = new List<PEAdjustment>();
            ContractorType = string.Empty;
            ContractorID = null;
            IsRFI = false;
            IsFinal = false;
            RetentionReleaseDate = null;
            RetentionMode = "P";
        }

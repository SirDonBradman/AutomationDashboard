using Aurigo.Brix.Platform.BusinessLayer.AbstractModels;
using Aurigo.Brix.Platform.BusinessLayer.XMLForm;

namespace Aurigo.Brix.Platform.UI.Modules.FRMBLDR
{
    public class AddSet : AddProcessor
    {
        public AddSet(BrixFormModel model) : base(model)
        {
        }

        public override string ControlName => "set";

        public override string Add(Command cmd, out ControlContainer control)
        {
            var err = string.Empty;
            control = null;

            ControlContainer dest = _model.form.GetContainer(cmd.destinationControl);
            if (dest != null)
            {
                if (dest.GetType() == typeof(Section))
                {
                    var xx = MakeNode(ControlType.Set, ControlType.Set, null);
                    var yy = MakeNode(ControlType.TextBox, ControlType.TextBox, null);
                    xx.Controls.Add(yy);
                    dest.Controls.Add(xx);

                    control = xx;
                }
                else
                    err = "A set must be a child of a Section";
            }
            else
                err = "Could not find destination control";

            return err;
        }
    }
}

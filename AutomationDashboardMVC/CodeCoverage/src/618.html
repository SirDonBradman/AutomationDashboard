using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using Aurigo.Brix.Platform.CoreUtilities.Utility;
using System.Data;
using Aurigo.AMP3.Common;
using Aurigo.Common;
using Aurigo.Brix.Platform.BusinessLayer.DataResouces;
using Aurigo.Brix.Construction.ContractManager.QuantityPlanningUI.BL;

namespace Aurigo.Brix.Construction.PlanningManagement.WFActivities.Forecast_WF
{
    [CustomResourceType("Forms", "XQTYPLN")]
    public class WFAUpdateForecastOnReDraft : FormsCustomResource
    {
        public WFAUpdateForecastOnReDraft()
        {
            _Namespace = "Forms";
            _Path = "Forms.PLANING.WFactivities.WFAUpdateForecastOnReDraft";
            _Name = "Update on Re-Draft";
            _Desc = "Update on Re-Draft";
            _OutParameters = new[] { "IsSuccessful,System.Boolean", "Message,System.String" };
        }
        public override void ExecuteDerived()
        {
            //Update the internal status column to Draft
            try
            {
                int isSuccessful = QuantityPlanningManager.Instance.ApprovePlanning(InstanceId.ToInt32_2(), null, null, QuantityPlanningStatus.Draft.ToInt32_2());

                if (isSuccessful != 0)
                {
                    throw new Exception("Re-Draft unsuccessful.");
                }

                SetOutParam("IsSuccessful", true, "System.Boolean");
                SetOutParam("Message", string.Empty, "System.String");
            }
            catch (Exception ex)
            {
                SetOutParam("IsSuccessful", false, "System.Boolean");
                SetOutParam("Message", ex.Message, "System.String");
                throw new Exception(ex.Message);
            }
        }
    }
}

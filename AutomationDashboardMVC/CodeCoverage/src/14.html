using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using Aurigo.AMP3.Common;
using Aurigo.AMP3.ModuleManagementBL;
using Aurigo.Brix.Platform.BusinessLayer.AbstractModels;

namespace Aurigo.Brix.Construction.BudgetManager.BusinessLayer.ConcreateModels
{
    public class BudgetManagementConfigurationsFormManager : XMLFormManagerModel
    {
        private Dictionary<string, string> _modules = null;

        private Dictionary<string, string> Modules
        {
            get
            {
                if (_modules == null)
                    _modules = ModuleManager.Instance.GetModuleList();

                return _modules;
            }
        }
        public override void OnPreInit(BrixFormModel model, XmlFormArgs e)
        {
            if (Modules.ContainsKey(Constants.MODID_PLANING))
            {
                model.form.GetControl("StartMonth").ReadOnly = true;
                model.form.GetControl("EndMonth").ReadOnly = true;
            }
        }
    }
}

using System;
using System.Collections.Generic;
using System.Data;
using Aurigo.AMP3.Common;

namespace Aurigo.AMP3.WORKORDDTO
{
    [Serializable]
    public class WODTO
    {
        public int ContractID;
        public int? ContractorID;
        public string ContractorType;
        public int? CreatedBy;
        public DateTime CreatedDate;
        public string Description;
        public string EstimatedValue;
        public string FundID;
        public DateTime IssuedDate;
        public string Notes;
        public int? ParentWOID;
        public int? Status;
        public DateTime WODate;
        public List<WODeletedItem> WODeletedItems;
        public List<WOFund> WOFunds;
        public int? WOID;
        public List<WOItem> WOItems;
        public string WONumber;

        public WODTO()
        {
            WOItems = new List<WOItem>();
            CreatedDate = DateTime.UtcNow;
            ContractorID = null;
            ContractorType = string.Empty;
        }
    }

    public class WorkOrderDTO
    {

        public DateTime ApprovedOn;
        public string ClosedBy;
        public int ClosedByID;
        public DateTime ClosedOn;
        public int ContractID;
        public string Contractor;
        public int ContractorID;
        public string CreatedBy;
        public int CreatedByID;
        public DateTime CreatedOn;
        public string Description;
        public DateTime EndDate;
        public bool HasAttachments;
        public string Insurance;
        public int? IsNew;
        public bool IsVersion;
        public string IssuedBy;
        public int IssuedByID;
        public DateTime IssuedOn;
        public string LicenseCity;
        public string LicenseState;
        public string Mode;

        public string ModifiedBy;
        public int ModifiedByID;
        public DateTime ModifiedOn;
        public string Notes;
        public int? ParentID;
        public string ScopeOfWork;
        public DateTime StartDate;
        public int Status;
        public string TaxPayerID;
        public decimal TotalAmount;
        public int? ValidateWOQty;
        public string VersionNo;
        public bool WithEquipment;
        public bool WithLabour;
        public bool WithMaterial;
        public DataSet WorkOrderActivities = new BrixDataSet();
        public DateTime WorkOrderDate;
        public int? WorkOrderID;
        public DataSet WorkOrderItems = new BrixDataSet();
        public string WorkOrderNo;
        public string SystemWONo;
        public DataSet WorkOrderQuotations;
        public string WorkOrderRequestedBy;
        public string WorkOrderType;
        public string xmlGeneralDetails;
        public string xmlPrepaymentItemList;
        public string xmlRetentionItemList;
        public string ExtReference;
        public IssueMode IssueMode;
    }

    [Serializable]
    public struct WOItem
    {
        public Decimal? Amount;
        public int? GroupID;
        public int? ItemID;
        public Decimal? Quantity;
    }

    [Serializable]
    public struct WOFund
    {
        public string FundID;
    }

    [Serializable]
    public struct WODeletedItem
    {
        public int GroupID;
        public int ItemID;
    }

    public enum IssueMode
    {
        WithQuotation,
        Direct
    }
}

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Web.UI.WebControls;
using Infragistics.WebUI.WebSchedule;
using Aurigo.Brix.Platform.CoreUtilities.Utility;
using Aurigo.AMP3.Common;
using Infragistics.WebUI.WebDataInput;
using Aurigo.Brix.Platform.BusinessLayer.CustomControls;
using System.Web.UI;

namespace Aurigo.Brix.Platform.BusinessLayer.XmlForm_Framework
{
    public class XMLListBox : ListBox, IControl
    {
        public bool RenderAsLabel { get; set; }
        protected override void Render(System.Web.UI.HtmlTextWriter writer)
        {
            if (RenderAsLabel)
            {
                string text = string.Empty;
                string value = string.Empty;
                foreach (ListItem listitem in this.Items)
                {
                    if (listitem.Selected)
                    {
                        text += listitem.Value == null && listitem.Text == null ? "" : listitem.Text + ",";
                        value += listitem.Value == null && listitem.Text == null ? "" : listitem.Value + ",";
                    }
                }
                Helper.RenderAsLabel(this, writer, text.TrimEnd(','), Width, 25, attributes: Style.Value);
                Helper.RenderAsLabel(this, writer, value.TrimEnd(','), Width, 25, true, attributes: Style.Value);
                base.Style.Add("display", "none");
                base.Render(writer);
            }
            else
                base.Render(writer);
        }
    }
}

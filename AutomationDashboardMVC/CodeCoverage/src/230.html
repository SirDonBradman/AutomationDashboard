using System;
using System.Data;
using Aurigo.AMP3.Common;
using Aurigo.AMP3.CONTMGTPEBL;
using Aurigo.Brix.Core.WebUiHelper;
using Aurigo.Brix.Platform.CoreUtilities.Utility;

namespace Aurigo.AMP3.CONTMGTPEUI
{
    public partial class NewRelease : BrixPageBase
    {
        protected void Page_Load(object sender, EventArgs e)
        {
            if (String.IsNullOrEmpty(Request[Constants.QRY_PROJECTID]) || String.IsNullOrEmpty(Request["ContractID"]))
                Response.Redirect("~/Common/BrixListPage.aspx?context=CONTMGT&PID=" + Request[Constants.QRY_PROJECTID],
                                  true);

            if (!String.IsNullOrEmpty(Request["Mode"]) && (Request["Mode"] == "View" || Request["Mode"] == "Edit") &&
                String.IsNullOrEmpty(Request["PEID"]))
                Response.Redirect("~/Common/BrixListPage.aspx?context=CONTMGT&PID=" + Request[Constants.QRY_PROJECTID],
                                  true);

            if (!IsPostBack)
            {
                DataRow dr;
                DataTable dt;
                if (String.IsNullOrEmpty(Request["Mode"]))
                {
                    dt = BL.Instance.GetPERet(Request["ContractID"].ToInt32_2(), 1, 1, null, null);
                    if (dt.Rows.Count != 1)
                        Response.Redirect(
                            "~/Common/BrixListPage.aspx?context=CONTMGT&PID=" + Request[Constants.QRY_PROJECTID], true);
                    dr = dt.Rows[0];
                }
            }

            SetLocalizedStrings();
        }

        private void SetLocalizedStrings()
        {
            if (String.IsNullOrEmpty(Request["Mode"]))
            {
                tdRetHdr.InnerText = "New Retainage Release";
            }
            else if (Request["Mode"] == "Edit")
            {
                tdRetHdr.InnerText = "Edit Retainage Release";
            }
            else if (Request["Mode"] == "View")
            {
                tdRetHdr.InnerText = "View Retainage Release";
            }
        }
    }
}

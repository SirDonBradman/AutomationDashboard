using Aurigo.AMP3.Common;
using System;
using System.Data.SqlClient;
using System.Data;
using System.Collections.Generic;

namespace Aurigo.AMP3.ContractManagementUI
{
    [ListModelContext(Name = _CONTEXT, Table = "CORITEMGlobalSubItemGroup")]
    public class GlobalSubItemGroupListModel : Core.AbstractModels.GlobalSubItemGroupListModel
    {
        private const string _CONTEXT = "GLOBALGROUPCONTMGT";

        public override string ModuleId
        {
            get { return Constants.MODID_CONTMGT; }
        }

        public override string CONTEXT
        {
            get { return _CONTEXT; }
        }      

        public override int HandleDelete()
        {
            try
            {
                return base.HandleDelete();
            }
            catch (SqlException ex)
            {
                if (ex.Message.Contains("REFERENCE constraint"))
                    throw new Exception("Cannot delete this global group because it is being used in a Sub Item");
                else
                    throw new Exception(ex.Message); 
            }

        }
    }
}

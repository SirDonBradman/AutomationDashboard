using Aurigo.AMP3.Common;
using Aurigo.AMP3.Core.UserControls;
using Aurigo.Brix.Core.WebUiHelper;
using Aurigo.Brix.Platform.BusinessLayer.AbstractModels;
using Aurigo.Brix.Platform.CoreUtilities.Utility;
using Infragistics.WebUI.UltraWebGrid;
using System;
using System.Collections.Generic;
using System.Data;

namespace Aurigo.AMP3.DWR.Models
{
    [ContextAttribute(Name = "DWRCUSTOMSEARCH")]
    public class DWRCustomSearch : CustomizePageModel
    {
        #region Public Methods

        public override void CustomizeGrid(System.Web.UI.Page page, UltraWebGrid grid, string mode, string callerContext, object additionalParameters = null)
        {
            DataTable dt = (page as BrixPageBase).columnInfo;
            if (dt != null)
            {
                dt.Select("Key='ContractorName'").ForEach(x => { dt.Rows.Remove(x); });
                dt.AcceptChanges();
            }
            if (callerContext.Equals("CustomSearch", StringComparison.CurrentCultureIgnoreCase))
            {
                if (grid.Columns.FromKey("Contractor") != null)
                {
                    grid.Columns.FromKey("Contractor").Hidden = true;
                }
                if (grid.Columns.FromKey("ContractorName") != null)
                {
                    grid.Columns.FromKey("ContractorName").Hidden = false;
                    grid.Columns.FromKey("ContractorName").Header.Caption = "Contractor Name";
                }
            }
        }

        public override void CustomizePage(System.Web.UI.Page page, ToolBar toolBar, string mode, string callerContext, object additionalParameters = null)
        {
        }

        public override void CustomizePageAfterCustomizeToolBar(System.Web.UI.Page page, ToolBar toolBar, string mode, string callerContext, object additionalParameters = null)
        {
        }

        public override void CustomizePageOnInit(System.Web.UI.Page page, ToolBar toolBar, string mode, string callerContext, object additionalParameters = null)
        {
        }

        public override void CustomizePageOnPageLoad(System.Web.UI.Page page, ToolBar toolBar, string mode, string callerContext, object additionalParameters = null)
        {
        }

        public override void CustomizePageOnPreInit(System.Web.UI.Page page, ToolBar toolBar, string mode, string callerContext, object additionalParameters = null)
        {
        }

        public override void CustomizePageOnPreRender(System.Web.UI.Page page, ToolBar toolBar, string mode, string callerContext, object additionalParameters = null)
        {
        }

        public override Dictionary<string, object> GetData(System.Web.UI.Page page, string mode, string callerContext, object additionalParameters = null)
        {
            throw new NotImplementedException();
        }

        #endregion Public Methods
    }
}

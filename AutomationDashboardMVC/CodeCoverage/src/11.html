using System.Web.UI;
using Aurigo.AMP3.Common;
using Aurigo.Brix.Construction.BudgetManager.BusinessLayer.BL;
using Aurigo.Brix.Platform.BusinessLayer.AbstractModels;
using Aurigo.Brix.Platform.BusinessLayer.DTO;

namespace Aurigo.Brix.Construction.BudgetManager.BusinessLayer.ConcreateModels
{
    [ContextAttribute(Name = "BDGPHAS")]
    public class AfterProjectPhaseSaveModel : AfterSaveRecordModel
    {
        public override void SaveDetails(object recordDetails, Page page, string mode, string callerContext, object additionalParameters = null)
        {
            if (callerContext == "PROJECT_PHASE")
            {
                ProjectPhaseAfterSave record = recordDetails as ProjectPhaseAfterSave;
                if (record != null)
                    BudgetManagementManager.Instance.InsertContainerAfterProjectPhaseSave(record.ProjectID, record.CreatedBy);
            }
        }
    }
}

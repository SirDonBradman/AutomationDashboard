using System.Data;
using Aurigo.AMP3.Core.IntegrationConnector;
using Aurigo.Brix.Platform.BusinessLayer.AbstractModels;
using Aurigo.Brix.Platform.CoreUtilities.Utility;

namespace Aurigo.Brix.Construction.Estimator.BusinessLayer.ConcreateModels.Estimate
{
    public class EstimatorEisCompanyDetails : ModuleDetails
    {
        public override string GetMappedProjectInEis(string additionalInfo, string projectId, string parentInstanceID)
        {
            string mappedCompany = additionalInfo;
            DataTable dtProjectMap = IntegrationConnectorManager.Instance.GetERPObjectMapInfo(projectId,
                                                                                              AMP3Object.PROJECT, null,
                                                                                              AMP3Object.UNKNOWN,
                                                                                              RegisteredEIS.AX);

            if (dtProjectMap.Rows.Count > 0)
                mappedCompany = dtProjectMap.Rows[0]["ERPCompany"].ToString2();

            return mappedCompany;
        }
    }
}

        //[HttpGet]
        public HttpResponseMessage GetServerDateAsText()
        {
            string serverDateTimeString = MWDateTimeHelper.MWNow.ToString_WithFormatDateTime();

            var response = new HttpResponseMessage(HttpStatusCode.OK);
            response.Content = new StringContent(serverDateTimeString, System.Text.Encoding.UTF8, "text/plain");
            return response;

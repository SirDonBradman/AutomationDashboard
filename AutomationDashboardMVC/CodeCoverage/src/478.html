using System;
using System.Web;
using Aurigo.AMP3.Common;
using Aurigo.AMP3.Core.AbstractModels;

namespace Aurigo.Brix.Construction.FundManagement
{
    public abstract class FundManagementListModel : ListModel
    {
        public override string ModuleId
        {
            get { return Constants.MODID_FNDMGMT; }
        }

        public override string QueryStringName
        {
            get { return "ID"; }
        }

        public override void SetUIDetails()
        {
            ModifyColumnProperties("ID", true, null, null, null, true);
            ModifyColumnProperties("RowNum", true, null, null, null, true);
            DisplayEdit = true;
            displayOtherDropMenu = false;
        }

        public override int GetPageCount(string filter)
        {
            return 0;
        }

        public void HandleNewEvent(object sender, EventArgs e)
        {
            HandleNew();
        }

        public void HandleEditEvent(object sender, EventArgs e)
        {
            HandleEdit();
        }

        public void HandleViewEvent(object sender, EventArgs e)
        {
            HandleView();
        }

        public void HandleDeleteEvent(object sender, EventArgs e)
        {
            HandleDelete();
            Response.Redirect(HttpContext.Current.Request.Url.ToString());
        }
    }
}

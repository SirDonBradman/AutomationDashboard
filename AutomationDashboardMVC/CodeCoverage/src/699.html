using System.Collections.Generic;
using System.Data;
using Aurigo.AMP3.DataAccess.Core;
using Aurigo.AMP3.LibraryDAC;
using Aurigo.Common.Utility;

namespace Aurigo.AMP3.LibraryBL
{
    public class SignatureTitlesManager : SingletonManagerBase<SignatureTitlesManager>
    {
        private SignatureTitlesManager()
        {
        }

        public DataSet GetSignatureTitles(string moduleId)
        {
            return LibraryComponent.Instance.GetSignatureTitles(moduleId);
        }

        public DataSet GetSignatureTitles(string moduleId, int contractId)
        {
            return LibraryComponent.Instance.GetSignatureTitles(moduleId, contractId);
        }

        public int SaveSignatureTitles(string xml, string moduleId)
        {
            return LibraryComponent.Instance.SaveSignatureTitles(xml, moduleId);
        }

        public int SaveContractSignatureTitles(string xml, int contractId)
        {
            return LibraryComponent.Instance.SaveContractSignatureTitles(xml, contractId);
        }

        public DataSet GetSignatureModules()
        {
            return LibraryComponent.Instance.GetSignatureModules();
        }
    }
}
